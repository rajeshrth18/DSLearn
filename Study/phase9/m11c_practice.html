Easy Problems (4)

#1 Find if Path Exists in Graph Easy

Given n nodes (0 to n-1) and a list of bidirectional edges, determine if there is a valid path from source to destination.

Example: n=3, edges=[[0,1],[1,2],[2,0]], source=0, dest=2 → true

Key Insight: Union-Find is perfect here — union all edges, then check if source and destination share the same root.
public bool ValidPath(int n, int[][] edges, int src, int dst)
{
    int[] parent = Enumerable.Range(0, n).ToArray();

    int Find(int x) => parent[x] == x ? x : (parent[x] = Find(parent[x]));
    void Union(int a, int b) { parent[Find(a)] = Find(b); }

    foreach (var e in edges) Union(e[0], e[1]);
    return Find(src) == Find(dst);
}
// Time: O(E α(V)) ≈ O(E), Space: O(V)
#2 Is Valid Anagram Easy

Given two strings s and t, return true if they are anagrams of each other.

Example: s="anagram", t="nagaram" → true

public bool IsAnagram(string s, string t)
{
    if (s.Length != t.Length) return false;
    int[] count = new int[26];
    for (int i = 0; i < s.Length; i++) { count[s[i]-'a']++; count[t[i]-'a']--; }
    return count.All(c => c == 0);
}
// Time: O(n), Space: O(1) — only 26 letters
#3 First Unique Character in a String Easy

Find the index of the first non-repeating character in a string. Return -1 if none exists.

Example: "leetcode" → 0 (the 'l')

public int FirstUniqChar(string s)
{
    int[] freq = new int[26];
    foreach (char c in s) freq[c - 'a']++;
    for (int i = 0; i < s.Length; i++)
        if (freq[s[i] - 'a'] == 1) return i;
    return -1;
}
// Time: O(n), Space: O(1)
#4 Course Schedule (Can Finish?) Easy

Given numCourses and prerequisites pairs, determine if you can finish all courses (i.e., no cycle exists).

Example: numCourses=2, prerequisites=[[1,0]] → true

Key Insight: Detect cycle in a directed graph using Kahn's topological sort. If all nodes can be processed (no remaining nodes with in-degree > 0), no cycle exists.
public bool CanFinish(int n, int[][] prereqs)
{
    var adj   = new List<int>[n].Select(_ => new List<int>()).ToArray();
    int[] ind = new int[n];
    foreach (var p in prereqs) { adj[p[1]].Add(p[0]); ind[p[0]]++; }

    var q = new Queue<int>();
    for (int i = 0; i < n; i++) if (ind[i] == 0) q.Enqueue(i);
    int done = 0;
    while (q.Count > 0)
    {
        int u = q.Dequeue(); done++;
        foreach (int v in adj[u]) if (--ind[v] == 0) q.Enqueue(v);
    }
    return done == n;
}
// Time: O(V+E), Space: O(V+E)

Medium Problems (7)

#5 Number of Islands (Union-Find) Medium

Count the number of islands in a 2D grid of '1' (land) and '0' (water). Solve using Union-Find.

Approach: Each land cell starts as its own island. Union adjacent land cells. Count remaining distinct roots.
public int NumIslands(char[][] grid)
{
    int m = grid.Length, n = grid[0].Length;
    int[] parent = Enumerable.Range(0, m * n).ToArray();
    int islands = 0;

    int Find(int x) => parent[x] == x ? x : (parent[x] = Find(parent[x]));
    void Union(int a, int b)
    {
        int ra = Find(a), rb = Find(b);
        if (ra != rb) { parent[ra] = rb; islands--; }
    }

    for (int r = 0; r < m; r++)
    for (int c = 0; c < n; c++)
        if (grid[r][c] == '1')
        {
            islands++;
            if (r > 0 && grid[r-1][c] == '1') Union(r*n+c, (r-1)*n+c);
            if (c > 0 && grid[r][c-1] == '1') Union(r*n+c, r*n+c-1);
        }
    return islands;
}
// Time: O(m*n*α(m*n)) ≈ O(m*n), Space: O(m*n)
#6 Minimum Spanning Tree — Min Cost to Connect All Points Medium

Given points[i] = [xi, yi], find the minimum cost to connect all points using Manhattan distance edges.

Prim's MST: Build the complete graph lazily — from the current MST, always pick the nearest unvisited point.
public int MinCostConnectPoints(int[][] pts)
{
    int n = pts.Length;
    bool[] inMST = new bool[n];
    int[] minCost = Enumerable.Repeat(int.MaxValue, n).ToArray();
    minCost[0] = 0;
    int total = 0;

    for (int iter = 0; iter < n; iter++)
    {
        // Pick the node with minimum cost not yet in MST
        int u = -1;
        for (int i = 0; i < n; i++)
            if (!inMST[i] && (u == -1 || minCost[i] < minCost[u])) u = i;

        inMST[u] = true;
        total += minCost[u];

        // Update costs to all other nodes
        for (int v = 0; v < n; v++)
        {
            if (inMST[v]) continue;
            int dist = Math.Abs(pts[u][0]-pts[v][0]) + Math.Abs(pts[u][1]-pts[v][1]);
            minCost[v] = Math.Min(minCost[v], dist);
        }
    }
    return total;
}
// Time: O(n²) — acceptable for dense graphs. Use PriorityQueue for O(n² log n).
#7 Network Delay Time (Dijkstra) Medium

There are n nodes (1..n) and times[i]=[u,v,w] (directed edge). Send signal from node k. Return the minimum time until all nodes receive it, or -1 if not possible.

public int NetworkDelayTime(int[][] times, int n, int k)
{
    var adj = new List<(int to, int w)>[n + 1];
    for (int i = 0; i <= n; i++) adj[i] = new();
    foreach (var t in times) adj[t[0]].Add((t[1], t[2]));

    int[] dist = Enumerable.Repeat(int.MaxValue, n + 1).ToArray();
    dist[k] = 0;
    var pq = new PriorityQueue<int, int>();
    pq.Enqueue(k, 0);

    while (pq.Count > 0)
    {
        pq.TryDequeue(out int u, out int d);
        if (d > dist[u]) continue;
        foreach (var (v, w) in adj[u])
            if (dist[u] + w < dist[v])
            {
                dist[v] = dist[u] + w;
                pq.Enqueue(v, dist[v]);
            }
    }
    int maxDist = dist[1..(n + 1)].Max();
    return maxDist == int.MaxValue ? -1 : maxDist;
}
// Time: O((V+E) log V), Space: O(V+E)
#8 Longest Palindromic Substring Medium

Given a string s, return the longest palindromic substring.

Example: "babad" → "bab"

Expand-around-center: For each center (single char or gap between chars), expand outward while characters match. O(n²) time, O(1) space — simpler than Manacher's for interviews.
public string LongestPalindrome(string s)
{
    int start = 0, maxLen = 1;

    void Expand(int l, int r)
    {
        while (l >= 0 && r < s.Length && s[l] == s[r]) { l--; r++; }
        if (r - l - 1 > maxLen) { maxLen = r - l - 1; start = l + 1; }
    }

    for (int i = 0; i < s.Length; i++)
    {
        Expand(i, i);     // Odd-length palindrome
        Expand(i, i + 1); // Even-length palindrome
    }
    return s.Substring(start, maxLen);
}
// Time: O(n²), Space: O(1)
#9 Group Anagrams Medium

Given an array of strings, group the anagrams together.

Example: ["eat","tea","tan","ate","nat","bat"] → [["bat"],["nat","tan"],["ate","eat","tea"]]

Key: Anagrams share the same sorted string. Use sorted string as dictionary key.
public IList<IList<string>> GroupAnagrams(string[] strs)
{
    var map = new Dictionary<string, List<string>>();
    foreach (var s in strs)
    {
        char[] arr = s.ToCharArray();
        Array.Sort(arr);
        string key = new string(arr);
        if (!map.ContainsKey(key)) map[key] = new();
        map[key].Add(s);
    }
    return new List<IList<string>>(map.Values);
}
// Time: O(n * m log m) where m = average string length
#10 Find All Anagrams in a String Medium

Given strings s and p, find all start indices of p's anagrams in s.

Example: s="cbaebabacd", p="abc" → [0,6]

Sliding Window: Maintain a fixed-size window of size p.Length and compare frequency arrays. Update incrementally in O(1) per step.
public IList<int> FindAnagrams(string s, string p)
{
    if (s.Length < p.Length) return new List<int>();

    int[] pCount = new int[26], wCount = new int[26];
    for (int i = 0; i < p.Length; i++) { pCount[p[i]-'a']++; wCount[s[i]-'a']++; }

    var result = new List<int>();
    if (pCount.SequenceEqual(wCount)) result.Add(0);

    for (int i = p.Length; i < s.Length; i++)
    {
        wCount[s[i] - 'a']++;
        wCount[s[i - p.Length] - 'a']--;
        if (pCount.SequenceEqual(wCount)) result.Add(i - p.Length + 1);
    }
    return result;
}
// Time: O(n * 26) = O(n), Space: O(1)
#11 Cheapest Flights Within K Stops Medium

Find the cheapest price from src to dst with at most k stops.

Bellman-Ford with iteration limit: Run exactly k+1 relaxation rounds. Use a snapshot (temp array) to prevent using edges from the same round.
public int FindCheapestPrice(int n, int[][] flights, int src, int dst, int k)
{
    int[] dist = Enumerable.Repeat(int.MaxValue, n).ToArray();
    dist[src] = 0;

    for (int i = 0; i <= k; i++)
    {
        int[] temp = (int[])dist.Clone();
        foreach (var f in flights)
        {
            int u = f[0], v = f[1], w = f[2];
            if (dist[u] != int.MaxValue && dist[u] + w < temp[v])
                temp[v] = dist[u] + w;
        }
        dist = temp;
    }
    return dist[dst] == int.MaxValue ? -1 : dist[dst];
}
// Time: O(k * E), Space: O(V)

Hard Problems (4)

#12 Critical Connections (Bridges in Graph) Hard

Find all critical connections (bridges) in a network. A bridge is an edge whose removal increases the number of connected components.

Tarjan's Bridge Algorithm: Use DFS with discovery time and low-link value. An edge (u,v) is a bridge if low[v] > disc[u] (v cannot reach u or its ancestors without going through this edge).
public IList<IList<int>> CriticalConnections(int n, IList<IList<int>> connections)
{
    var adj  = new List<int>[n].Select(_ => new List<int>()).ToArray();
    foreach (var c in connections) { adj[c[0]].Add(c[1]); adj[c[1]].Add(c[0]); }

    int[] disc = new int[n], low = new int[n];
    bool[] vis = new bool[n];
    var result = new List<IList<int>>();
    int timer = 0;

    void Dfs(int u, int parent)
    {
        vis[u] = true;
        disc[u] = low[u] = timer++;
        foreach (int v in adj[u])
        {
            if (!vis[v])
            {
                Dfs(v, u);
                low[u] = Math.Min(low[u], low[v]);
                if (low[v] > disc[u]) result.Add(new[] { u, v }); // Bridge!
            }
            else if (v != parent)
                low[u] = Math.Min(low[u], disc[v]);
        }
    }

    Dfs(0, -1);
    return result;
}
// Time: O(V+E), Space: O(V+E)
#13 Word Ladder (Shortest Transformation Sequence) Hard

Transform beginWord to endWord by changing one letter at a time, using only words in wordList. Find the minimum number of steps.

BFS on an Implicit Graph: Nodes = words, edges = words that differ by exactly one character. BFS gives shortest path. Use wildcard patterns to efficiently find neighbors.
public int LadderLength(string begin, string end, IList<string> wordList)
{
    var wordSet = new HashSet<string>(wordList);
    if (!wordSet.Contains(end)) return 0;

    // Build adjacency via wildcard patterns: "hot" → "_ot","h_t","ho_"
    var pattern = new Dictionary<string, List<string>>();
    foreach (var w in wordSet)
        for (int i = 0; i < w.Length; i++)
        {
            string key = w.Substring(0,i) + "*" + w.Substring(i+1);
            if (!pattern.ContainsKey(key)) pattern[key] = new();
            pattern[key].Add(w);
        }

    var queue   = new Queue<string>();
    var visited = new HashSet<string>();
    queue.Enqueue(begin); visited.Add(begin);
    int level = 1;

    while (queue.Count > 0)
    {
        int sz = queue.Count;
        while (sz-- > 0)
        {
            var word = queue.Dequeue();
            if (word == end) return level;
            for (int i = 0; i < word.Length; i++)
            {
                string key = word.Substring(0,i) + "*" + word.Substring(i+1);
                foreach (var nb in pattern.GetValueOrDefault(key, new()))
                    if (visited.Add(nb)) queue.Enqueue(nb);
            }
        }
        level++;
    }
    return 0;
}
// Time: O(M² * N) where M=word length, N=wordList size; Space: O(M² * N)
#14 Minimum Window Substring (Revisited) Hard

Given strings s and t, find the minimum window in s that contains all characters of t.

Two Pointer + Frequency Map: Expand right pointer until window contains all of t. Then shrink left pointer as much as possible. Track how many characters are "satisfied" to avoid O(26) check each step.
public string MinWindow(string s, string t)
{
    var need  = new Dictionary<char, int>();
    foreach (char c in t) need[c] = need.GetValueOrDefault(c) + 1;

    var window   = new Dictionary<char, int>();
    int have = 0, required = need.Count;
    int l = 0, minLen = int.MaxValue, start = 0;

    for (int r = 0; r < s.Length; r++)
    {
        char rc = s[r];
        window[rc] = window.GetValueOrDefault(rc) + 1;
        if (need.ContainsKey(rc) && window[rc] == need[rc]) have++;

        while (have == required)
        {
            if (r - l + 1 < minLen) { minLen = r - l + 1; start = l; }
            char lc = s[l++];
            window[lc]--;
            if (need.ContainsKey(lc) && window[lc] < need[lc]) have--;
        }
    }
    return minLen == int.MaxValue ? "" : s.Substring(start, minLen);
}
// Time: O(|s| + |t|), Space: O(|s| + |t|)
#15 Reconstruct Itinerary (Eulerian Path) Hard

Given a list of airline tickets [from, to], reconstruct the itinerary using all tickets. Start from "JFK". Use smallest lexical order when multiple choices exist.

Hierholzer's Algorithm (Eulerian Path): Greedy DFS — always pick the smallest next destination. Add to result only when backtracking (dead-end first). Reverse at end. This correctly handles dead-ends that must be visited last.
public IList<string> FindItinerary(IList<IList<string>> tickets)
{
    var adj = new Dictionary<string, SortedList<string, int>>();

    foreach (var t in tickets)
    {
        if (!adj.ContainsKey(t[0])) adj[t[0]] = new SortedList<string, int>(Comparer<string>.Create((a,b) => string.Compare(a,b)));
        var list = adj[t[0]];
        // SortedList doesn't allow duplicates — use count as key trick
        list[t[1] + "#" + list.Count] = 0;
    }

    // Simpler: use sorted list per node
    var graph  = new Dictionary<string, SortedSet<string>>();
    var counts = new Dictionary<string, Queue<string>>();

    foreach (var t in tickets)
    {
        if (!counts.ContainsKey(t[0])) counts[t[0]] = new();
    }
    // Sort tickets, then enqueue in order
    var sorted = tickets.OrderBy(t => t[0]).ThenBy(t => t[1]).ToList();
    foreach (var t in sorted)
    {
        if (!counts.ContainsKey(t[0])) counts[t[0]] = new();
        counts[t[0]].Enqueue(t[1]);
    }

    var result = new LinkedList<string>();
    var stack  = new Stack<string>();
    stack.Push("JFK");

    while (stack.Count > 0)
    {
        string curr = stack.Peek();
        if (counts.ContainsKey(curr) && counts[curr].Count > 0)
            stack.Push(counts[curr].Dequeue()); // Go deeper
        else
            result.AddFirst(stack.Pop()); // Dead end → add to front
    }
    return result.ToList();
}
// Time: O(E log E) for sorting, O(E) for traversal; Space: O(E)
