1. String Hashing (Polynomial Rolling Hash)

Hashing transforms a substring into an integer so two substrings can be compared in O(1) instead of O(n). The rolling hash moves a window along the string without recomputing from scratch.

Polynomial hash of string s: hash = s[0]*p^(n-1) + s[1]*p^(n-2) + ... + s[n-1]*p^0 (mod M) where p = small prime (31 for lowercase letters), M = large prime (10^9+9) Prefix hash array: h[i] = hash of s[0..i-1] h[0] = 0 h[i] = h[i-1] * p + s[i-1] Substring hash s[l..r] = (h[r+1] - h[l] * p^(r-l+1)) mod M ↑ O(1) after O(n) preprocessing!
public class StringHash
{
    private const long P = 31, MOD = 1_000_000_007;
    private long[] h, pw;

    public StringHash(string s)
    {
        int n = s.Length;
        h  = new long[n + 1];
        pw = new long[n + 1];
        pw[0] = 1;

        for (int i = 0; i < n; i++)
        {
            h[i + 1]  = (h[i] * P + (s[i] - 'a' + 1)) % MOD;
            pw[i + 1] = pw[i] * P % MOD;
        }
    }

    // Hash of substring s[l..r] (0-indexed, inclusive)
    public long Get(int l, int r)
        => (h[r + 1] - h[l] * pw[r - l + 1] % MOD + MOD * 2) % MOD;
}

// Use case: check if two substrings are equal in O(1)
// Longest duplicate substring: binary search on length + hash set
Hash Collision Risk: With one hash, there's a ~1/MOD chance of false match. For critical applications, use double hashing (two different base/mod pairs) to reduce collision probability to ~1/MOD².

2. Rabin-Karp Algorithm

Uses rolling hash to search for a pattern in a text. Compare hashes first (O(1)); only do character-by-character comparison on hash match. Average O(n+m), worst case O(n*m) if many hash collisions.

public IList<int> RabinKarp(string text, string pattern)
{
    int n = text.Length, m = pattern.Length;
    const long BASE = 31, MOD = 1_000_000_007;

    long patHash = 0, winHash = 0, power = 1;
    var result = new List<int>();

    // Compute pattern hash and BASE^(m-1)
    for (int i = 0; i < m; i++)
    {
        patHash = (patHash * BASE + pattern[i]) % MOD;
        winHash = (winHash * BASE + text[i])    % MOD;
        if (i < m - 1) power = power * BASE % MOD; // BASE^(m-1)
    }

    for (int i = 0; i <= n - m; i++)
    {
        if (winHash == patHash && text.Substring(i, m) == pattern)
            result.Add(i); // Verify on match to avoid false positives

        if (i < n - m) // Roll the window: remove leftmost, add next char
            winHash = (winHash - text[i] * power % MOD + MOD) * BASE % MOD + text[i + m];
    }
    return result;
}
// Average: O(n + m), good for multiple pattern search (Aho-Corasick) or long text

3. KMP — Knuth-Morris-Pratt Pattern Matching

KMP avoids re-examining characters after a mismatch by precomputing a failure function (also called the LPS — Longest Proper Prefix that is also a Suffix array). This lets the algorithm skip ahead intelligently.

Pattern: "ABABC" LPS array: [0, 0, 1, 2, 0] ↑ For each position i, LPS[i] = length of longest proper prefix of pattern[0..i] that is also a suffix. Pattern: A B A B C LPS: 0 0 1 2 0 LPS[2]=1: "ABA" → longest proper prefix that is also suffix = "A" (length 1) LPS[3]=2: "ABAB" → "AB" is both prefix and suffix → length 2 When mismatch at pattern[j]: Don't restart from 0 — jump to LPS[j-1] and retry This saves O(m) work per mismatch!
// Build the LPS (failure function) array
private int[] BuildLPS(string pattern)
{
    int m = pattern.Length;
    int[] lps = new int[m];
    int len = 0, i = 1;

    while (i < m)
    {
        if (pattern[i] == pattern[len])
        {
            lps[i++] = ++len; // Extend the matching prefix
        }
        else if (len > 0)
        {
            len = lps[len - 1]; // Fall back to shorter prefix (don't increment i)
        }
        else
        {
            lps[i++] = 0; // No prefix matches
        }
    }
    return lps;
}

// KMP search: find all occurrences of pattern in text
public List<int> KMPSearch(string text, string pattern)
{
    int[] lps = BuildLPS(pattern);
    var result = new List<int>();
    int i = 0, j = 0; // text index, pattern index

    while (i < text.Length)
    {
        if (text[i] == pattern[j]) { i++; j++; }

        if (j == pattern.Length)
        {
            result.Add(i - j); // Match found at position i-j
            j = lps[j - 1];   // Look for next occurrence
        }
        else if (i < text.Length && text[i] != pattern[j])
        {
            if (j > 0) j = lps[j - 1]; // Use LPS to skip
            else      i++;              // No prefix match, advance text
        }
    }
    return result;
}
// Time: O(n + m), Space: O(m) for LPS array

KMP Application: Repeated String Match

// How many times must we repeat 'a' so that 'b' is a substring? (LeetCode 686)
public int RepeatedStringMatch(string a, string b)
{
    int reps = (int)Math.Ceiling((double)b.Length / a.Length);
    string candidate = string.Concat(Enumerable.Repeat(a, reps + 1));
    return KMPSearch(candidate, b).Count > 0 ? reps : (KMPSearch(String.Concat(Enumerable.Repeat(a, reps+1)), b).Count > 0 ? reps+1 : -1);
}

4. Z-Algorithm

The Z-array for string S: Z[i] = length of the longest substring starting at S[i] that is also a prefix of S. Another O(n) string tool — often simpler than KMP for basic pattern matching.

String S = "aabxaa" Z array: [6, 1, 0, 0, 2, 1] ↑ Z[0] = length of string (by convention) Z[1] = 1: S[1..] = "abxaa", prefix match = "a" (length 1) Z[4] = 2: S[4..] = "aa", prefix match = "aa" (length 2) Pattern matching trick: build pattern + "$" + text Positions i in text where Z[i] == pattern.Length → pattern found there!
public int[] ZFunction(string s)
{
    int n = s.Length;
    int[] z = new int[n];
    z[0] = n;
    int l = 0, r = 0;

    for (int i = 1; i < n; i++)
    {
        if (i < r)
            z[i] = Math.Min(r - i, z[i - l]); // Reuse already computed values
        while (i + z[i] < n && s[z[i]] == s[i + z[i]])
            z[i]++;             // Extend match
        if (i + z[i] > r) { l = i; r = i + z[i]; }
    }
    return z;
}

// Pattern matching using Z-function
public List<int> ZSearch(string text, string pattern)
{
    string combined = pattern + "$" + text; // "$" is a sentinel not in alphabet
    int[] z = ZFunction(combined);
    var matches = new List<int>();

    for (int i = pattern.Length + 1; i < combined.Length; i++)
        if (z[i] == pattern.Length)
            matches.Add(i - pattern.Length - 1); // Index in original text
    return matches;
}
// Time: O(n + m), Space: O(n + m)

5. Manacher's Algorithm — Longest Palindromic Substring in O(n)

Manacher's exploits palindrome symmetry: if we know palindromes inside a larger palindrome, we can skip recomputation. Finds all palindrome radii in O(n).

Naive approach: for each center, expand outward → O(n²) Manacher's key insight: If center c has palindrome radius r, and we're computing radius at i (inside the palindrome): → i's mirror i' = 2c - i is already computed → radius[i] >= min(radius[i'], r - (i - c)) → Then extend from there (amortized O(n)) Trick: insert '#' between chars so even-length palindromes become odd: "abba" → "#a#b#b#a#" Now every palindrome has an odd length with a center character!
public string LongestPalindrome(string s)
{
    // Transform: "abc" → "#a#b#c#"
    var t = "#" + string.Join("#", s.ToCharArray()) + "#";
    int n = t.Length;
    int[] p = new int[n]; // p[i] = palindrome radius at i (in transformed string)
    int c = 0, r = 0;    // c = center of rightmost palindrome, r = its right boundary

    for (int i = 0; i < n; i++)
    {
        int mirror = 2 * c - i;
        if (i < r) p[i] = Math.Min(r - i, p[mirror]); // Use symmetry

        // Attempt to expand around center i
        while (i + p[i] + 1 < n && i - p[i] - 1 >= 0 &&
               t[i + p[i] + 1] == t[i - p[i] - 1])
            p[i]++;

        if (i + p[i] > r) { c = i; r = i + p[i]; } // Update rightmost palindrome
    }

    // Find the maximum radius
    int maxLen = 0, center = 0;
    for (int i = 0; i < n; i++)
        if (p[i] > maxLen) { maxLen = p[i]; center = i; }

    // Convert back to original string indices: start = (center - maxLen) / 2
    return s.Substring((center - maxLen) / 2, maxLen);
}
// Time: O(n), Space: O(n)

6. Trie — Advanced Operations

Beyond basic insert/search, Tries support prefix counting, wildcard matching, and word deletion. Common in autocomplete, spell-check, and IP routing (CIDR prefixes).

public class Trie
{
    private class TrieNode
    {
        public TrieNode[] Children = new TrieNode[26];
        public int      Count   = 0;  // Words passing through this node
        public bool     IsEnd   = false;
    }

    private readonly TrieNode root = new();

    public void Insert(string word)
    {
        var node = root;
        foreach (char c in word)
        {
            int idx = c - 'a';
            node.Children[idx] ??= new TrieNode();
            node = node.Children[idx];
            node.Count++;
        }
        node.IsEnd = true;
    }

    // Count words with a given prefix (for autocomplete suggestions)
    public int CountWithPrefix(string prefix)
    {
        var node = root;
        foreach (char c in prefix)
        {
            node = node.Children[c - 'a'];
            if (node == null) return 0;
        }
        return node.Count;
    }

    // Wildcard search: '.' matches any character (DFS)
    public bool SearchWildcard(string word)
    {
        return Dfs(root, word, 0);

        bool Dfs(TrieNode node, string w, int idx)
        {
            if (node == null) return false;
            if (idx == w.Length) return node.IsEnd;

            char c = w[idx];
            if (c != '.') return Dfs(node.Children[c - 'a'], w, idx + 1);

            // Wildcard: try all 26 children
            foreach (var child in node.Children)
                if (Dfs(child, w, idx + 1)) return true;
            return false;
        }
    }
}
// Insert/Search/Prefix: O(m) where m = word length, Space: O(total chars * 26)

7. Word Break (Trie + DP)

Combining Trie lookup with DP solves the word break problem efficiently: can string s be segmented into dictionary words?

// Word Break (LeetCode 139) — DP approach
public bool WordBreak(string s, IList<string> wordDict)
{
    var dict = new HashSet<string>(wordDict);
    int n = s.Length;
    bool[] dp = new bool[n + 1];
    dp[0] = true; // Empty string can always be segmented

    for (int i = 1; i <= n; i++)
        for (int j = 0; j < i; j++)
            if (dp[j] && dict.Contains(s.Substring(j, i - j)))
            {
                dp[i] = true; break;
            }
    return dp[n];
}
// Time: O(n² * m) where m = avg word length for substring check
// Optimize: use Trie to check prefixes and only iterate to max word length
