Section 4 — Building Graphs in C#
// ── Method 1: Dictionary<int, List<int>> (most flexible)
var graph = new Dictionary<int, List<int>>();

// Add edge u→v (directed) or u—v (undirected)
void AddEdge(int u, int v, bool directed = false)
{
    if (!graph.ContainsKey(u)) graph[u] = new();
    if (!graph.ContainsKey(v)) graph[v] = new();
    graph[u].Add(v);
    if (!directed) graph[v].Add(u);
}

// ── Method 2: List<int>[] adj (faster, when n is known)
int n = 5;
List<int>[] adj = Enumerable.Range(0, n).Select(_ => new List<int>()).ToArray();
adj[0].Add(1); adj[1].Add(0); // undirected 0—1

// ── Method 3: Build from edges array (common LeetCode pattern)
int[][] edges = { new[]{0,1}, new[]{1,2}, new[]{2,3} };
int numNodes = 4;
List<int>[] g = new List<int>[numNodes];
for (int i = 0; i < numNodes; i++) g[i] = new();
foreach (var e in edges)
{
    g[e[0]].Add(e[1]);
    g[e[1]].Add(e[0]); // omit for directed
}

// ── Method 4: Weighted graph with (neighbor, weight) pairs
List<(int to, int weight)>[] wg = new List<(int,int)>[numNodes];
for (int i = 0; i < numNodes; i++) wg[i] = new();
wg[0].Add((1, 5)); // edge 0→1 with weight 5

// ── Method 5: Node class (for Clone Graph type problems)
public class GraphNode
{
    public int Val;
    public List<GraphNode> Neighbors = new();
    public GraphNode(int val) => Val = val;
}
Section 5 — BFS (Breadth-First Search)

BFS explores all neighbors of a node before going deeper. It processes nodes level by level. Key property: BFS finds the shortest path in an unweighted graph.

BFS on graph: 0─1, 0─2, 1─3, 1─4, 2─5 [0] / \ [1] [2] / \ \ [3] [4] [5] Start at node 0. Queue: [0], visited: {0} Level 0: Dequeue 0 → process. Add neighbors 1,2. Queue:[1,2], visited:{0,1,2} Level 1: Dequeue 1 → process. Add unvisited neighbors 3,4. Dequeue 2 → process. Add unvisited neighbor 5. Queue:[3,4,5], visited:{0,1,2,3,4,5} Level 2: Dequeue 3,4,5 → process. No unvisited neighbors. BFS order: 0 → 1 → 2 → 3 → 4 → 5 Distances from node 0: dist[0]=0, dist[1]=1, dist[2]=1, dist[3]=2, dist[4]=2, dist[5]=2
// BFS template — works for any graph
public static int[] BFS(List<int>[] graph, int start)
{
    int n = graph.Length;
    int[] dist = new int[n];
    Array.Fill(dist, -1);      // -1 = unvisited
    dist[start] = 0;

    var queue = new Queue<int>();
    queue.Enqueue(start);

    while (queue.Count > 0)
    {
        int node = queue.Dequeue();
        foreach (int neighbor in graph[node])
        {
            if (dist[neighbor] == -1)  // not visited
            {
                dist[neighbor] = dist[node] + 1;
                queue.Enqueue(neighbor);
            }
        }
    }
    return dist;  // dist[i] = shortest path from start to i (-1 if unreachable)
}
// Time: O(V + E)  Space: O(V)
// The visited check (dist[neighbor] == -1) is CRITICAL
// Without it: infinite loops in cyclic graphs

// BFS to reconstruct path (not just distances)
public static List<int>? BFSPath(List<int>[] graph, int start, int end)
{
    int n = graph.Length;
    int[] prev = new int[n];
    Array.Fill(prev, -1);
    bool[] visited = new bool[n];
    visited[start] = true;

    var q = new Queue<int>();
    q.Enqueue(start);
    while (q.Count > 0)
    {
        int u = q.Dequeue();
        if (u == end) break;
        foreach (int v in graph[u])
        {
            if (!visited[v]) { visited[v] = true; prev[v] = u; q.Enqueue(v); }
        }
    }
    if (!visited[end]) return null; // unreachable
    var path = new List<int>();
    for (int v = end; v != -1; v = prev[v]) path.Add(v);
    path.Reverse();
    return path;
}
Section 6 — BFS on Grids

A 2D grid is a graph in disguise. Each cell is a vertex; edges go to adjacent cells (up/down/left/right). Grid BFS is one of the most common interview patterns.

GRID BFS — Shortest path from S to E in grid: Grid (. = open, # = wall): S . . # . . # . . . . # . # . . . . # E BFS distances from S (row 0, col 0): 0 1 2 # . 1 # 3 4 5 2 # 4 # 6 3 4 5 # 7 Shortest path length = 7 steps. Direction arrays: int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}}; // right,left,down,up Encoding: use (row*cols + col) or tuple (row,col) as queue element
// Generic Grid BFS — shortest path or reachability
public int GridBFS(char[][] grid, int sr, int sc, int er, int ec)
{
    int rows = grid.Length, cols = grid[0].Length;
    bool[,] visited = new bool[rows, cols];
    int[][] dirs = { new[]{0,1}, new[]{0,-1}, new[]{1,0}, new[]{-1,0} };

    var q = new Queue<(int r, int c, int dist)>();
    q.Enqueue((sr, sc, 0));
    visited[sr, sc] = true;

    while (q.Count > 0)
    {
        var (r, c, d) = q.Dequeue();
        if (r == er && c == ec) return d;
        foreach (var dir in dirs)
        {
            int nr = r + dir[0], nc = c + dir[1];
            if (nr >= 0 && nr < rows && nc >= 0 && nc < cols
                && !visited[nr, nc] && grid[nr][nc] != '#')
            {
                visited[nr, nc] = true;
                q.Enqueue((nr, nc, d + 1));
            }
        }
    }
    return -1; // unreachable
}
// Time: O(rows × cols), Space: O(rows × cols)

// Number of Islands (LC 200) — BFS flood fill approach
public int NumIslands(char[][] grid)
{
    int rows = grid.Length, cols = grid[0].Length;
    int islands = 0;
    int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
            if (grid[r][c] == '1') { islands++; BFSMark(grid, r, c, rows, cols, dirs); }
    return islands;
}
private void BFSMark(char[][] grid, int r, int c, int rows, int cols, int[][] dirs)
{
    var q = new Queue<(int,int)>();
    q.Enqueue((r, c)); grid[r][c] = '0'; // mark visited
    while (q.Count > 0)
    {
        var (cr,cc) = q.Dequeue();
        foreach (var d in dirs)
        {
            int nr=cr+d[0], nc=cc+d[1];
            if (nr>=0 && nr<rows && nc>=0 && nc<cols && grid[nr][nc]=='1')
            { grid[nr][nc]='0'; q.Enqueue((nr,nc)); }
        }
    }
}
Section 7 — Multi-Source BFS

Sometimes we need to start BFS from multiple sources simultaneously — as if all sources fire at step 0. Classic examples: Rotting Oranges, Walls and Gates.

MULTI-SOURCE BFS — "Rotting Oranges" Grid: 2=rotten orange, 1=fresh, 0=empty Initial: 2 1 1 1 1 0 0 1 1 Enqueue ALL rotten oranges at time=0: Queue: [(0,0,0), ...] (row, col, time) Minute 0: Sources (rotten) = {(0,0)} Minute 1: Fresh oranges adjacent to rotten rot → {(0,1),(1,0)} Minute 2: {(0,2),(1,1)} Minute 3: {(2,1),(1,0) already done} Minute 4: {(2,2)} Answer: 4 minutes (or -1 if any fresh remain unreachable)
// LC 994 — Rotting Oranges (Multi-source BFS template)
public int OrangesRotting(int[][] grid)
{
    int rows = grid.Length, cols = grid[0].Length;
    int fresh = 0;
    int[][] dirs = { new[]{0,1},new[]{0,-1},new[]{1,0},new[]{-1,0} };
    var q = new Queue<(int r, int c)>();

    // Seed all rotten oranges at once
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
        {
            if (grid[r][c] == 2) q.Enqueue((r, c));
            if (grid[r][c] == 1) fresh++;
        }

    int minutes = 0;
    while (q.Count > 0 && fresh > 0)
    {
        minutes++;
        int sz = q.Count;
        for (int i = 0; i < sz; i++)
        {
            var (r, c) = q.Dequeue();
            foreach (var d in dirs)
            {
                int nr = r+d[0], nc = c+d[1];
                if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && grid[nr][nc] == 1)
                {
                    grid[nr][nc] = 2; fresh--; q.Enqueue((nr, nc));
                }
            }
        }
    }
    return fresh == 0 ? minutes : -1;
}
// Time: O(rows×cols), Space: O(rows×cols)
// KEY: enqueue ALL sources before starting BFS — they all expand simultaneously
Section 8 — Bipartite Check (2-Coloring)

A graph is bipartite if you can color all vertices with 2 colors such that no two adjacent vertices share the same color. This is equivalent to: the graph has no odd-length cycles.

BIPARTITE CHECK via BFS 2-coloring: Graph: 1─2, 2─3, 3─4, 4─1 Color 1 = RED, start BFS from 1 Node 1 → color RED, queue:[2,4] Node 2 → color BLUE, queue:[3] (1 already colored) Node 4 → color BLUE, queue:[3] (1 already colored) Node 3 → color RED, queue:[] (2 and 4 already BLUE) RED: {1,3}, BLUE: {2,4} All edges go RED→BLUE or BLUE→RED → BIPARTITE ✓ NON-bipartite graph: triangle 1─2, 2─3, 3─1 Color 1=RED, 2=BLUE, 3=? 3 adjacent to 1(RED) → must be BLUE 3 adjacent to 2(BLUE) → must be RED CONFLICT → NOT BIPARTITE ✗
// LC 785 — Is Graph Bipartite?
public bool IsBipartite(int[][] graph)
{
    int n = graph.Length;
    int[] color = new int[n]; // 0=uncolored, 1=red, 2=blue

    for (int start = 0; start < n; start++)
    {
        if (color[start] != 0) continue; // already processed
        var q = new Queue<int>();
        q.Enqueue(start); color[start] = 1;

        while (q.Count > 0)
        {
            int node = q.Dequeue();
            foreach (int nb in graph[node])
            {
                if (color[nb] == 0)
                {
                    color[nb] = color[node] == 1 ? 2 : 1; // alternate color
                    q.Enqueue(nb);
                }
                else if (color[nb] == color[node])
                    return false; // same color → not bipartite
            }
        }
    }
    return true;
}
// Outer loop handles disconnected graphs (multiple components)
// Time: O(V+E), Space: O(V)

Key Takeaways:
• Graph = V vertices + E edges. Most problems use adjacency list.
• BFS = queue-based. Visits all neighbors before going deeper.
• BFS guarantees shortest path in unweighted graphs.
• Always mark visited when enqueuing, not when dequeuing (avoids duplicates).
• Multi-source BFS: enqueue ALL sources at step 0 before the main loop.
• Bipartite: BFS 2-coloring. Conflict = not bipartite.
