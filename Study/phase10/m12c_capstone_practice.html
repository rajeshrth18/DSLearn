#1 Trapping Rain Water Medium Two Pointers Arrays

Given an array of bar heights, compute how much water can be trapped between bars.

Example: [0,1,0,2,1,0,1,3,2,1,2,1] → 6

Key Insight: Water above position i = min(maxLeft[i], maxRight[i]) - height[i]. Use two-pointer to avoid O(n) prefix arrays.
public int Trap(int[] h)
{
    int l = 0, r = h.Length - 1, lMax = 0, rMax = 0, water = 0;
    while (l < r)
    {
        if (h[l] < h[r])
        {
            if (h[l] >= lMax) lMax = h[l];
            else water += lMax - h[l];
            l++;
        }
        else
        {
            if (h[r] >= rMax) rMax = h[r];
            else water += rMax - h[r];
            r--;
        }
    }
    return water;
}
// Time: O(n), Space: O(1)
#2 LRU Cache Medium LinkedList HashMap

Design a data structure that follows the LRU (Least Recently Used) cache eviction policy. O(1) get and put.

Design: Dictionary maps key → LinkedListNode for O(1) access. LinkedList maintains order (most-recent at front). On access/add: move to front. On eviction: remove from back.
public class LRUCache
{
    private readonly int _cap;
    private readonly Dictionary<int, LinkedListNode<(int key, int val)>> _map;
    private readonly LinkedList<(int key, int val)> _list;

    public LRUCache(int capacity)
    {
        _cap  = capacity;
        _map  = new();
        _list = new();
    }

    public int Get(int key)
    {
        if (!_map.TryGetValue(key, out var node)) return -1;
        _list.Remove(node);
        _list.AddFirst(node);
        return node.Value.val;
    }

    public void Put(int key, int value)
    {
        if (_map.TryGetValue(key, out var existing))
        {
            _list.Remove(existing);
            _map.Remove(key);
        }
        else if (_list.Count == _cap)
        {
            _map.Remove(_list.Last.Value.key);
            _list.RemoveLast();
        }
        var node = _list.AddFirst((key, value));
        _map[key] = node;
    }
}
// Get: O(1), Put: O(1), Space: O(capacity)
#3 Combination Sum IV (Count Ways) Medium Dynamic Programming

Given an array of distinct integers and a target, return the number of possible combinations that add up to target. Order matters: [1,2] and [2,1] are different.

Example: nums=[1,2,3], target=4 → 7

Unlike 0/1 knapsack: Order matters → iterate target in outer loop, numbers in inner loop. This counts permutations, not combinations.
public int CombinationSum4(int[] nums, int target)
{
    int[] dp = new int[target + 1];
    dp[0] = 1; // One way to make 0: pick nothing
    for (int t = 1; t <= target; t++)
        foreach (int n in nums)
            if (n <= t) dp[t] += dp[t - n];
    return dp[target];
}
// Time: O(target * nums.Length), Space: O(target)
#4 Decode Ways Medium Dynamic Programming Strings

A string of digits can be decoded (1→A, 2→B, ..., 26→Z). Count how many ways to decode it.

Example: "226" → 3 (2-2-6 = BBF, 22-6 = VF, 2-26 = BZ)

public int NumDecodings(string s)
{
    int n = s.Length;
    int[] dp = new int[n + 1];
    dp[0] = 1;                                   // Empty string = 1 way
    dp[1] = s[0] == '0' ? 0 : 1;                // Leading zero = 0 ways

    for (int i = 2; i <= n; i++)
    {
        int one = s[i-1] - '0';                  // Single digit
        int two = (s[i-2] - '0') * 10 + one;   // Two-digit number
        if (one >= 1)             dp[i] += dp[i - 1]; // Valid single
        if (two >= 10 && two <= 26) dp[i] += dp[i - 2]; // Valid double
    }
    return dp[n];
}
// Time: O(n), Space: O(n) → can reduce to O(1) with rolling variables
#5 Partition Equal Subset Sum Medium DP / 0-1 Knapsack

Given an array, determine if it can be partitioned into two subsets with equal sum.

Reduce to: Can we find a subset summing to total/2? 0/1 knapsack with boolean DP.
public bool CanPartition(int[] nums)
{
    int sum = nums.Sum();
    if (sum % 2 != 0) return false;
    int target = sum / 2;

    bool[] dp = new bool[target + 1];
    dp[0] = true;

    foreach (int num in nums)
        for (int j = target; j >= num; j--) // Reverse to avoid using num twice
            dp[j] |= dp[j - num];

    return dp[target];
}
// Time: O(n * sum), Space: O(sum)
#6 Rotting Oranges (Multi-Source BFS) Medium BFS Graphs

Grid contains 0 (empty), 1 (fresh), 2 (rotten). Each minute, fresh oranges adjacent to rotten become rotten. Return minimum minutes to rot all oranges, or -1 if impossible.

public int OrangesRotting(int[][] grid)
{
    int m = grid.Length, n = grid[0].Length;
    var q = new Queue<(int, int)>();
    int fresh = 0;

    for (int r = 0; r < m; r++)
    for (int c = 0; c < n; c++)
    {
        if (grid[r][c] == 2) q.Enqueue((r, c));
        if (grid[r][c] == 1) fresh++;
    }
    if (fresh == 0) return 0;

    int[] dr = {-1, 1, 0, 0}, dc = {0, 0, -1, 1};
    int minutes = 0;
    while (q.Count > 0)
    {
        int sz = q.Count; minutes++;
        while (sz-- > 0)
        {
            var (r, c) = q.Dequeue();
            for (int d = 0; d < 4; d++)
            {
                int nr = r + dr[d], nc = c + dc[d];
                if (nr >= 0 && nr < m && nc >= 0 && nc < n && grid[nr][nc] == 1)
                { grid[nr][nc] = 2; fresh--; q.Enqueue((nr, nc)); }
            }
        }
    }
    return fresh == 0 ? minutes - 1 : -1;
}
// Time: O(m*n), Space: O(m*n)
#7 Kth Largest Element in a Stream Medium Heap

Design a class that finds the kth largest element in a stream. add(val) adds a value and returns the kth largest.

Min-heap of size k: The smallest element in the heap is the k-th largest overall. When heap exceeds size k, pop the minimum.
public class KthLargest
{
    private readonly int _k;
    private readonly PriorityQueue<int, int> _minHeap = new();

    public KthLargest(int k, int[] nums)
    {
        _k = k;
        foreach (int n in nums) Add(n);
    }

    public int Add(int val)
    {
        _minHeap.Enqueue(val, val);
        while (_minHeap.Count > _k) _minHeap.Dequeue();
        return _minHeap.Peek();
    }
}
// Add: O(log k), Space: O(k)
#8 Jump Game II (Minimum Jumps) Medium Greedy Arrays

Given jump lengths at each position, find the minimum number of jumps to reach the last index.

Greedy: At each step, track the farthest reachable position. When you reach the end of the current jump range, you must take another jump — greedily extend to the farthest seen.
public int Jump(int[] nums)
{
    int jumps = 0, currEnd = 0, farthest = 0;
    for (int i = 0; i < nums.Length - 1; i++)
    {
        farthest = Math.Max(farthest, i + nums[i]);
        if (i == currEnd) // Must jump — current range exhausted
        {
            jumps++;
            currEnd = farthest;
        }
    }
    return jumps;
}
// Time: O(n), Space: O(1)
#9 Coin Change II (Number of Ways) Medium DP / Unbounded Knapsack

Given coins of different denominations and amount, return the number of combinations to make that amount.

Example: amount=5, coins=[1,2,5] → 4 (5=5, 5=2+2+1, 5=2+1+1+1, 5=1+1+1+1+1)

Unbounded knapsack (combinations, order doesn't matter): Iterate coins in outer loop, amount in inner. This ensures each combination is counted once.
public int Change(int amount, int[] coins)
{
    int[] dp = new int[amount + 1];
    dp[0] = 1;
    foreach (int coin in coins)        // Outer = coins (prevents duplicate combinations)
        for (int a = coin; a <= amount; a++) // Inner = amount (forward = reuse coin)
            dp[a] += dp[a - coin];
    return dp[amount];
}
// Time: O(amount * coins), Space: O(amount)

#10 Burst Balloons (Interval DP) Hard Interval DP

Given n balloons, each with a number. Bursting balloon i earns coins[i-1] * coins[i] * coins[i+1]. Burst all to maximize coins. Boundary balloons have value 1.

Example: [3,1,5,8] → 167

Think backwards: Instead of "which to burst first", think "which balloon is burst LAST in interval (i,j)". The last balloon left earns coins[i] * coins[k] * coins[j] where i and j are the boundaries.
public int MaxCoins(int[] nums)
{
    int n = nums.Length;
    int[] arr = new int[n + 2];
    arr[0] = arr[n + 1] = 1;
    for (int i = 0; i < n; i++) arr[i + 1] = nums[i];

    int m = n + 2;
    int[,] dp = new int[m, m];

    for (int len = 2; len < m; len++)
    for (int i = 0; i < m - len; i++)
    {
        int j = i + len;
        for (int k = i + 1; k < j; k++)
            dp[i, j] = Math.Max(dp[i, j],
                dp[i, k] + arr[i] * arr[k] * arr[j] + dp[k, j]);
    }
    return dp[0, m - 1];
}
// Time: O(n³), Space: O(n²)
#11 Count Vowel Strings in Range (Digit DP Style) Hard DP Strings

Count strings of length n using only vowels (a,e,i,o,u) in lexicographically non-decreasing order. This generalizes to: count combinations with repetition.

Example: n=1 → 5, n=2 → 15 (aa,ae,ai,ao,au,ee,ei,...)

DP: dp[i][v] = count of non-decreasing strings of length i ending with vowel index v. Transition: dp[i][v] = sum of dp[i-1][0..v].
public int CountVowelStrings(int n)
{
    int[] dp = {1, 1, 1, 1, 1}; // 5 vowels, each 1 string of length 1
    for (int len = 2; len <= n; len++)
        for (int v = 3; v >= 0; v--) // v=0→'a', accumulate from right
            dp[v] += dp[v + 1];
    return dp.Sum();
}
// This is C(n+4, 4) = stars-and-bars. DP is O(5n), closed form O(1).

#12 Maximum Profit with K Transactions Hard Stock DP State Machine

You can complete at most k buy-sell transactions. Find the maximum profit.

public int MaxProfit(int k, int[] prices)
{
    int n = prices.Length;
    if (n == 0) return 0;

    // If k >= n/2, unlimited transactions (can do every profitable day)
    if (k >= n / 2)
        return prices.Zip(prices.Skip(1), (a, b) => Math.Max(0, b-a)).Sum();

    // buy[j]: max profit when we OWN stock with j completed transactions so far
    // sell[j]: max profit when we DO NOT own stock with j completed transactions
    int[] buy  = Enumerable.Repeat(int.MinValue / 2, k + 1).ToArray();
    int[] sell = new int[k + 1];

    foreach (int price in prices)
    for (int j = k; j >= 1; j--)
    {
        sell[j] = Math.Max(sell[j], buy[j] + price);    // Sell: complete tx j
        buy[j]  = Math.Max(buy[j], sell[j-1] - price);  // Buy: start tx j
    }
    return sell[k];
}
// Time: O(n * k), Space: O(k)
#13 Minimum Cost to Hire K Workers (Bitmask / Heap) Hard Greedy Heap

Hire exactly K workers. Pay a "wage group" where every worker is paid proportional to their quality relative to wage ratio. Minimize total cost.

Key Insight: Sort by wage/quality ratio. For each worker as the "captain" (highest ratio), all K workers must be paid at captain's ratio. To minimize total: pick K workers with minimum quality sum from workers considered so far.
public double MincostToHireWorkers(int[] quality, int[] wage, int k)
{
    int n = quality.Length;
    // Sort by wage/quality ratio ascending
    var workers = Enumerable.Range(0, n)
        .Select(i => ((double)wage[i] / quality[i], quality[i]))
        .OrderBy(w => w.Item1)
        .ToArray();

    // Max-heap to track the k smallest quality values seen so far
    var maxHeap = new PriorityQueue<int, int>(); // (quality, -quality) for max
    int qualSum = 0;
    double ans = double.MaxValue;

    foreach (var (ratio, q) in workers)
    {
        maxHeap.Enqueue(q, -q); // Negative priority → max heap behavior
        qualSum += q;
        if (maxHeap.Count > k) // Remove the largest quality to keep minimum k
        {
            maxHeap.TryDequeue(out int maxQ, out _);
            qualSum -= maxQ;
        }
        if (maxHeap.Count == k)
            ans = Math.Min(ans, ratio * qualSum);
    }
    return ans;
}
// Time: O(n log n + n log k), Space: O(k)
#14 Serialize and Deserialize Binary Tree Hard Trees BFS

Design an algorithm to serialize a binary tree to a string and deserialize it back. No constraints on format.

public class Codec
{
    // BFS level-order serialization (like LeetCode's own format)
    public string Serialize(TreeNode root)
    {
        if (root == null) return "#";
        var sb = new StringBuilder();
        var q  = new Queue<TreeNode>();
        q.Enqueue(root);
        while (q.Count > 0)
        {
            var node = q.Dequeue();
            if (node == null) { sb.Append("#,"); continue; }
            sb.Append(node.val).Append(',');
            q.Enqueue(node.left);
            q.Enqueue(node.right);
        }
        return sb.ToString().TrimEnd(',');
    }

    public TreeNode Deserialize(string data)
    {
        if (data == "#") return null;
        var parts = data.Split(',');
        var root  = new TreeNode(int.Parse(parts[0]));
        var q     = new Queue<TreeNode>();
        q.Enqueue(root);
        int i = 1;
        while (q.Count > 0 && i < parts.Length)
        {
            var node = q.Dequeue();
            if (i < parts.Length && parts[i] != "#")
                { node.left = new TreeNode(int.Parse(parts[i])); q.Enqueue(node.left); }
            i++;
            if (i < parts.Length && parts[i] != "#")
                { node.right = new TreeNode(int.Parse(parts[i])); q.Enqueue(node.right); }
            i++;
        }
        return root;
    }
}
// Time: O(n), Space: O(n)
#15 Median of Two Sorted Arrays Hard Binary Search Arrays

Given two sorted arrays nums1 and nums2, find the median of both combined in O(log(m+n)) time.

Binary Search on Partition: Find a partition of both arrays such that left half and right half contain equal elements and max(left) ≤ min(right). Binary search on the partition index of the smaller array.
public double FindMedianSortedArrays(int[] A, int[] B)
{
    if (A.Length > B.Length) (A, B) = (B, A); // Always binary search on smaller
    int m = A.Length, n = B.Length;
    int lo = 0, hi = m;

    while (lo <= hi)
    {
        int pA = (lo + hi) / 2;   // Partition A: pA elements on left
        int pB = (m + n + 1) / 2 - pA; // Partition B: pB elements on left

        int maxLeftA  = pA == 0 ? int.MinValue : A[pA - 1];
        int minRightA = pA == m ? int.MaxValue : A[pA];
        int maxLeftB  = pB == 0 ? int.MinValue : B[pB - 1];
        int minRightB = pB == n ? int.MaxValue : B[pB];

        if (maxLeftA <= minRightB && maxLeftB <= minRightA)
        {
            // Perfect partition found
            if ((m + n) % 2 == 1)
                return Math.Max(maxLeftA, maxLeftB);
            return (Math.Max(maxLeftA, maxLeftB) + Math.Min(minRightA, minRightB)) / 2.0;
        }
        else if (maxLeftA > minRightB) hi = pA - 1; // Move partition left in A
        else                           lo = pA + 1; // Move partition right in A
    }
    throw new InvalidOperationException(); // Should never reach here
}
// Time: O(log(min(m,n))), Space: O(1)
