Heap Problems — Easy (1–4)
#1Kth Largest Element in an Array
EasyMin-HeapTop-K

Given an integer array nums and integer k, return the k-th largest element.

Example: nums = [3,2,1,5,6,4], k = 2 → 5
Approach: Maintain a min-heap of size K. At the end, heap top = K-th largest.
public int FindKthLargest(int[] nums, int k)
{
    var pq = new PriorityQueue<int, int>();
    foreach (int n in nums)
    {
        pq.Enqueue(n, n);
        if (pq.Count > k) pq.Dequeue();
    }
    pq.TryPeek(out int res, out _);
    return res;
}
// Time: O(n log k), Space: O(k)
#2Last Stone Weight
EasyMax-Heap

Stones have weights. Each round, smash the two heaviest: if equal → both gone; if different → surviving stone has weight difference. Return weight of last stone (0 if none).

Example: [2,7,4,1,8,1] → 1
Approach: Max-heap (negate values for min-heap). Each round, dequeue two heaviest. If different, push back the difference.
public int LastStoneWeight(int[] stones)
{
    var pq = new PriorityQueue<int, int>();
    foreach (int s in stones) pq.Enqueue(s, -s); // max-heap
    while (pq.Count > 1)
    {
        int a = pq.Dequeue(), b = pq.Dequeue(); // two heaviest
        if (a != b) pq.Enqueue(a - b, -(a - b));
    }
    return pq.Count == 0 ? 0 : pq.Dequeue();
}
// Time: O(n log n), Space: O(n)
#3K Closest Points to Origin
EasyMax-Heap

Given an array of points on a plane, return the K closest to the origin (0,0). Distance = x²+y² (no need to sqrt).

Example: points=[[1,3],[-2,2]], k=1 → [[-2,2]]
Approach: Max-heap of size K on distance. Keep K smallest distances → when heap exceeds K, remove the farthest.
public int[][] KClosest(int[][] points, int k)
{
    // max-heap: negate distance so largest distance is popped first
    var pq = new PriorityQueue<int[], int>();
    foreach (var p in points)
    {
        int dist = p[0] * p[0] + p[1] * p[1];
        pq.Enqueue(p, -dist);      // max-heap on distance
        if (pq.Count > k) pq.Dequeue(); // remove farthest
    }
    return pq.UnorderedItems().Select(x => x.Element).ToArray();
}
// Time: O(n log k), Space: O(k)
#4Relative Ranks
EasyMax-Heap

Assign medals to athletes: highest score = "Gold Medal", 2nd = "Silver Medal", 3rd = "Bronze Medal", rest get their rank as a string.

Approach: Max-heap on score. Pop in order and assign ranks. Track original index to fill result array.
public string[] FindRelativeRanks(int[] score)
{
    var pq = new PriorityQueue<int, int>();  // (index, -score)
    for (int i = 0; i < score.Length; i++)
        pq.Enqueue(i, -score[i]);
    string[] medals = { "Gold Medal", "Silver Medal", "Bronze Medal" };
    string[] res = new string[score.Length];
    int rank = 1;
    while (pq.Count > 0)
    {
        int idx = pq.Dequeue();
        res[idx] = rank <= 3 ? medals[rank - 1] : rank.ToString();
        rank++;
    }
    return res;
}
// Time: O(n log n), Space: O(n)
Heap Problems — Medium (5–8)
#5Top K Frequent Elements
MediumMin-HeapHashMap

Given nums, return the k most frequent elements.

Example: [1,1,1,2,2,3], k=2 → [1,2]
Approach: Build frequency map. Min-heap of size K on frequency — top = K-th most frequent, so we drop least frequent when heap exceeds K.
public int[] TopKFrequent(int[] nums, int k)
{
    var freq = new Dictionary<int, int>();
    foreach (int n in nums) freq[n] = freq.GetValueOrDefault(n) + 1;
    var pq = new PriorityQueue<int, int>();
    foreach (var (num, cnt) in freq)
    {
        pq.Enqueue(num, cnt);
        if (pq.Count > k) pq.Dequeue();
    }
    var res = new int[k];
    for (int i = k - 1; i >= 0; i--) res[i] = pq.Dequeue();
    return res;
}
// Time: O(n log k), Space: O(n)
#6Task Scheduler
MediumMax-HeapGreedy

CPU tasks in array. Same tasks must be separated by n intervals. Return minimum number of intervals needed.

Example: tasks=["A","A","A","B","B","B"], n=2 → 8 (A→B→idle→A→B→idle→A→B)
Approach: Greedy with max-heap. Always run the most frequent available task. Use a queue to track cooldown: (freq, readyTime). At each time step, pop from cooldown queue if ready, pick max-freq task from heap.
public int LeastInterval(char[] tasks, int n)
{
    var freq = new int[26];
    foreach (char t in tasks) freq[t - 'A']++;
    var maxH = new PriorityQueue<int, int>(); // max-heap (negate)
    foreach (int f in freq) if (f > 0) maxH.Enqueue(f, -f);
    // Queue: (remaining_count, next_available_time)
    var cooldown = new Queue<(int cnt, int ready)>();
    int time = 0;
    while (maxH.Count > 0 || cooldown.Count > 0)
    {
        time++;
        if (maxH.Count > 0)
        {
            int cnt = maxH.Dequeue() - 1;
            if (cnt > 0) cooldown.Enqueue((cnt, time + n));
        }
        if (cooldown.Count > 0 && cooldown.Peek().ready == time)
        {
            var (c, _) = cooldown.Dequeue();
            maxH.Enqueue(c, -c);
        }
    }
    return time;
}
// Time: O(n log 26) = O(n), Space: O(26) = O(1)
#7Reorganize String
MediumMax-HeapGreedy

Rearrange a string so no two adjacent characters are the same. Return empty string if impossible.

Example: "aab" → "aba"
Approach: Greedy: always pick most frequent available character that isn't same as last placed. Use max-heap. After placing, put previous char back if it still has count remaining.
public string ReorganizeString(string s)
{
    int[] freq = new int[26];
    foreach (char c in s) freq[c - 'a']++;
    var pq = new PriorityQueue<char, int>();
    for (int i = 0; i < 26; i++)
        if (freq[i] > 0) pq.Enqueue((char)('a' + i), -freq[i]);
    var sb = new System.Text.StringBuilder();
    char prev = ' '; int prevCnt = 0;
    while (pq.Count > 0)
    {
        var c = pq.Dequeue();
        sb.Append(c);
        if (prevCnt > 0) pq.Enqueue(prev, -prevCnt); // re-add prev char
        prev = c; prevCnt = freq[c - 'a'] - 1;
        freq[c - 'a']--;
    }
    return sb.Length == s.Length ? sb.ToString() : "";
}
// Time: O(n log 26) = O(n), Space: O(26)
#8Kth Smallest Element in a Sorted Matrix
MediumMin-Heap

Given an n×n matrix where each row and column is sorted in ascending order, find the k-th smallest element.

Approach: Min-heap: start with first element of each row. Pop minimum, push next element in same row. After K pops, return the K-th popped value.
public int KthSmallest(int[][] matrix, int k)
{
    int n = matrix.Length;
    // (value, row, col)
    var pq = new PriorityQueue<(int v, int r, int c), int>();
    for (int r = 0; r < n; r++)
        pq.Enqueue((matrix[r][0], r, 0), matrix[r][0]);
    for (int i = 0; i < k - 1; i++)
    {
        var (v, r, c) = pq.Dequeue();
        if (c + 1 < n) pq.Enqueue((matrix[r][c+1], r, c+1), matrix[r][c+1]);
    }
    pq.TryPeek(out var res, out _);
    return res.v;
}
// Time: O(k log n), Space: O(n)
Heap Problems — Hard (9–11)
#9Merge K Sorted Lists
HardMin-HeapLinked List

Merge k sorted linked lists and return one sorted list.

Approach: Min-heap holds the current front node of each list. Pop min, advance that list, push next node. Attach all popped nodes to a dummy head.
public ListNode? MergeKLists(ListNode?[] lists)
{
    var pq = new PriorityQueue<ListNode, int>();
    foreach (var head in lists)
        if (head != null) pq.Enqueue(head, head.Val);
    var dummy = new ListNode(); var cur = dummy;
    while (pq.Count > 0)
    {
        var node = pq.Dequeue();
        cur.Next = node; cur = cur.Next;
        if (node.Next != null) pq.Enqueue(node.Next, node.Next.Val);
    }
    return dummy.Next;
}
// Time: O(N log k), N = total nodes, k = number of lists
#10Find Median from Data Stream
HardTwo HeapsDesign

Design a class supporting AddNum(int num) and FindMedian() in O(log n) and O(1) respectively.

Approach: Two heaps: max-heap for lower half, min-heap for upper half. Invariant: maxH.size == minH.size ± 1. Median = top of larger heap, or average of both tops.
public class MedianFinder
{
    private PriorityQueue<int, int> _lo = new(); // max-heap (lower half)
    private PriorityQueue<int, int> _hi = new(); // min-heap (upper half)

    public void AddNum(int num)
    {
        _lo.Enqueue(num, -num);         // push to lower half
        // Ensure lower ≤ upper: move lo's max to hi if needed
        int loTop = _lo.Peek();
        if (_hi.Count > 0 && loTop > _hi.Peek())
            _hi.Enqueue(_lo.Dequeue(), _lo.Count > 0 ? _lo.Peek() : loTop);
        // Balance sizes
        if (_lo.Count > _hi.Count + 1)
        {
            int v = _lo.Dequeue(); _hi.Enqueue(v, v);
        }
        else if (_hi.Count > _lo.Count)
        {
            int v = _hi.Dequeue(); _lo.Enqueue(v, -v);
        }
    }
    public double FindMedian()
    {
        if (_lo.Count > _hi.Count) return _lo.Peek();
        return (_lo.Peek() + _hi.Peek()) / 2.0;
    }
}
// AddNum: O(log n), FindMedian: O(1)
#11Sliding Window Maximum
HardMonotonic Deque

Given array nums and integer k, return max of each window of size k.

Example: [1,3,-1,-3,5,3,6,7], k=3 → [3,3,5,5,6,7]
Approach: Monotonic deque storing indices in decreasing order of value. Front = max of current window. Remove indices that left the window from front. Remove smaller elements from back before inserting (they can never be max).
public int[] MaxSlidingWindow(int[] nums, int k)
{
    var dq  = new LinkedList<int>(); // stores indices, decreasing values
    var res = new List<int>();
    for (int i = 0; i < nums.Length; i++)
    {
        // Remove indices outside window
        while (dq.Count > 0 && dq.First!.Value <= i - k)
            dq.RemoveFirst();
        // Maintain decreasing order — remove smaller elements from back
        while (dq.Count > 0 && nums[dq.Last!.Value] < nums[i])
            dq.RemoveLast();
        dq.AddLast(i);
        if (i >= k - 1) res.Add(nums[dq.First!.Value]); // window is full
    }
    return res.ToArray();
}
// Time: O(n) — each element pushed/popped at most once
// Space: O(k) for deque
Trie Problems — Easy to Hard (12–15)
#12Implement Trie (Prefix Tree)
EasyTrieDesign

Implement a trie with insert(word), search(word), and startsWith(prefix) methods.

Approach: TrieNode with Dictionary of children + isEnd flag. All operations traverse character by character.
public class Trie
{
    private class Node { public Dictionary<char,Node> C=new(); public bool End; }
    private Node _root = new();

    public void Insert(string w)
    {
        var cur = _root;
        foreach (char c in w) { if(!cur.C.ContainsKey(c)) cur.C[c]=new(); cur=cur.C[c]; }
        cur.End = true;
    }
    public bool Search(string w)   { var n=Get(w); return n!=null&&n.End; }
    public bool StartsWith(string p) { return Get(p) != null; }
    private Node? Get(string s)
    {
        var cur = _root;
        foreach (char c in s) { if(!cur.C.TryGetValue(c, out cur!)) return null; }
        return cur;
    }
}
// All operations: O(m) where m = word length
#13Replace Words
MediumTrie

Given a dictionary of roots and a sentence, replace every word with the shortest root that is a prefix of it.

Example: dictionary=["cat","bat","rat"], sentence="the cattle was rattled by the battery" → "the cat was rat by the bat"
Approach: Insert all roots into trie. For each word in sentence, traverse trie until we find an isEnd node (= shortest root prefix). If found, replace word with that prefix.
public string ReplaceWords(IList<string> dict, string sentence)
{
    var trie = new Trie();  // use Trie from problem 12
    foreach (var root in dict) trie.Insert(root);

    var words = sentence.Split(' ');
    for (int i = 0; i < words.Length; i++)
    {
        string prefix = ShortestRoot(trie, words[i]);
        if (prefix != null) words[i] = prefix;
    }
    return string.Join(" ", words);
}
// ShortestRoot: traverse trie, return prefix when isEnd hit
// Time: O(d + sentence_len), d = total chars in dictionary
#14Design Add and Search Words (Wildcard)
MediumTrieDFS

Design a data structure that supports addWord(word) and search(word) where '.' matches any character.

Approach: Trie with recursive search. For '.', try all children recursively. For exact chars, follow trie normally.
public class WordDictionary
{
    private class Node { public Dictionary<char,Node> C=new(); public bool End; }
    private Node _root = new();

    public void AddWord(string w)
    {
        var c = _root;
        foreach (char ch in w) { if(!c.C.ContainsKey(ch)) c.C[ch]=new(); c=c.C[ch]; }
        c.End = true;
    }
    public bool Search(string w) => Dfs(_root, w, 0);
    private bool Dfs(Node n, string w, int i)
    {
        if (i == w.Length) return n.End;
        char c = w[i];
        if (c != '.')
            return n.C.TryGetValue(c, out var nx) && Dfs(nx, w, i+1);
        foreach (var child in n.C.Values)
            if (Dfs(child, w, i+1)) return true;
        return false;
    }
}
// Search: O(m) exact, O(26^dots × m) with wildcards
#15Maximum XOR of Two Numbers in an Array
HardBinary TrieBit Manipulation

Given integer array nums, return the maximum XOR of any two numbers. (LeetCode 421)

Insight: Build a trie of bit representations. For each number, greedily choose the opposite bit at each level to maximize XOR.
Approach: Binary trie — each node has children[0] and children[1] for bits. Insert all numbers (bit by bit from MSB). For each number, try to pick the opposite bit at each step to maximize XOR.
public int FindMaximumXOR(int[] nums)
{
    // Binary Trie node: two children (bit 0 and bit 1)
    int[][] trie = new int[nums.Length * 32][];
    for (int i = 0; i < trie.Length; i++) trie[i] = new int[2] { -1, -1 };
    int idx = 1;

    foreach (int n in nums)
    {
        int cur = 0;
        for (int b = 31; b >= 0; b--)
        {
            int bit = (n >> b) & 1;
            if (trie[cur][bit] == -1) trie[cur][bit] = idx++;
            cur = trie[cur][bit];
        }
    }
    int maxXor = 0;
    foreach (int n in nums)
    {
        int cur = 0, xorVal = 0;
        for (int b = 31; b >= 0; b--)
        {
            int bit = (n >> b) & 1;
            int want = 1 - bit; // want the opposite to maximize XOR
            if (trie[cur][want] != -1) { xorVal |= (1 << b); cur = trie[cur][want]; }
            else                          { cur = trie[cur][bit]; }
        }
        maxXor = Math.Max(maxXor, xorVal);
    }
    return maxXor;
}
// Time: O(32n) = O(n), Space: O(32n) = O(n)
// Binary trie: each level = one bit. Depth = 32 (int size).
