P1 Kth Largest Element in a Stream
Easy Heap Min-Heap

Design a class to find the k-th largest element in a stream. The class is initialized with an integer k and an initial array of numbers. The Add(val) method adds a new value and returns the k-th largest element.

Approach: Maintain a min-heap of size k. The root (minimum) of this heap is always the k-th largest element overall. When adding a new value: if heap size < k, push directly. If new value > heap root, pop root and push new value. Return heap root as the answer.
Time: O(log k) per add — heap has at most k elements &nbsp;|&nbsp; Space: O(k) — min-heap stores k largest elements
C# 
public class KthLargest
{
    private readonly PriorityQueue<int, int> _heap = new();
    private readonly int _k;

    public KthLargest(int k, int[] nums) {
        _k = k;
        foreach (int n in nums) Add(n);
    }

    public int Add(int val) {
        _heap.Enqueue(val, val);
        while (_heap.Count > _k) _heap.Dequeue();
        return _heap.Peek();
    }
}
// Usage: var kl = new KthLargest(3, new[]{4,5,8,2});
// kl.Add(3) = 4, kl.Add(5) = 5, kl.Add(10) = 5, kl.Add(9) = 8, kl.Add(4) = 8
Python

        self.k, self.heap = k, []
        for n in nums: self.add(n)

    def add(self, val: int) -> int:
        heapq.heappush(self.heap, val)
        while len(self.heap) > self.k: heapq.heappop(self.heap)
        return self.heap[0]
P2 Implement Trie (Prefix Tree)
Easy Trie

Implement the Trie class with Insert(word), Search(word), and StartsWith(prefix). LeetCode #208.

Approach: Each TrieNode has a dictionary mapping chars to children and an IsEnd flag. Insert: walk the string creating nodes as needed, set IsEnd=true at the end. Search: walk the string, return true if all chars found AND last node IsEnd=true. StartsWith: walk the string, return true if all chars found (don't check IsEnd).
Time: O(L) per operation — L = word or prefix length &nbsp;|&nbsp; Space: O(N×L) — N words, each L characters stored in Trie nodes
C# 
public class Trie
{
    private class Node { public Dictionary<char,Node> C=new(); public bool E; }
    private readonly Node _root = new();

    public void Insert(string w) {
        var n=_root; foreach(var c in w){ if(!n.C.ContainsKey(c))n.C[c]=new(); n=n.C[c]; } n.E=true;
    }
    private Node? Walk(string s) {
        var n=_root; foreach(var c in s){ if(!n.C.TryGetValue(c,out n!)) return null; } return n;
    }
    public bool Search(string w) { var n=Walk(w); return n!=null&&n.E; }
    public bool StartsWith(string p) => Walk(p)!=null;
}
Python 
class Trie:
    def __init__(self): self.root = {}
    def insert(self, w):
        n = self.root
        for c in w: n = n.setdefault(c, {})
        n['#'] = True
    def _walk(self, s):
        n = self.root
        for c in s:
            if c not in n: return None
            n = n[c]
        return n
    def search(self, w): n=self._walk(w); return n is not None and '#' in n
    def startsWith(self, p): return self._walk(p) is not None
P3 Last Stone Weight
Easy Heap Max-Heap

You have an array of stone weights. Repeatedly smash the two heaviest stones: if equal, both destroyed; if different, the smaller is destroyed and the larger loses weight equal to the smaller. Return the weight of the last stone (0 if none remain). LeetCode #1046.

Approach: Use a max-heap. Repeatedly extract top 2, compute difference, push back if non-zero. Stop when = 1 stone remains.
Time: O(n log n) — n stones, each enqueue/dequeue is O(log n) &nbsp;|&nbsp; Space: O(n) — heap holds all stones
C# 
public int LastStoneWeight(int[] stones) {
    var pq = new PriorityQueue<int,int>();
    foreach(int s in stones) pq.Enqueue(s,-s);  // Max-heap: negate priority
    while(pq.Count > 1) {
        int a=pq.Dequeue(), b=pq.Dequeue();
        if(a!=b) pq.Enqueue(a-b, -(a-b));
    }
    return pq.Count==0 ? 0 : pq.Dequeue();
}
Python 
import heapq
def lastStoneWeight(stones):
    h = [-s for s in stones]
    heapq.heapify(h)
    while len(h) > 1:
        a, b = -heapq.heappop(h), -heapq.heappop(h)
        if a != b: heapq.heappush(h, -(a-b))
    return -h[0] if h else 0
P4 Word Search II (Trie + DFS Backtracking)
Easy Trie DFS Backtracking

Given a 2D board and a list of words, find all words that exist in the board (adjacent cells, no reuse). LeetCode #212.

Approach: Build a Trie from all words. DFS from every cell: at each step, check if current char is in the Trie child. If IsEnd reached, add word to results. Mark visited during DFS (restore on backtrack). This avoids re-searching for each word independently.
Time: O(W×L + m×n×3^L) — build Trie + DFS from each cell; W=words, L=avg length, m×n=board &nbsp;|&nbsp; Space: O(W×L) — Trie storage
C# 
public IList<string> FindWords(char[][] board, string[] words) {
    // Build Trie
    var root = new Dictionary<char,object>();
    foreach(var w in words) {
        var n = root;
        foreach(var c in w) { if(!n.ContainsKey(c)) n[c]= new Dictionary<char,object>(); n=(Dictionary<char,object>)n[c]; }
        n['$'] = w;
    }
    int R=board.Length, C=board[0].Length;
    var res = new HashSet<string>();
    void DFS(int r, int c, Dictionary<char,object> node) {
        if(r<0||r>=R||c<0||c>=C||board[r][c]=='#'||!node.ContainsKey(board[r][c])) return;
        char ch=board[r][c]; board[r][c]='#';
        var next=(Dictionary<char,object>)node[ch];
        if(next.ContainsKey('$')) res.Add((string)next['$']);
        foreach(var(dr,dc) in new[]{(0,1),(0,-1),(1,0),(-1,0)}) DFS(r+dr,c+dc,next);
        board[r][c]=ch;
    }
    for(int r=0;r<R;r++) for(int c=0;c<C;c++) DFS(r,c,root);
    return new List<string>(res);
}
Python 
def findWords(board, words):
    root = {}
    for w in words:
        n = root
        for c in w: n = n.setdefault(c, {})
        n['$'] = w
    R, C, res = len(board), len(board[0]), set()
    def dfs(r, c, node):
        if not (0<=r<R and 0<=c<C) or board[r][c] not in node: return
        ch = board[r][c]; nxt = node[ch]
        if '$' in nxt: res.add(nxt['$'])
        board[r][c] = '#'
        for dr,dc in [(0,1),(0,-1),(1,0),(-1,0)]: dfs(r+dr, c+dc, nxt)
        board[r][c] = ch
    for r in range(R):
        for c in range(C): dfs(r, c, root)
    return list(res)
P5 Find Median from Data Stream
Easy Two Heaps Max-Heap + Min-Heap

Design a data structure supporting AddNum(int num) and FindMedian(). LeetCode #295.

Approach (Two Heaps): Maintain a max-heap (lower half) and a min-heap (upper half). After each insert, rebalance so |lower.size - upper.size| = 1. Median: if equal sizes, average of both tops; if unequal, top of the larger heap.
Time: O(log n) per add, O(1) find median — heap operations &nbsp;|&nbsp; Space: O(n) — two heaps storing all n elements
C# 
public class MedianFinder
{
    // lo = max-heap (lower half, negated for PQ), hi = min-heap (upper half)
    private readonly PriorityQueue<int,int> _lo=new(), _hi=new();

    public void AddNum(int n) {
        _lo.Enqueue(n, -n);                            // Push to max-heap
        _hi.Enqueue(_lo.Peek(), _lo.Peek());           // Balance: push lo's max to hi
        _lo.Dequeue();
        if(_hi.Count > _lo.Count) { int v=_hi.Dequeue(); _lo.Enqueue(v,-v); }
    }

    public double FindMedian() =>
        _lo.Count == _hi.Count
            ? (_lo.Peek() + _hi.Peek()) / 2.0
            : _lo.Peek();
}
Python 
import heapq

class MedianFinder:
    def __init__(self): self.lo, self.hi = [], []  # lo=max-heap(neg), hi=min-heap
    def addNum(self, n: int) -> None:
        heapq.heappush(self.lo, -n)
        heapq.heappush(self.hi, -heapq.heappop(self.lo))
        if len(self.hi) > len(self.lo):
            heapq.heappush(self.lo, -heapq.heappop(self.hi))
    def findMedian(self) -> float:
        return (-self.lo[0] + self.hi[0]) / 2.0 if len(self.lo)==len(self.hi) else -self.lo[0]

P6 Merge K Sorted Lists
Medium Heap K-Way Merge

Given k sorted linked lists, merge them into one sorted linked list. LeetCode #23.

Approach: Use a min-heap of size k. Push the head of each list. At each step: pop min node, append to result, push that node's next (if exists). O(n log k) where n = total nodes.
Time: O(n log k) — n total nodes, each heap operation is O(log k) &nbsp;|&nbsp; Space: O(k) — min-heap holds one node per list
C# 
public ListNode? MergeKLists(ListNode?[] lists) {
    var pq = new PriorityQueue<ListNode,int>();
    foreach(var n in lists) if(n!=null) pq.Enqueue(n, n.val);
    var dummy = new ListNode(); var cur = dummy;
    while(pq.Count>0) {
        cur.next = pq.Dequeue(); cur = cur.next;
        if(cur.next!=null) pq.Enqueue(cur.next, cur.next.val);
    }
    return dummy.next;
}
Python 
import heapq

def mergeKLists(lists):
    h, dummy = [], type('N',(),{'val':0,'next':None})()
    for n in lists:
        if n: heapq.heappush(h, (n.val, id(n), n))
    cur = dummy
    while h:

        cur.next = node; cur = cur.next
        if node.next: heapq.heappush(h, (node.next.val, id(node.next), node.next))
    return dummy.next
P7 Replace Words with Root (Trie)
Medium Trie String

Given a dictionary of roots and a sentence, replace each word in the sentence with its shortest matching root. If a word has multiple roots, use the shortest one. LeetCode #648.

Approach: Build a Trie from the dictionary. For each word in the sentence, walk the Trie until an IsEnd node is reached — that's the shortest root. If no root found, keep the original word.
Time: O(N×L + S×L) — build Trie from N roots + scan S sentence words &nbsp;|&nbsp; Space: O(N×L) — Trie for N dictionary roots
C# 
public string ReplaceWords(IList<string> dict, string sentence) {
    var root = new Dictionary<char,object>();
    foreach(var w in dict) {
        var n = root;
        foreach(var c in w) { if(!n.ContainsKey(c)) n[c]=new Dictionary<char,object>(); n=(Dictionary<char,object>)n[c]; }
        n['$'] = true;
    }
    string Replace(string word) {
        var n=root; var sb=new System.Text.StringBuilder();
        foreach(var c in word) {
            if(!n.ContainsKey(c)) break;
            sb.Append(c); n=(Dictionary<char,object>)n[c];
            if(n.ContainsKey('$')) return sb.ToString();
        }
        return word;
    }
    return string.Join(" ", sentence.Split(' ').Select(Replace));
}

def replaceWords(dictionary, sentence):

P8 K Closest Points to Origin
Medium Heap Max-Heap of Size K

Given a list of points, find the k closest to the origin. LeetCode #973.

Approach: Maintain a max-heap of size k. For each point, if distance < heap top, remove top and add current. Alternatively, push all and extract k times. Using a max-heap of size k avoids sorting all n points.
Time: O(n log k) — n points, heap ops on size-k heap &nbsp;|&nbsp; Space: O(k) — max-heap stores k closest points
C# 
public int[][] KClosest(int[][] points, int k) {
    // Max-heap of size k: priority = distance (negate for max-heap behavior)
    var pq = new PriorityQueue<int[], int>();
    foreach(var p in points) {
        int d = p[0]*p[0]+p[1]*p[1];
        pq.Enqueue(p, -d);         // Negate: largest dist = lowest priority = dequeued first
        if(pq.Count>k) pq.Dequeue();
    }
    return pq.UnorderedItems().Select(x=>x.Element).ToArray();
}
Python 
import heapq
def kClosest(points, k):
    h = []
    for x, y in points:
        d = x*x + y*y
        heapq.heappush(h, (-d, x, y))   # Max-heap via negation
        if len(h) > k: heapq.heappop(h)
    return [[x, y] for _, x, y in h]
P9 Maximum Width of Binary Tree
Medium Binary Tree BFS + Index Tricks

Find the maximum width of a binary tree (max number of nodes between the leftmost and rightmost non-null nodes at each level, inclusive). LeetCode #662.

Approach: BFS level-order, assign index to each node (root=1, left child=2i, right child=2i+1). At each level, width = last_index - first_index + 1. Normalize indices per level to avoid integer overflow.
Time: O(n) — BFS visits every node exactly once &nbsp;|&nbsp; Space: O(w) — w = maximum width (nodes in widest level), queue holds one level at a time
C# 
public int WidthOfBinaryTree(TreeNode? root) {
    if(root==null) return 0;
    int ans=0;
    var q=new Queue<(TreeNode,long)>(); q.Enqueue((root,1));
    while(q.Count>0) {
        int cnt=q.Count; long first=0,last=0;
        for(int i=0;i<cnt;i++) {
            var(node,idx)=q.Dequeue();
            if(i==0) first=idx;
            last=idx;
            if(node.left!=null)  q.Enqueue((node.left,  2*(idx-first)));
            if(node.right!=null) q.Enqueue((node.right, 2*(idx-first)+1));
        }
        ans=(int)Math.Max(ans,last-first+1);
    }
    return ans;
}
Python 
from collections import deque
def widthOfBinaryTree(root):
    if not root: return 0
    ans = 0; q = deque([(root, 1)])
    while q:
        first = last = 0
        for i in range(len(q)):
            node, idx = q.popleft()
            if i == 0: first = idx
            last = idx
            if node.left:  q.append((node.left,  2*(idx-first)))
            if node.right: q.append((node.right, 2*(idx-first)+1))
        ans = max(ans, last-first+1)
    return ans
P10 Longest Word in Dictionary (Trie)
Medium Trie String

Given a list of words, find the longest word that can be built one character at a time. Each intermediate prefix must also be in the dictionary. LeetCode #720.

Approach: Build a Trie. BFS/DFS on the Trie: only follow edges where the current node has IsEnd=true (meaning the prefix is itself a valid word). Track the longest word found.
Time: O(n log n + n×L) — sort + HashSet lookups per word &nbsp;|&nbsp; Space: O(n×L) — HashSet stores all valid words
C# 
public string LongestWord(string[] words) {
    Array.Sort(words);  // Alphabetical sort ensures shorter words processed first
    var built = new HashSet<string> { "" };
    string best = "";
    foreach(var w in words) {
        if(built.Contains(w[..(w.Length-1)])) {  // All prefixes already in built
            built.Add(w);
            if(w.Length > best.Length) best=w;
        }
    }
    return best;
}
// O(n * L) using sort + HashSet — equivalent to Trie approach

def longestWord(words):

    return best

P11 Smallest Range Covering K Lists
Hard Heap K-Way Merge Variant

Given k sorted lists, find the smallest range [a, b] that includes at least one number from each list. LeetCode #632.

Approach: Use a min-heap initialized with the first element of each list. Track the current maximum. At each step: range = [heap_min, current_max]. When popping, advance that list's pointer and update max. Stop when any list is exhausted.
Time: O(n log k) — n = total elements across all lists, k lists in heap &nbsp;|&nbsp; Space: O(k) — heap holds one element per list
C# 
public int[] SmallestRange(IList<IList<int>> nums) {
    var pq = new PriorityQueue<(int val,int row,int col),int>();
    int curMax = int.MinValue;
    for(int i=0;i<nums.Count;i++) {
        pq.Enqueue((nums[i][0],i,0), nums[i][0]);
        curMax=Math.Max(curMax,nums[i][0]);
    }
    int[] ans={0,int.MaxValue};
    while(pq.Count==nums.Count) {
        var(val,row,col)=pq.Dequeue();
        if(curMax-val < ans[1]-ans[0]) ans=new[]{val,curMax};
        if(col+1>=nums[row].Count) break;
        int nxt=nums[row][col+1];
        pq.Enqueue((nxt,row,col+1),nxt);
        curMax=Math.Max(curMax,nxt);
    }
    return ans;
}
Python 
import heapq
def smallestRange(nums):
    h = [(lst[0], i, 0) for i, lst in enumerate(nums)]
    heapq.heapify(h)
    cur_max = max(v for v, _, _ in h)
    ans = [-float('inf'), float('inf')]
    while len(h) == len(nums):
        val, row, col = heapq.heappop(h)
        if cur_max - val < ans[1] - ans[0]: ans = [val, cur_max]
        if col + 1 >= len(nums[row]): break
        nxt = nums[row][col+1]
        heapq.heappush(h, (nxt, row, col+1))
        cur_max = max(cur_max, nxt)
    return ans
P12 Design Search Autocomplete System
Hard Trie Design Min-Heap

Design an autocomplete system. Initialize with a list of sentences and their frequencies. Input(c) adds character c to current query and returns the top 3 matching sentences by frequency (ties broken alphabetically). LeetCode #642.

Approach: Build a Trie where each node stores a dict of {sentence: frequency}. For each character typed, walk the Trie and collect all sentences at the prefix node. Use a min-heap of size 3 to find top-3. On '#', save the current query with its frequency.
Time: O(L) per input — walk Trie + heap of size 3 &nbsp;|&nbsp; Space: C# 
public class AutocompleteSystem {
    private class TNode {
        public Dictionary<char, TNode> ch = new();
        public Dictionary<string, int> sents = new();
    }
    private TNode root = new(), cur;
    private Dictionary<string, int> freq = new();
    private System.Text.StringBuilder buf = new();
    public AutocompleteSystem(string[] sentences, int[] times) {
        cur = root;
        for (int i = 0; i < sentences.Length; i++)
            { freq[sentences[i]] = times[i]; Add(sentences[i]); }
    }
    private void Add(string s) {
        var n = root;
        foreach (char c in s) {
            if (!n.ch.ContainsKey(c)) n.ch[c] = new TNode();
            n = n.ch[c]; n.sents[s] = freq.GetValueOrDefault(s);
        }
    }
    public IList<string> Input(char c) {
        if (c == '#') {
            string s = buf.ToString();
            freq[s] = freq.GetValueOrDefault(s) + 1;
            Add(s); cur = root; buf.Clear(); return new List<string>();
        }
        buf.Append(c);
        if (cur == null || !cur.ch.ContainsKey(c)) { cur = null; return new List<string>(); }
        cur = cur.ch[c];
        return cur.sents
            .OrderByDescending(x => x.Value).ThenBy(x => x.Key)
            .Take(3).Select(x => x.Key).ToList();
    }
}
Python 
import heapq
from collections import defaultdict

class AutocompleteSystem:
    def __init__(self, sentences, times):
        self.trie = {}
        self.freq = defaultdict(int)
        self.cur = []          # current input buffer
        self.node = self.trie  # current trie node
        for s, t in zip(sentences, times):
            self.freq[s] = t
            self._add(s)

    def _add(self, sentence):
        n = self.trie
        for c in sentence:
            n = n.setdefault(c, {})
            n.setdefault('#sents', {})
            n['#sents'][sentence] = self.freq[sentence]

    def input(self, c: str):
        if c == '#':
            s = ''.join(self.cur)
            self.freq[s] += 1
            self._add(s)
            self.cur, self.node = [], self.trie
            return []
        self.cur.append(c)
        if self.node is None or c not in self.node:
            self.node = None; return []
        self.node = self.node[c]
        sents = self.node.get('#sents', {})
        # Top 3 by (-freq, name) using min-heap
        h = []
        for s, f in sents.items():
            heapq.heappush(h, (-f, s))
        res = []
        while h and len(res) < 3: _, s = heapq.heappop(h); res.append(s)
        return res
P13 Closest Nodes Queries in BST
Hard BST Inorder Binary Search

Given a BST and queries [q_i], for each query return [max value = q_i, min value = q_i] (or -1 if none exists). LeetCode #2476.

Approach: Inorder traversal gives sorted array. For each query, binary search for lower bound (= q) and upper bound (= q) in the sorted array. O(n + Q log n).
Time: O(n + Q log n) — O(n) inorder + O(log n) binary search per query Q &nbsp;|&nbsp; Space: O(n) — sorted array from inorder traversal
C# 
public IList<IList<int>> ClosestNodes(TreeNode root, IList<int> queries) {
    var sorted = new List<int>();
    void Inorder(TreeNode? n) { if(n==null)return; Inorder(n.left); sorted.Add(n.val); Inorder(n.right); }
    Inorder(root);
    var res = new List<IList<int>>();
    foreach(int q in queries) {
        int lo = sorted.BinarySearch(q);
        int idx = lo < 0 ? ~lo : lo;
        int maxLe = idx>0 ? sorted[idx-1] : (idx==0&&sorted[0]==q ? q : -1);
        if(lo>=0) maxLe = sorted[lo];
        int minGe = idx < sorted.Count ? sorted[idx] : -1;
        res.Add(new[]{maxLe, minGe});
    }
    return res;
}

import bisect

def closestNodes(root, queries):

    def inorder(n):

        res.append([max_le, min_ge])
    return res
P14 Path Sum III (All Paths)
Hard Binary Tree Prefix Sum DFS

Count paths in a binary tree that sum to a target. Paths can start/end anywhere (not necessarily root-to-leaf), but must go downward. LeetCode #437.

Approach: DFS with a prefix sum HashMap. At each node, cumulative sum = running total. Count how many previous prefix sums = (cumSum - target). This gives the number of paths ending at the current node that sum to target. Add/remove from HashMap during DFS (backtrack).
Time: O(n) — single DFS, O(1) HashMap lookup per node &nbsp;|&nbsp; Space: O(n) — recursion depth + prefix sum HashMap
C# 
public int PathSum(TreeNode root, int targetSum) {
    var prefixCount = new Dictionary<long,int> {{ 0L, 1 }};
    int DFS(TreeNode? n, long curr) {
        if(n==null) return 0;
        curr += n.val;
        int count = prefixCount.GetValueOrDefault(curr - targetSum, 0);
        prefixCount[curr] = prefixCount.GetValueOrDefault(curr,0)+1;
        count += DFS(n.left, curr) + DFS(n.right, curr);
        prefixCount[curr]--;
        return count;
    }
    return DFS(root, 0);
}

from collections import defaultdict
def pathSum(root, targetSum):
    prefix = defaultdict(int); prefix[0] = 1
    def dfs(node, curr):
        if not node: return 0
        curr += node.val
        count = prefix[curr - targetSum]
        prefix[curr] += 1
        count += dfs(node.left, curr) + dfs(node.right, curr)
        prefix[curr] -= 1   # Backtrack
        return count
    return dfs(root, 0)
P15 LCA in Binary Tree + Maximum Distance Between Two Nodes
Hard LCA Tree Diameter DFS

Given a binary tree, find the maximum distance between any two nodes (the tree diameter). LeetCode #543. Relate this to the LCA concept — the longest path must pass through some LCA.

Approach (Diameter via Postorder): For each node as LCA, the longest path through it = left_height + right_height. Track global max. This is equivalent to: for every potential LCA node, compute the longest path that has this node as its LCA.
Time: O(n) — single postorder DFS visits each node once &nbsp;|&nbsp; Space: O(h) — h = tree height (call stack depth)
💡 LCA Connection: The diameter of a binary tree is the longest path between any two leaf nodes. This path must pass through some node — the LCA of the two endpoint leaves. So finding the diameter = for every node, compute (left subtree height + right subtree height), take the max. The node achieving this max is the LCA of the two farthest leaves.
C# 
public int DiameterOfBinaryTree(TreeNode root) {
    int diameter = 0;
    int Height(TreeNode? n) {
        if(n==null) return -1;
        int l=Height(n.left), r=Height(n.right);
        diameter = Math.Max(diameter, l+r+2);  // l+r+2 = path length through n
        return Math.Max(l,r)+1;
    }
    Height(root);
    return diameter;
}
// Note: diameter = max(l+r+2) across all nodes
// When both l=r=-1 (leaf), path length = 0 (just the leaf itself)

def diameterOfBinaryTree(root):
    ans = [0]
    def height(node):
        if not node: return -1
        l, r = height(node.left), height(node.right)
        ans[0] = max(ans[0], l + r + 2)  # Longest path through this node as LCA
        return max(l, r) + 1
    height(root)
    return ans[0]
