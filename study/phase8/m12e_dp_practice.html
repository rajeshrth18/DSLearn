#1 · LeetCode 746 — Min Cost Climbing Stairs
Easy
Pattern: Linear DP
Array cost[] where cost[i] is the cost of step i. You can climb 1 or 2 steps from each step. Find min cost to reach the top (one past the last step). You can start from index 0 or 1.
dp[i] = min cost to reach step i. dp[i] = min(dp[i-1]+cost[i-1], dp[i-2]+cost[i-2]). Answer is dp[n].
C# 
public int MinCostClimbingStairs(int[] cost) {
    int n=cost.Length;
    int[] dp=new int[n+1];
    for(int i=2;i<=n;i++)
        dp[i]=Math.Min(dp[i-1]+cost[i-1],dp[i-2]+cost[i-2]);
    return dp[n];
}
Python 
def minCostClimbingStairs(cost):
    n = len(cost)
    dp = [0] * (n + 1)
    for i in range(2, n + 1):
        dp[i] = min(dp[i-1] + cost[i-1], dp[i-2] + cost[i-2])
    return dp[n]
#2 · LeetCode 413 — Arithmetic Slices
Medium
Pattern: Linear DP
Count the number of arithmetic subarrays (length = 3) in array A.
dp[i] = number of arithmetic slices ending at index i. If A[i]-A[i-1] == A[i-1]-A[i-2], then dp[i] = dp[i-1] + 1 (extend all previous slices plus a new 3-element one). Sum all dp[i].
C# 
public int NumberOfArithmeticSlices(int[] A) {
    int dp=0, ans=0;
    for(int i=2;i<A.Length;i++){
        if(A[i]-A[i-1]==A[i-1]-A[i-2]) dp++; else dp=0;
        ans+=dp;
    }
    return ans;
}
Python 
def numberOfArithmeticSlices(A):
    dp = ans = 0
    for i in range(2, len(A)):
        if A[i] - A[i-1] == A[i-1] - A[i-2]: dp += 1
        else: dp = 0
        ans += dp
    return ans
#3 · LeetCode 983 — Minimum Cost For Tickets
Medium
Pattern: Linear DP with lookback
Travel days are given. 1-day pass costs[0], 7-day pass costs[1], 30-day pass costs[2]. Find minimum cost to cover all travel days.
dp[i] = min cost to cover all travel days up to day i. If day i is not a travel day, dp[i] = dp[i-1]. Otherwise dp[i] = min(dp[i-1]+c1, dp[max(0,i-7)]+c7, dp[max(0,i-30)]+c30).
C# 
public int MincostTickets(int[] days, int[] costs) {
    var travel=new HashSet<int>(days);
    int last=days[^1];
    int[] dp=new int[last+1];
    for(int i=1;i<=last;i++){
        if(!travel.Contains(i)){dp[i]=dp[i-1];continue;}
        dp[i]=Math.Min(dp[i-1]+costs[0],
               Math.Min(dp[Math.Max(0,i-7)]+costs[1],
                        dp[Math.Max(0,i-30)]+costs[2]));
    }
    return dp[last];
}
Python 
def mincostTickets(days, costs):
    travel = set(days)
    last = days[-1]
    dp = [0] * (last + 1)
    for i in range(1, last + 1):
        if i not in travel: dp[i] = dp[i-1]; continue
        dp[i] = min(dp[i-1] + costs[0],
                    dp[max(0, i-7)]  + costs[1],
                    dp[max(0, i-30)] + costs[2])
    return dp[last]
#4 · LeetCode 309 — Best Time to Buy and Sell Stock with Cooldown
Medium
Pattern: State machine DP
Prices array. Can buy/sell, but after selling must wait 1 day (cooldown). Maximize profit.
Three states: hold (holding stock), sold (just sold, in cooldown), rest (not holding, not in cooldown). Transitions: hold ? hold or rest?hold; sold?rest; rest?rest or hold?sold.
C# 
public int MaxProfit(int[] prices) {
    int hold=int.MinValue, sold=0, rest=0;
    foreach(int p in prices)
        (hold,sold,rest)=(Math.Max(hold,rest-p),hold+p,Math.Max(rest,sold));
    return Math.Max(sold,rest);
}
Python 
def maxProfit(prices):
    hold = -float('inf'); sold = 0; rest = 0
    for p in prices:
        hold, sold, rest = (
            max(hold, rest - p),   # buy from rest state
            hold + p,              # sell
            max(rest, sold)        # cooldown ends
        )
    return max(sold, rest)

public int FindLengthOfLCIS(int[] nums) {

}

def findLengthOfLCIS(nums):

    for i in range(1, len(nums)):
        cur = cur + 1 if nums[i] > nums[i-1] else 1
        best = max(best, cur)
    return best
#6 · LeetCode 152 — Maximum Product Subarray
Medium
Pattern: Dual-state DP
Find the contiguous subarray with the largest product.
Track both max and min products ending at i (because a negative × negative = positive). max_dp[i] = max(nums[i], nums[i]*max_dp[i-1], nums[i]*min_dp[i-1]).
C# 
public int MaxProduct(int[] nums) {
    int best=nums[0], mx=nums[0], mn=nums[0];
    for(int i=1;i<nums.Length;i++){
        int n=nums[i];
        (mx,mn)=(Math.Max(n,Math.Max(mx*n,mn*n)),Math.Min(n,Math.Min(mx*n,mn*n)));
        best=Math.Max(best,mx);
    }
    return best;
}
Python 
def maxProduct(nums):
    best = mx = mn = nums[0]
    for n in nums[1:]:
        mx, mn = (max(n, mx*n, mn*n), min(n, mx*n, mn*n))
        best = max(best, mx)
    return best

#7 · LeetCode 516 — Longest Palindromic Subsequence
Medium
Pattern: Interval DP
Find the length of the longest palindromic subsequence in string s.
dp[i][j] = LPS of s[i..j]. If s[i]==s[j]: dp[i][j]=dp[i+1][j-1]+2; else dp[i][j]=max(dp[i+1][j], dp[i][j-1]). Fill diagonally (length 1 up to n).
C# 
public int LongestPalindromeSubseq(string s) {
    int n=s.Length;
    int[,] dp=new int[n,n];
    for(int i=0;i<n;i++) dp[i,i]=1;
    for(int len=2;len<=n;len++)
        for(int i=0;i<=n-len;i++){
            int j=i+len-1;
            if(s[i]==s[j]) dp[i,j]=dp[i+1,j-1]+2;
            else dp[i,j]=Math.Max(dp[i+1,j],dp[i,j-1]);
        }
    return dp[0,n-1];
}
Python 
def longestPalindromeSubseq(s):
    n = len(s)
    dp = [[0]*n for _ in range(n)]
    for i in range(n): dp[i][i] = 1
    for length in range(2, n+1):
        for i in range(n - length + 1):
            j = i + length - 1
            if s[i] == s[j]: dp[i][j] = dp[i+1][j-1] + 2
            else: dp[i][j] = max(dp[i+1][j], dp[i][j-1])
    return dp[0][n-1]

#9 · LeetCode 494 — Target Sum
Medium
Pattern: Knapsack / partition
Assign +/- to each number in nums to reach target. Count ways.
Let P = sum of positive items, N = sum of negative items. P + N = sum, P - N = target. So P = (sum+target)/2. Count subsets with sum P. dp[i] += dp[i-num] for each num.
C# 
public int FindTargetSumWays(int[] nums, int target) {
    int total=nums.Sum();
    if((total+target)%2!=0||Math.Abs(target)>total) return 0;
    int s=(total+target)/2;
    int[] dp=new int[s+1]; dp[0]=1;
    foreach(int n in nums)
        for(int j=s;j>=n;j--) dp[j]+=dp[j-n];
    return dp[s];
}
Python 
def findTargetSumWays(nums, target):
    total = sum(nums)
    if (total + target) % 2 != 0 or abs(target) > total: return 0
    s = (total + target) // 2
    dp = [0] * (s + 1); dp[0] = 1
    for n in nums:
        for j in range(s, n-1, -1): dp[j] += dp[j-n]
    return dp[s]
#10 · LeetCode 115 — Distinct Subsequences
Hard
Pattern: 2D DP
Count distinct subsequences of string s that equal string t.
dp[i][j] = ways to match t[0..j-1] using s[0..i-1]. If s[i-1]==t[j-1]: dp[i][j]=dp[i-1][j-1]+dp[i-1][j] (use or skip this match). Else dp[i][j]=dp[i-1][j].
C# 
public int NumDistinct(string s, string t) {
    int m=s.Length, n=t.Length;
    long[,] dp=new long[m+1,n+1];
    for(int i=0;i<=m;i++) dp[i,0]=1;
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++){
            dp[i,j]=dp[i-1,j];
            if(s[i-1]==t[j-1]) dp[i,j]+=dp[i-1,j-1];
        }
    return (int)dp[m,n];
}
Python 
def numDistinct(s, t):
    m, n = len(s), len(t)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = 1  # empty t matched by any s prefix
    for i in range(1, m+1):
        for j in range(1, n+1):
            dp[i][j] = dp[i-1][j]
            if s[i-1] == t[j-1]: dp[i][j] += dp[i-1][j-1]
    return dp[m][n]
#11 · LeetCode 97 — Interleaving String
Medium
Pattern: 2D DP
Given s1, s2, s3, determine if s3 is an interleaving of s1 and s2.
dp[i][j] = can s3[0..i+j-1] be formed from s1[0..i-1] and s2[0..j-1]. dp[i][j] = (dp[i-1][j] and s1[i-1]==s3[i+j-1]) OR (dp[i][j-1] and s2[j-1]==s3[i+j-1]).
C# 
public bool IsInterleave(string s1, string s2, string s3) {
    int m=s1.Length, n=s2.Length;
    if(m+n!=s3.Length) return false;
    bool[,] dp=new bool[m+1,n+1]; dp[0,0]=true;
    for(int i=1;i<=m;i++) dp[i,0]=dp[i-1,0]&&s1[i-1]==s3[i-1];
    for(int j=1;j<=n;j++) dp[0,j]=dp[0,j-1]&&s2[j-1]==s3[j-1];
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++)
            dp[i,j]=(dp[i-1,j]&&s1[i-1]==s3[i+j-1])||(dp[i,j-1]&&s2[j-1]==s3[i+j-1]);
    return dp[m,n];
}
Python 
def isInterleave(s1, s2, s3):
    m, n = len(s1), len(s2)
    if m + n != len(s3): return False
    dp = [[False]*(n+1) for _ in range(m+1)]
    dp[0][0] = True
    for i in range(1, m+1): dp[i][0] = dp[i-1][0] and s1[i-1]==s3[i-1]
    for j in range(1, n+1): dp[0][j] = dp[0][j-1] and s2[j-1]==s3[j-1]
    for i in range(1, m+1):
        for j in range(1, n+1):
            dp[i][j] = (dp[i-1][j] and s1[i-1]==s3[i+j-1]) or \
                       (dp[i][j-1] and s2[j-1]==s3[i+j-1])
    return dp[m][n]
#12 · LeetCode 312 — Burst Balloons
Hard
Pattern: Interval DP
Given n balloons with nums[i] values, burst them all to maximize coins. Bursting i gives nums[left]*nums[i]*nums[right].
Think "last balloon to burst" in interval [l,r]. dp[l][r] = max coins from interval (l,r). Try each k as last burst: dp[l][r] = max(dp[l][k]+dp[k][r]+nums[l]*nums[k]*nums[r]).
C# 
public int MaxCoins(int[] nums) {
    int[] a=new int[nums.Length+2]; a[0]=a[^1]=1;
    for(int i=0;i<nums.Length;i++) a[i+1]=nums[i];
    int n=a.Length;
    int[,] dp=new int[n,n];
    for(int len=2;len<n;len++)
        for(int l=0;l<n-len;l++){
            int r=l+len;
            for(int k=l+1;k<r;k++)
                dp[l,r]=Math.Max(dp[l,r],dp[l,k]+a[l]*a[k]*a[r]+dp[k,r]);
        }
    return dp[0,n-1];
}
Python 
def maxCoins(nums):
    nums = [1] + nums + [1]  # sentinel boundaries
    n = len(nums)
    dp = [[0]*n for _ in range(n)]
    for length in range(2, n):
        for l in range(0, n - length):
            r = l + length
            for k in range(l+1, r):
                dp[l][r] = max(dp[l][r],
                    dp[l][k] + nums[l]*nums[k]*nums[r] + dp[k][r])
    return dp[0][n-1]

#14 · LeetCode 1245 — Tree Diameter
Medium
Pattern: Tree DP
Given a tree as edge list (not binary), find the diameter (longest path in edges).
DFS returns max depth from current node. At each node, sort children depths and use top 2 for diameter candidate. Answer = max over all nodes of (top1 + top2) depths.
C# 
public int TreeDiameter(int[][] edges) {
    var adj=new Dictionary<int,List<int>>();
    foreach(var e in edges){
        if(!adj.ContainsKey(e[0]))adj[e[0]]=new();
        if(!adj.ContainsKey(e[1]))adj[e[1]]=new();
        adj[e[0]].Add(e[1]); adj[e[1]].Add(e[0]);
    }
    int ans=0;
    int Dfs(int u, int par){
        var deps=adj.GetValueOrDefault(u,new())
            .Where(v=>v!=par).Select(v=>Dfs(v,u))
            .OrderDescending().Take(2).ToArray();
        ans=Math.Max(ans,deps.Sum());
        return 1+(deps.Length>0?deps[0]:0);
    }
    if(edges.Length>0) Dfs(edges[0][0],-1);
    return ans;
}
Python 
from collections import defaultdict
def treeDiameter(edges):
    adj = defaultdict(list)
    for u, v in edges: adj[u].append(v); adj[v].append(u)
    ans = [0]
    def dfs(u, parent):
        depths = [dfs(v, u) for v in adj[u] if v != parent]
        depths.sort(reverse=True)
        ans[0] = max(ans[0], sum(depths[:2]))
        return 1 + (depths[0] if depths else 0)
    if edges: dfs(edges[0][0], -1)
    return ans[0]

#16 · LeetCode 329 — Longest Increasing Path in Matrix
Hard
Pattern: DAG DP + memoization
Find the longest strictly increasing path in an m×n matrix (move 4 directions).
The implicit DAG has edges from smaller to larger values — no cycles. Use memoized DFS: dp[r][c] = longest path starting at (r,c). For each neighbor with larger value, dp[r][c] = 1 + max(dp[neighbor]).
C# 
public int LongestIncreasingPath(int[][] matrix) {
    int m=matrix.Length, n=matrix[0].Length;
    var memo=new int[m,n];
    int[] ds={0,1,0,-1,1,0,-1,0};
    int Dfs(int r, int c){
        if(memo[r,c]>0) return memo[r,c];
        int best=1;
        for(int d=0;d<4;d++){
            int nr=r+ds[d*2], nc=c+ds[d*2+1];
            if(nr>=0&&nr<m&&nc>=0&&nc<n&&matrix[nr][nc]>matrix[r][c])
                best=Math.Max(best,1+Dfs(nr,nc));
        }
        return memo[r,c]=best;
    }
    return Enumerable.Range(0,m)
        .SelectMany(r=>Enumerable.Range(0,n).Select(c=>Dfs(r,c))).Max();
}
Python 
from functools import lru_cache
def longestIncreasingPath(matrix):
    m, n = len(matrix), len(matrix[0])
    @lru_cache(None)
    def dfs(r, c):
        best = 1
        for dr, dc in ((0,1),(0,-1),(1,0),(-1,0)):
            nr, nc = r+dr, c+dc
            if 0 <= nr < m and 0 <= nc < n and matrix[nr][nc] > matrix[r][c]:
                best = max(best, 1 + dfs(nr, nc))
        return best
    return max(dfs(r, c) for r in range(m) for c in range(n))

#18 · LeetCode 647 — Palindromic Substrings
Medium
Pattern: Interval DP / expand-around-center
Count the number of palindromic substrings (contiguous, not subsequences).
Expand around each center (n odd centers + n-1 even centers). Count all expansions. O(n˛). Alternatively: dp[i][j]=True if s[i..j] is palindrome; s[i]==s[j] and dp[i+1][j-1] is True.
C# 
public int CountSubstrings(string s) {
    int count=0;
    for(int c=0;c<s.Length;c++)
        foreach(var(L0,R0) in new[]{ (c,c),(c,c+1) }){
            int L=L0, R=R0;
            while(L>=0&&R<s.Length&&s[L]==s[R]){count++;L--;R++;}
        }
    return count;
}
Python 
def countSubstrings(s):
    count = 0
    for c in range(len(s)):
        for L, R in ((c,c), (c,c+1)):  # odd and even centers
            while L >= 0 and R < len(s) and s[L] == s[R]:
                count += 1; L -= 1; R += 1
    return count
#19 · LeetCode 10 — Regular Expression Matching
Hard
Pattern: 2D DP
Implement regex matching with '.' (any char) and '*' (zero or more of preceding element).
dp[i][j] = does s[0..i-1] match p[0..j-1]. If p[j-1]!='*': dp[i][j] = dp[i-1][j-1] and match(s[i-1],p[j-1]). If p[j-1]=='*': dp[i][j] = dp[i][j-2] (zero uses) OR (dp[i-1][j] and match(s[i-1],p[j-2])) (one+ uses).
C# 
public bool IsMatch(string s, string p) {
    int m=s.Length, n=p.Length;
    bool[,] dp=new bool[m+1,n+1]; dp[0,0]=true;
    for(int j=2;j<=n;j++) if(p[j-1]=='*') dp[0,j]=dp[0,j-2];
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++){
            if(p[j-1]!='*') dp[i,j]=dp[i-1,j-1]&&(p[j-1]==s[i-1]||p[j-1]=='.');
            else dp[i,j]=dp[i,j-2]||(dp[i-1,j]&&(p[j-2]==s[i-1]||p[j-2]=='.'));
        }
    return dp[m,n];
}
Python 
def isMatch(s, p):
    m, n = len(s), len(p)
    dp = [[False]*(n+1) for _ in range(m+1)]
    dp[0][0] = True
    for j in range(2, n+1):
        if p[j-1] == '*': dp[0][j] = dp[0][j-2]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if p[j-1] != '*':
                dp[i][j] = dp[i-1][j-1] and (p[j-1] in (s[i-1], '.'))
            else:
                dp[i][j] = dp[i][j-2] or \
                    (dp[i-1][j] and p[j-2] in (s[i-1], '.'))
    return dp[m][n]
#20 · LeetCode 44 — Wildcard Matching
Hard
Pattern: 2D DP
Wildcard matching with '?' (any single char) and '*' (any sequence including empty).
Simpler than regex: dp[i][j] = does s[0..i-1] match p[0..j-1]. If p[j-1]=='*': dp[i][j] = dp[i][j-1] (skip *) OR dp[i-1][j] (use * for one more char). If '?' or match: dp[i][j]=dp[i-1][j-1].
C# 
public bool IsMatchWild(string s, string p) {
    int m=s.Length, n=p.Length;
    bool[,] dp=new bool[m+1,n+1]; dp[0,0]=true;
    for(int j=1;j<=n;j++) if(p[j-1]=='*') dp[0,j]=dp[0,j-1];
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++){
            if(p[j-1]=='*') dp[i,j]=dp[i,j-1]||dp[i-1,j];
            else dp[i,j]=dp[i-1,j-1]&&(p[j-1]==s[i-1]||p[j-1]=='?');
        }
    return dp[m,n];
}
Python
