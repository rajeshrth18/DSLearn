Easy Problems (1–4)
#1Two Sum
EasyHashMap

Given an array nums and a target integer, return the indices of two numbers that add up to target. Assume exactly one solution exists. You may not use the same element twice.

Examples:
  • [2,7,11,15], target=9 → [0,1]
  • [3,2,4], target=6 → [1,2]
Approach: One-pass hash map. For each element, check if its complement (target - num) is already in the map. If so, found. Otherwise store index.
public int[] TwoSum(int[] nums, int target)
{
    var map = new Dictionary<int, int>();
    for (int i = 0; i < nums.Length; i++)
    {
        if (map.TryGetValue(target - nums[i], out int j))
            return new[] { j, i };
        map[nums[i]] = i;
    }
    return Array.Empty<int>();
}
// Time: O(n), Space: O(n)
#2Contains Duplicate
EasyHashSet

Given an integer array, return true if any value appears at least twice, or false if every element is distinct.

Examples:
  • [1,2,3,1] → true
  • [1,2,3,4] → false
Approach: HashSet.Add returns false if element already exists.
public bool ContainsDuplicate(int[] nums)
{
    var seen = new HashSet<int>();
    foreach (int n in nums)
        if (!seen.Add(n)) return true;
    return false;
}
// Time: O(n), Space: O(n)
#3Valid Anagram
EasyFrequency Array

Given two strings s and t, return true if t is an anagram of s (same characters, same frequencies).

Examples:
  • "anagram", "nagaram" → true
  • "rat", "car" → false
Approach: Use int[26] frequency array — increment for s, decrement for t, then verify all zeros. O(n) time, O(1) space (fixed-size array).
public bool IsAnagram(string s, string t)
{
    if (s.Length != t.Length) return false;
    int[] count = new int[26];
    foreach (char c in s) count[c - 'a']++;
    foreach (char c in t) count[c - 'a']--;
    foreach (int n in count)
        if (n != 0) return false;
    return true;
}
// Time: O(n), Space: O(1) — array[26] is constant size
#4Word Pattern
EasyBijectionHashMap

Given a pattern string and a string s (space-separated words), determine if s follows the pattern. A bijection (one-to-one mapping both ways) must exist.

Examples:
  • pattern="abba", s="dog cat cat dog" → true
  • pattern="abba", s="dog cat cat fish" → false
  • pattern="aaaa", s="dog cat cat dog" → false
Approach: Two maps: char→word and word→char. A bijection requires both mappings to be consistent. If char already maps to a different word (or word maps to different char), return false.
public bool WordPattern(string pattern, string s)
{
    string[] words = s.Split(' ');
    if (pattern.Length != words.Length) return false;
    var charToWord = new Dictionary<char, string>();
    var wordToChar = new Dictionary<string, char>();
    for (int i = 0; i < pattern.Length; i++)
    {
        char ch = pattern[i]; string w = words[i];
        if (charToWord.TryGetValue(ch, out string? mapped) && mapped != w) return false;
        if (wordToChar.TryGetValue(w,  out char  rev)    && rev   != ch) return false;
        charToWord[ch] = w; wordToChar[w] = ch;
    }
    return true;
}
// Time: O(n), Space: O(n)
Medium Problems (5–10)
#5Group Anagrams
MediumHashMapGrouping

Given an array of strings, group the anagrams together. Return in any order.

Example: ["eat","tea","tan","ate","nat","bat"] → [["bat"],["nat","tan"],["ate","eat","tea"]]
Approach: Canonical key = sorted characters of each word. All anagrams produce the same sorted key → group them.
public IList<IList<string>> GroupAnagrams(string[] strs)
{
    var groups = new Dictionary<string, List<string>>();
    foreach (string s in strs)
    {
        char[] ch = s.ToCharArray();
        Array.Sort(ch);
        string key = new string(ch);
        if (!groups.ContainsKey(key)) groups[key] = new List<string>();
        groups[key].Add(s);
    }
    return new List<IList<string>>(groups.Values);
}
// Time: O(n·k·log k), Space: O(n·k)  where k = max string length
#6Top K Frequent Elements
MediumFrequencyBucket Sort

Given an integer array and integer k, return the k most frequent elements in any order. You must solve it in better than O(n log n) time.

Example: [1,1,1,2,2,3], k=2 → [1,2]
Approach: Frequency map + Bucket Sort. Bucket[i] holds all numbers with frequency i. Scan from high to low — O(n).
public int[] TopKFrequent(int[] nums, int k)
{
    var freq = new Dictionary<int, int>();
    foreach (int n in nums) freq[n] = freq.GetValueOrDefault(n) + 1;
    var buckets = new List<int>[nums.Length + 1];
    foreach (var (val, cnt) in freq)
    {
        buckets[cnt] ??= new List<int>();
        buckets[cnt].Add(val);
    }
    var result = new List<int>();
    for (int i = buckets.Length - 1; i >= 1 && result.Count < k; i--)
        if (buckets[i] != null) result.AddRange(buckets[i]);
    return result.Take(k).ToArray();
}
// Time: O(n), Space: O(n)
#7Subarray Sum Equals K
MediumPrefix SumHashMap

Given an integer array and an integer k, return the total number of subarrays whose sum equals k.

Examples:
  • [1,1,1], k=2 → 2
  • [1,2,3], k=3 → 2
Approach: Running prefix sum. Store how many times each prefix sum has been seen. For each position, check if (current_prefix - k) exists in the map — those are subarrays ending here with sum k.
public int SubarraySum(int[] nums, int k)
{
    var map = new Dictionary<int, int> { [0] = 1 };
    int prefix = 0, count = 0;
    foreach (int n in nums)
    {
        prefix += n;
        count += map.GetValueOrDefault(prefix - k);
        map[prefix] = map.GetValueOrDefault(prefix) + 1;
    }
    return count;
}
// Time: O(n), Space: O(n)
#8Longest Consecutive Sequence
MediumHashSet

Given an unsorted integer array, find the length of the longest consecutive elements sequence. Must run in O(n) time.

Examples:
  • [100,4,200,1,3,2] → 4 — sequence [1,2,3,4]
  • [0,3,7,2,5,8,4,6,0,1] → 9
Approach: Add all numbers to a HashSet. For each number n, only start counting if (n-1) is NOT in the set (meaning n is the start of a sequence). Then count consecutive n, n+1, n+2... — each element is processed at most twice overall = O(n).
public int LongestConsecutive(int[] nums)
{
    var set = new HashSet<int>(nums);
    int best = 0;
    foreach (int n in set)
    {
        if (set.Contains(n - 1)) continue; // only start of sequence
        int len = 1;
        while (set.Contains(n + len)) len++;
        best = Math.Max(best, len);
    }
    return best;
}
// [100,4,200,1,3,2] → 4
// Time: O(n), Space: O(n)
#9Ransom Note
MediumFrequency

Given two strings ransomNote and magazine, return true if the ransomNote can be constructed using letters from magazine. Each letter may only be used once.

Examples:
  • "a", "b" → false
  • "aa", "aab" → true
Approach: Count magazine letter frequencies. Decrement for each ransom note letter. If any count goes negative, return false.
public bool CanConstruct(string ransomNote, string magazine)
{
    int[] count = new int[26];
    foreach (char c in magazine) count[c - 'a']++;
    foreach (char c in ransomNote)
    {
        if (--count[c - 'a'] < 0) return false;
    }
    return true;
}
// Time: O(m+r), Space: O(1) — fixed-size array
#10Minimum Window Substring
MediumSliding WindowHashMap

Given strings s and t, return the minimum window substring of s that contains all characters in t (including duplicates). Return "" if no such window exists.

Example: s="ADOBECODEBANC", t="ABC" → "BANC"
Approach: Sliding window with two hash maps (need, window). Track how many distinct characters are satisfied. Expand right, contract left when all satisfied. Track minimum window.
public string MinWindow(string s, string t)
{
    if (s.Length == 0 || t.Length == 0) return "";
    var need = new Dictionary<char, int>();
    foreach (char c in t) need[c] = need.GetValueOrDefault(c) + 1;
    var window = new Dictionary<char, int>();
    int formed = 0, required = need.Count;
    int l = 0, minLen = int.MaxValue, minL = 0;
    for (int r = 0; r < s.Length; r++)
    {
        char c = s[r];
        window[c] = window.GetValueOrDefault(c) + 1;
        if (need.ContainsKey(c) && window[c] == need[c]) formed++;
        while (formed == required)
        {
            if (r - l + 1 < minLen) { minLen = r - l + 1; minL = l; }
            char left = s[l++];
            window[left]--;
            if (need.ContainsKey(left) && window[left] < need[left]) formed--;
        }
    }
    return minLen == int.MaxValue ? "" : s.Substring(minL, minLen);
}
// "ADOBECODEBANC","ABC" → "BANC"
// Time: O(|s| + |t|), Space: O(|t|)
Hard Problems (11–12)
#11LRU Cache
HardDesignDLL + HashMap

Design a data structure that follows LRU (Least Recently Used) cache eviction policy. Implement LRUCache(capacity), Get(key) (returns -1 if absent), and Put(key, value) (evicts LRU if capacity exceeded). Both operations must be O(1).

Approach: Dictionary for O(1) key → node lookup. Doubly Linked List (with dummy head/tail) for O(1) recency ordering. Most recently used = front; LRU = back.
public class LRUCache
{
    private class Node
    {
        public int Key, Val;
        public Node? Prev, Next;
        public Node(int k, int v) { Key = k; Val = v; }
    }
    private readonly int _cap;
    private readonly Dictionary<int, Node> _map;
    private readonly Node _head, _tail;
    public LRUCache(int capacity)
    {
        _cap = capacity; _map = new(capacity);
        _head = new(-1,-1); _tail = new(-1,-1);
        _head.Next = _tail; _tail.Prev = _head;
    }
    public int Get(int key)
    {
        if (!_map.TryGetValue(key, out var n)) return -1;
        MoveToFront(n); return n.Val;
    }
    public void Put(int key, int value)
    {
        if (_map.TryGetValue(key, out var n)) { n.Val = value; MoveToFront(n); return; }
        if (_map.Count == _cap) { var lru = _tail.Prev!; Unlink(lru); _map.Remove(lru.Key); }
        var node = new Node(key, value); InsertFront(node); _map[key] = node;
    }
    private void MoveToFront(Node n) { Unlink(n); InsertFront(n); }
    private void Unlink(Node n) { n.Prev!.Next = n.Next; n.Next!.Prev = n.Prev; }
    private void InsertFront(Node n)
    { n.Next = _head.Next; n.Prev = _head; _head.Next!.Prev = n; _head.Next = n; }
}
// All operations: O(1) time
#12Find All Anagrams in a String
HardSliding WindowRolling Frequency

Given strings s and p, return an array of all start indices of anagrams of p in s.

Examples:
  • s="cbaebabacd", p="abc" → [0,6]
  • s="abab", p="ab" → [0,1,2]
Approach: Fixed-size sliding window of length |p|. Maintain a rolling character frequency array for the window. Slide right: add new char, remove old char. Compare with pattern frequency.
public IList<int> FindAnagrams(string s, string p)
{
    var result = new List<int>();
    if (s.Length < p.Length) return result;
    int[] need = new int[26], win = new int[26];
    int k = p.Length;
    foreach (char c in p) need[c - 'a']++;
    for (int i = 0; i < s.Length; i++)
    {
        win[s[i] - 'a']++;                          // add right
        if (i >= k) win[s[i - k] - 'a']--;          // remove left
        if (i >= k - 1 && win.SequenceEqual(need))   // window is anagram?
            result.Add(i - k + 1);
    }
    return result;
}
// "cbaebabacd","abc" → [0, 6]
// Time: O(n), Space: O(1) — fixed 26-element arrays
