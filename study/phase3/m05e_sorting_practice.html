Sort Colors (Dutch National Flag) — Given array with values 0, 1, 2 only, sort in-place in a single pass without using sort().
Easy
🧠 Approach: 3-pointer (low, mid, high). O(n) time, O(1) space.
Approach: Use 3 pointers (low, mid, high) to partition in-place. Elements before low = 0, between low and mid = 1, after high = 2. Swap 0s left, 2s right, advance mid for 1s — all in a single O(n) pass.
Time: O(n) — single pass with 3 pointers &nbsp;|&nbsp; Space: O(1) — in-place, no extra memory
using System;

// LeetCode 75 — Sort Colors
// Constraint: single pass, in-place

class SortColors
{
    static void Sort(int[] nums)
    {
        int low = 0, mid = 0, high = nums.Length - 1;

        while (mid <= high)
        {
            if      (nums[mid] == 0) (nums[low++], nums[mid++]) = (nums[mid], nums[low]);
            else if (nums[mid] == 1) mid++;
            else                   (nums[mid], nums[high--]) = (nums[high], nums[mid]);
        }
    }

    static void Main()
    {
        int[] arr = { 2,0,2,1,1,0 };
        Sort(arr);
        Console.WriteLine(string.Join(", ", arr)); // 0,0,1,1,2,2
    }
}
from typing import List

def sort_colors(nums: List[int]) -> None:
    low, mid, high = 0, 0, len(nums) - 1
    while mid <= high:
        if   nums[mid] == 0: nums[low], nums[mid] = nums[mid], nums[low]; low+=1; mid+=1
        elif nums[mid] == 1: mid+=1
        else: nums[mid], nums[high] = nums[high], nums[mid]; high-=1

arr = [2,0,2,1,1,0]
sort_colors(arr)
print(arr)  # [0,0,1,1,2,2]

Kth Largest Element — Find the kth largest element in an unsorted array without fully sorting it.
Easy
🧠 Approach: Quickselect — partition-based O(n) average. Or min-heap of size k: O(n log k).
Approach: Quickselect: randomly choose pivot, partition so pivot lands in final sorted position. The kth largest is the (n-k)th smallest. Recurse only on the half containing the target index — expected O(n) with random pivot.
Time: O(n) average, O(n²) worst &mdash; Quickselect with random pivot &nbsp;|&nbsp; Space: O(log n) — recursion stack depth (log n levels)
using System;

// LeetCode 215 — Kth Largest Element
// Quickselect: partition then recurse only on the side containing k

class KthLargest
{
    static Random _rng = new Random();

    static int FindKthLargest(int[] nums, int k)
    {
        // kth largest = (n-k)th smallest (0-indexed)
        return Quickselect(nums, 0, nums.Length-1, nums.Length-k);
    }

    static int Quickselect(int[] arr, int lo, int hi, int targetIdx)
    {
        if (lo == hi) return arr[lo];
        int pivotIdx = Partition(arr, lo, hi);
        if      (pivotIdx == targetIdx) return arr[pivotIdx];
        else if (pivotIdx < targetIdx)  return Quickselect(arr, pivotIdx+1, hi, targetIdx);
        else                             return Quickselect(arr, lo, pivotIdx-1, targetIdx);
    }

    static int Partition(int[] arr, int lo, int hi)
    {
        int ri = _rng.Next(lo, hi+1);
        (arr[ri], arr[hi]) = (arr[hi], arr[ri]);
        int pivot = arr[hi], i = lo-1;
        for(int j=lo;j<hi;j++) if(arr[j]<=pivot)(arr[++i],arr[j])=(arr[j],arr[i]);
        (arr[i+1],arr[hi])=(arr[hi],arr[i+1]);
        return i+1;
    }

    static void Main()
    {
        Console.WriteLine(FindKthLargest(new[]{3,2,1,5,6,4}, 2)); // 5
        Console.WriteLine(FindKthLargest(new[]{3,2,3,1,2,4,5,5,6},4)); // 4
    }
}
import random
from typing import List

def find_kth_largest(nums: List[int], k: int) -> int:
    target = len(nums) - k  # kth largest = (n-k)th smallest
    def quickselect(lo, hi):
        ri = random.randint(lo, hi)
        nums[ri], nums[hi] = nums[hi], nums[ri]
        pivot, i = nums[hi], lo - 1
        for j in range(lo, hi):
            if nums[j] <= pivot: i+=1; nums[i],nums[j]=nums[j],nums[i]
        nums[i+1],nums[hi]=nums[hi],nums[i+1]
        pi = i+1
        if   pi == target: return nums[pi]
        elif pi <  target: return quickselect(pi+1, hi)
        else:              return quickselect(lo, pi-1)
    return quickselect(0, len(nums)-1)

print(find_kth_largest([3,2,1,5,6,4], 2))              # 5
print(find_kth_largest([3,2,3,1,2,4,5,5,6], 4))   # 4

Meeting Rooms — Given meeting intervals [[s1,e1],[s2,e2],...], determine if a person can attend all meetings (no overlaps).
Easy
🧠 Approach: Sort by start time. Check if any meeting starts before previous one ends.
Approach: Sort meetings by start time. Then do a linear scan: if any meeting starts before the previous one ends (curr.start < prev.end), there's an overlap and the person cannot attend all meetings.
Time: O(n log n) — sorting dominates &nbsp;|&nbsp; Space: O(1) — in-place sort comparison
using System;

class MeetingRooms
{
    static bool CanAttendAll(int[][] intervals)
    {
        // Sort by start time
        Array.Sort(intervals, (a, b) => a[0].CompareTo(b[0]));

        for (int i = 1; i < intervals.Length; i++)
        {
            // If current meeting starts before previous one ends ? conflict
            if (intervals[i][0] < intervals[i-1][1]) return false;
        }
        return true;
    }

    static void Main()
    {
        Console.WriteLine(CanAttendAll(new[]{ new[]{0,30},new[]{5,10},new[]{15,20} })); // False
        Console.WriteLine(CanAttendAll(new[]{ new[]{7,10},new[]{2,4} }));            // True
    }
}
from typing import List

def can_attend_all(intervals: List[List[int]]) -> bool:
    intervals.sort(key=lambda x: x[0])
    for i in range(1, len(intervals)):
        if intervals[i][0] < intervals[i-1][1]:
            return False
    return True

print(can_attend_all([[0,30],[5,10],[15,20]]))  # False
print(can_attend_all([[7,10],[2,4]]))            # True

Valid Anagram — Given two strings, determine if one is an anagram of the other using sorting.
Easy
🧠 Approach: Sort both strings, compare. O(n log n). Also solvable with counting sort O(n).
Approach: Sort both strings and compare (O(n log n)). Or faster: use a 26-element frequency array, increment for chars in s, decrement for chars in t. If all frequencies are zero, strings are anagrams.
Time: O(n log n) sort, or O(n) with counting array &nbsp;|&nbsp; Space: O(1) — 26-element fixed-size array
using System;

class ValidAnagram
{
    // Approach 1: Sort O(n log n)
    static bool IsAnagramSort(string s, string t) {
        if (s.Length != t.Length) return false;
        char[] a = s.ToCharArray(), b = t.ToCharArray();
        Array.Sort(a); Array.Sort(b);
        return new string(a) == new string(b);
    }

    // Approach 2: Counting sort O(n)
    static bool IsAnagramCount(string s, string t) {
        if (s.Length != t.Length) return false;
        int[] count = new int[26];
        for(int i=0;i<s.Length;i++) { count[s[i]-'a']++; count[t[i]-'a']--; }
        foreach(int c in count) if(c!=0) return false;
        return true;
    }

    static void Main() {
        Console.WriteLine(IsAnagramCount("anagram", "nagaram")); // True
        Console.WriteLine(IsAnagramCount("rat", "car"));         // False
    }
}

Merge Intervals — Given a list of intervals, merge all overlapping intervals.
Easy
🧠 Approach: Sort by start. Merge current with previous if overlapping.
Approach: Sort intervals by start time. Initialize result with the first interval. For each subsequent interval: if it overlaps the last result interval (curr.start ≤ last.end), extend the end; otherwise append as a new interval.
Time: O(n log n) — sorting dominates; merge scan is O(n) &nbsp;|&nbsp; Space: O(n) — output list of merged intervals
using System;
using System.Collections.Generic;

class MergeIntervals
{
    static int[][] Merge(int[][] intervals)
    {
        Array.Sort(intervals, (a,b)=>a[0].CompareTo(b[0]));
        var res = new List<int[]>();
        res.Add(intervals[0]);
        foreach(var iv in intervals) {
            var last = res[res.Count-1];
            if(iv[0] <= last[1]) last[1] = Math.Max(last[1], iv[1]);
            else res.Add(iv);
        }
        return res.ToArray();
    }

    static void Main() {
        var merged = Merge(new[]{new[]{1,3},new[]{2,6},new[]{8,10},new[]{15,18}});
        foreach(var iv in merged) Console.Write($"[{iv[0]},{iv[1]}] ");
        // [1,6] [8,10] [15,18]
    }
}
from typing import List

def merge_intervals(intervals: List[List[int]]) -> List[List[int]]:
    intervals.sort(key=lambda x: x[0])
    merged = [intervals[0]]
    for start, end in intervals[1:]:
        if start <= merged[-1][1]: merged[-1][1] = max(merged[-1][1], end)
        else: merged.append([start, end])
    return merged

print(merge_intervals([[1,3],[2,6],[8,10],[15,18]]))  # [[1,6],[8,10],[15,18]]

Largest Number — Arrange an array of non-negative integers to form the largest possible number.
Medium
🧠 Approach: Custom comparator — compare ab vs ba string concatenations. O(n log n).
Approach: Custom comparator: for two numbers a and b, compare string concatenations ab vs ba. If "ab" > "ba", a should come first. Sort all numbers using this comparator descending. Edge case: if result starts with 0 (all zeros), return "0".
Time: O(n log n × L) — sort with string comparator; L = avg digit length &nbsp;|&nbsp; Space: O(n × L) — string output list
using System;

class LargestNumber
{
    static string Solve(int[] nums) {
        string[] s = Array.ConvertAll(nums, n => n.ToString());
        Array.Sort(s, (a, b) => string.Compare(b+a, a+b, StringComparison.Ordinal));
        string result = string.Concat(s);
        return result[0] == '0' ? "0" : result;
    }
    static void Main() {
        Console.WriteLine(Solve(new[]{3,30,34,5,9}));  // 9534330
        Console.WriteLine(Solve(new[]{10,2}));            // 210
    }
}
from functools import cmp_to_key
from typing import List

def largest_number(nums: List[int]) -> str:
    s = list(map(str, nums))
    s.sort(key=cmp_to_key(lambda a,b: (1 if a+b > b+a else -1)), reverse=False)
    # Simpler:
    s.sort(key=cmp_to_key(lambda a,b: -1 if a+b>b+a else (1 if a+b<b+a else 0)))
    result = "".join(s)
    return "0" if result[0]=="0" else result

print(largest_number([3,30,34,5,9]))  # 9534330

H-Index — A researcher has citations[i] for ith paper. Find the H-Index: maximum h such that h papers have at least h citations.
Medium
🧠 Approach: Sort descending. Scan until citations[i] < i+1. O(n log n).
Approach: Sort citations in descending order. At index i (0-based), at least i+1 papers have citations[i] or more citations. The h-index is the largest i+1 where citations[i] ≥ i+1 — stop scanning when citations[i] < i+1.
Time: O(n log n) — sorting &nbsp;|&nbsp; Space: O(1) — linear scan with one variable
using System;

class HIndex
{
    static int Solve(int[] citations) {
        Array.Sort(citations, (a,b) => b.CompareTo(a)); // sort descending
        int h = 0;
        while (h < citations.Length && citations[h] > h) h++;
        return h;
    }
    static void Main() {
        Console.WriteLine(Solve(new[]{3,0,6,1,5}));  // 3
        Console.WriteLine(Solve(new[]{1,3,1}));         // 1
    }
}
def h_index(citations):
    citations.sort(reverse=True)
    h = 0
    while h < len(citations) and citations[h] > h:
        h += 1
    return h

print(h_index([3,0,6,1,5]))  # 3
print(h_index([1,3,1]))         # 1

Find All Duplicates — Given an integer array where each element is between 1 and n, find all elements that appear twice. Use O(1) extra space.
Medium
🧠 Approach: Use array indices as a "visited" marker by negating values. O(n) time, O(1) space.
Approach: Use array indices as a hash set without extra space. For each value v, negate arr[|v|-1] (use absolute value since values may already be negative). After marking, scan for positive values: their (index+1) is a missing number.
Time: O(n) — two linear passes over the array &nbsp;|&nbsp; Space: O(1) — negation trick uses no extra space
using System;
using System.Collections.Generic;

class FindDuplicates
{
    static IList<int> Solve(int[] nums) {
        var res = new List<int>();
        foreach(int n in nums) {
            int idx = Math.Abs(n) - 1;
            if(nums[idx] < 0) res.Add(Math.Abs(n)); // seen before ? duplicate
            else nums[idx] = -nums[idx];                // mark as seen
        }
        return res;
    }
    static void Main() {
        var r = Solve(new[]{4,3,2,7,8,2,3,1});
        Console.WriteLine(string.Join(", ",r)); // 2,3
    }
}
from typing import List

def find_duplicates(nums: List[int]) -> List[int]:
    res = []
    for n in nums:
        idx = abs(n) - 1
        if nums[idx] < 0: res.append(abs(n))
        else: nums[idx] = -nums[idx]
    return res

print(find_duplicates([4,3,2,7,8,2,3,1]))  # [2, 3]

Sort Array By Parity — Move all even numbers before odd numbers. The order within each group doesn't matter.
Medium
🧠 Approach: Two-pointer in-place O(n). Or stable version: sort by parity key.
Approach: Two pointers in-place: left pointer scans from start (looking for odd number), right pointer from end (looking for even number). Swap when left is odd and right is even, then advance both. Continue until pointers meet.
Time: O(n) — each element examined at most once &nbsp;|&nbsp; Space: O(1) — in-place swaps only
using System;

class SortParity
{
    static int[] Solve(int[] arr) {
        int l=0, r=arr.Length-1;
        while(l<r) {
            if(arr[l]%2!=0 && arr[r]%2==0) (arr[l],arr[r])=(arr[r],arr[l]);
            if(arr[l]%2==0) l++;
            if(arr[r]%2!=0) r--;
        }
        return arr;
    }
    static void Main() {
        Console.WriteLine(string.Join(", ",Solve(new[]{3,1,2,4}))); // 4,2,1,3 (or similar)
    }
}
def sort_array_by_parity(arr):
    # One-liner using key (stable, O(n log n))
    return sorted(arr, key=lambda x: x % 2)  # even=0 first, odd=1 second

print(sort_array_by_parity([3,1,2,4]))  # [2,4,3,1]

Non-overlapping Intervals — Given intervals, find the minimum number to remove to make the rest non-overlapping.
Medium
🧠 Approach: Greedy — sort by end time. Keep interval if it doesn't overlap. Count removals.
Approach: Greedy: sort intervals by end time. Keep an interval if it does not overlap with the last kept interval. When two intervals overlap, discard the one with the later end time (keep the one ending sooner — it leaves maximum room for future intervals).
Time: O(n log n) — sort by end time; scan is O(n) &nbsp;|&nbsp; Space: O(1) — scan with one variable tracking last kept end
using System;

class NonOverlapping
{
    static int EraseOverlapIntervals(int[][] intervals) {
        Array.Sort(intervals, (a,b) => a[1].CompareTo(b[1])); // sort by END
        int count=0, prevEnd=int.MinValue;
        foreach(var iv in intervals) {
            if(iv[0] >= prevEnd) prevEnd=iv[1]; // keep: no overlap
            else count++;                            // remove: overlaps
        }
        return count;
    }
    static void Main() {
        Console.WriteLine(EraseOverlapIntervals(new[]{new[]{1,2},new[]{2,3},new[]{3,4},new[]{1,3}})); // 1
        Console.WriteLine(EraseOverlapIntervals(new[]{new[]{1,2},new[]{1,2},new[]{1,2}}));         // 2
    }
}
def erase_overlap_intervals(intervals):
    intervals.sort(key=lambda x: x[1])
    count, prev_end = 0, float('-inf')
    for start, end in intervals:
        if start >= prev_end: prev_end = end
        else: count += 1
    return count

print(erase_overlap_intervals([[1,2],[2,3],[3,4],[1,3]]))  # 1

Sort Characters by Frequency — Given a string, sort its characters in decreasing order based on the frequency of characters.
Medium
🧠 Approach: Count frequencies, sort by count descending (bucket sort for O(n)).
Approach: Count element frequencies using a HashMap. Then use bucket sort: create n+1 buckets indexed by frequency; add each element to its frequency bucket. Scan buckets from highest to lowest to collect the top k elements.
Time: O(n) — HashMap counting + bucket sort &nbsp;|&nbsp; Space: O(n) — frequency map + n+1 bucket array
using System;
using System.Collections.Generic;
using System.Linq;

class FreqSort
{
    static string Solve(string s) {
        var freq = new Dictionary<char,int>();
        foreach(char c in s) { freq[c] = freq.GetValueOrDefault(c) + 1; }
        return string.Concat(freq.OrderByDescending(kv=>kv.Value)
            .Select(kv=>new string(kv.Key,kv.Value)));
    }
    static void Main() {
        Console.WriteLine(Solve("tree"));    // "eert" or "eetr"
        Console.WriteLine(Solve("cccaaa")); // "cccaaa" or "aaaccc"
    }
}
from collections import Counter

def frequency_sort(s: str) -> str:
    freq = Counter(s)
    return "".join(c * n for c, n in freq.most_common())

print(frequency_sort("tree"))    # "eert"
print(frequency_sort("cccaaa")) # "cccaaa"

Count Smaller Numbers After Self — For each number in nums, count how many smaller numbers appear to its right.
Hard
🧠 Approach: Modified merge sort — count during merge phase. O(n log n).
Approach: Modified merge sort with index tracking. Sort an index array by value. During each merge step: when a right-side element is placed before remaining left-side elements, add the count of remaining left elements to that right element's answer (each is an inversion).
Time: O(n log n) — merge sort with O(1) extra work per comparison &nbsp;|&nbsp; Space: O(n) — index array + temporary merge arrays
using System;
using System.Collections.Generic;

class CountSmaller
{
    static int[] _counts;
    static int[] _indices;

    static IList<int> Solve(int[] nums) {
        int n = nums.Length;
        _counts = new int[n];
        _indices = new int[n];
        for(int i=0;i<n;i++) _indices[i]=i;
        MergeSort(nums, 0, n-1);
        return _counts;
    }

    static void MergeSort(int[] nums, int lo, int hi) {
        if(lo>=hi) return;
        int mid=(lo+hi)/2;
        MergeSort(nums,lo,mid); MergeSort(nums,mid+1,hi);
        Merge(nums,lo,mid,hi);
    }

    static void Merge(int[] nums, int lo, int mid, int hi) {
        var tmp = new int[hi-lo+1]; int i=lo,j=mid+1,k=0;
        while(i<=mid && j<=hi) {
            if(nums[_indices[i]] > nums[_indices[j]]) {
                _counts[_indices[i]] += hi-j+1; // all remaining right elements are smaller
                tmp[k++]=_indices[i++];
            } else tmp[k++]=_indices[j++];
        }
        while(i<=mid) tmp[k++]=_indices[i++];
        while(j<=hi) tmp[k++]=_indices[j++];
        for(int x=0;x<tmp.Length;x++) _indices[lo+x]=tmp[x];
    }

    static void Main() {
        var r = Solve(new[]{5,2,6,1});
        Console.WriteLine(string.Join(", ",r)); // 2, 1, 1, 0
    }
}
from typing import List

def count_smaller(nums: List[int]) -> List[int]:
    n = len(nums)
    counts = [0] * n
    indices = list(range(n))

    def merge_sort(idx_list):
        if len(idx_list) <= 1: return idx_list
        mid = len(idx_list) // 2
        L = merge_sort(idx_list[:mid])
        R = merge_sort(idx_list[mid:])
        merged, i, j = [], 0, 0
        while i < len(L) and j < len(R):
            if nums[L[i]] > nums[R[j]]:
                counts[L[i]] += len(R) - j  # all remaining R are smaller
                merged.append(L[i]); i += 1
            else: merged.append(R[j]); j += 1
        merged.extend(L[i:]); merged.extend(R[j:])
        return merged

    merge_sort(indices)
    return counts

print(count_smaller([5,2,6,1]))  # [2,1,1,0]

Maximum Gap — Given an unsorted array, find the maximum difference between successive elements in its sorted form. Must run in O(n) time and space.
Hard
🧠 Approach: Bucket sort + pigeonhole principle. Gap must come from different buckets.
Approach: Pigeonhole principle: n numbers spanning [min, max] have n-1 gaps; the max gap is ≥ (max-min)/(n-1). Create buckets of that minimum gap size. The maximum gap must span across adjacent non-empty bucket boundaries — check only consecutive non-empty buckets.
Time: O(n) — bucket sort; each element placed once &nbsp;|&nbsp; Space: O(n) — n-1 buckets each storing (min, max) pair
import math
from typing import List

def maximum_gap(nums: List[int]) -> int:
    if len(nums) < 2: return 0
    lo, hi = min(nums), max(nums)
    if lo == hi: return 0
    n = len(nums)
    # Bucket width (by pigeonhole, max gap >= ceil((hi-lo)/(n-1)))
    bw = max(1, math.ceil((hi - lo) / (n - 1)))
    nb = (hi - lo) // bw + 1
    # Each bucket stores (min, max) of elements it contains
    buckets = [[float('inf'), float('-inf')] for _ in range(nb)]
    for x in nums:
        bi = (x - lo) // bw
        buckets[bi][0] = min(buckets[bi][0], x)
        buckets[bi][1] = max(buckets[bi][1], x)
    # Max gap must come between buckets (elements in same bucket have gap < bw)
    max_gap, prev_max = 0, lo
    for bmin, bmax in buckets:
        if bmin == float('inf'): continue   # empty bucket
        max_gap = max(max_gap, bmin - prev_max)
        prev_max = bmax
    return max_gap

print(maximum_gap([3,6,9,1]))   # 3
print(maximum_gap([10]))          # 0
print(maximum_gap([1,10000000])) # 9999999
using System;

class MaximumGap
{
    static int Solve(int[] nums) {
        int n=nums.Length; if(n<2) return 0;
        int lo=int.MaxValue, hi=int.MinValue;
        foreach(int x in nums){if(x<lo)lo=x;if(x>hi)hi=x;}
        if(lo==hi) return 0;
        int bw=Math.Max(1,(hi-lo+n-2)/(n-1)), nb=(hi-lo)/bw+1;
        int[] bMin=new int[nb], bMax=new int[nb];
        Array.Fill(bMin,int.MaxValue); Array.Fill(bMax,int.MinValue);
        foreach(int x in nums){int bi=(x-lo)/bw;bMin[bi]=Math.Min(bMin[bi],x);bMax[bi]=Math.Max(bMax[bi],x);}
        int gap=0,prevMax=lo;
        for(int i=0;i<nb;i++){if(bMin[i]==int.MaxValue)continue;gap=Math.Max(gap,bMin[i]-prevMax);prevMax=bMax[i];}
        return gap;
    }
    static void Main() {
        Console.WriteLine(Solve(new[]{3,6,9,1}));  // 3
    }
}

Median of Two Sorted Arrays — Find the median of two sorted arrays in O(log(min(m,n))) time.
Hard
🧠 Approach: Binary search on partition of smaller array.
Approach: Binary search on a partition of the smaller array. Find split i of A and j=(m+n+1)/2-i of B such that A[i-1] ≤ B[j] and B[j-1] ≤ A[i]. This ensures the left half has the correct elements. Compute median from the four boundary values.
Time: O(log(min(m,n))) — binary search on shorter array &nbsp;|&nbsp; Space: O(1) — just partition index variables
using System;

class MedianTwoSortedArrays
{
    static double FindMedianSortedArrays(int[] nums1, int[] nums2)
    {
        // Ensure nums1 is the smaller array for binary search
        if (nums1.Length > nums2.Length) return FindMedianSortedArrays(nums2, nums1);
        int m=nums1.Length, n=nums2.Length, half=(m+n+1)/2;
        int lo=0, hi=m;
        while(lo<=hi) {
            int i=(lo+hi)/2, j=half-i;
            int maxL1=(i==0?int.MinValue:nums1[i-1]), minR1=(i==m?int.MaxValue:nums1[i]);
            int maxL2=(j==0?int.MinValue:nums2[j-1]), minR2=(j==n?int.MaxValue:nums2[j]);
            if(maxL1<=minR2 && maxL2<=minR1) {
                if((m+n)%2==1) return Math.Max(maxL1,maxL2);
                return (Math.Max(maxL1,maxL2)+Math.Min(minR1,minR2))/2.0;
            } else if(maxL1>minR2) hi=i-1; else lo=i+1;
        }
        return 0;
    }
    static void Main() {
        Console.WriteLine(FindMedianSortedArrays(new[]{1,3},new[]{2}));       // 2.0
        Console.WriteLine(FindMedianSortedArrays(new[]{1,2},new[]{3,4}));   // 2.5
    }
}
from typing import List

def find_median_sorted_arrays(nums1: List[int], nums2: List[int]) -> float:
    if len(nums1) > len(nums2): nums1, nums2 = nums2, nums1
    m, n = len(nums1), len(nums2)
    half = (m + n + 1) // 2
    lo, hi = 0, m
    while lo <= hi:
        i = (lo + hi) // 2; j = half - i
        maxL1 = nums1[i-1] if i > 0 else float('-inf')
        minR1 = nums1[i]   if i < m else float('inf')
        maxL2 = nums2[j-1] if j > 0 else float('-inf')
        minR2 = nums2[j]   if j < n else float('inf')
        if maxL1 <= minR2 and maxL2 <= minR1:
            if (m+n) % 2: return max(maxL1,maxL2)
            return (max(maxL1,maxL2)+min(minR1,minR2)) / 2
        elif maxL1 > minR2: hi = i - 1
        else: lo = i + 1

Wiggle Sort II — Given unsorted array, reorder such that nums[0] < nums[1] > nums[2] < nums[3]...
Hard
🧠 Approach: Find median (Quickselect), then 3-way partition + virtual indexing trick.
Approach: Find the median using Quickselect (O(n)). Apply 3-way Dutch National Flag partition around the median. Use virtual index mapping (odd indices: n-2, n-4,... then even: n-1, n-3,...) to interleave small and large values in-place without needing a separate array.
Time: O(n) — Quickselect for median + 3-way partition &nbsp;|&nbsp; Space: O(1) — in-place with virtual index mapping
using System;

class WiggleSortII
{
    static void WiggleSort(int[] nums) {
        int[] sorted = (int[])nums.Clone();
        Array.Sort(sorted);
        int n=nums.Length, lo=(n-1)/2, hi=n-1;
        // Place larger half at odd indices (descending), smaller half at even (descending)
        for(int i=0;i<n;i++) {
            nums[i] = (i%2==0) ? sorted[lo--] : sorted[hi--];
        }
    }
    static void Main() {
        int[] a1={1,5,1,1,6,4}; WiggleSort(a1);
        Console.WriteLine(string.Join(",",a1)); // e.g. 1,6,1,5,1,4
        int[] a2={4,5,5,6}; WiggleSort(a2);
        Console.WriteLine(string.Join(",",a2)); // e.g. 4,6,5,5 ? verify pattern
    }
}
from typing import List

def wiggle_sort(nums: List[int]) -> None:
    n = len(nums)
    s = sorted(nums)
    lo = (n-1) // 2   # median index
    hi = n - 1
    # Fill even indices with smaller half (descending), odd with larger half (descending)
    for i in range(n):
        if i % 2 == 0: nums[i] = s[lo]; lo -= 1
        else:           nums[i] = s[hi]; hi -= 1
