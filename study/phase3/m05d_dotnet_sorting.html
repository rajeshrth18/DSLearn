Section 1 — C# Array.Sort() and List<T>.Sort()

1.1 Basic Sorting

The simplest way to sort in C# is to call Array.Sort() or list.Sort(). These use the introspective sort internally — a hybrid of quick sort, heap sort, and insertion sort that guarantees O(n log n) worst-case.

using System;
using System.Collections.Generic;
using System.Linq;

class BasicSortingApis
{
    static void Main()
    {
        // -- Array.Sort (in-place, introspective sort) --
        int[] nums = { 5, 2, 8, 1, 9, 3 };
        Array.Sort(nums);
        Console.WriteLine("Array.Sort: " + string.Join(", ", nums)); // 1,2,3,5,8,9

        // -- Array.Sort Descending (with Comparison lambda) --
        Array.Sort(nums, (a, b) => b.CompareTo(a));
        Console.WriteLine("Descending: " + string.Join(", ", nums)); // 9,8,5,3,2,1

        // -- Array.Sort sub-range (sort only part of array) --
        int[] arr = { 9,7,5,3,1,2,4,6,8 };
        Array.Sort(arr, 2, 4);  // sort 4 elements starting at index 2
        Console.WriteLine("Sub-range: " + string.Join(", ", arr));
        // 9,7,[1,2,3,5],4,6,8 — only indices 2-5 sorted

        // -- List<T>.Sort --
        List<string> names = new List<string> { "Charlie", "Alice", "Bob", "Dave" };
        names.Sort();
        Console.WriteLine("Names: " + string.Join(", ", names)); // Alice,Bob,Charlie,Dave

        // Sort by string length, then alphabetically
        names.Sort((a, b) => {
            int lenCmp = a.Length.CompareTo(b.Length);
            return lenCmp != 0 ? lenCmp : string.Compare(a, b, StringComparison.Ordinal);
        });
        Console.WriteLine("By length: " + string.Join(", ", names)); // Bob,Alice,Dave,Charlie
    }
}
# --- Python Sorting Basics ----------------------------------------------------

print("sorted():", result)   # [1, 2, 3, 5, 8, 9]

print("After .sort():", nums)  # [1, 2, 3, 5, 8, 9]

# Reverse sort

print("Descending:", nums)  # [9, 8, 5, 3, 2, 1]

# Strings sort lexicographically by default
names = ["Charlie", "Alice", "Bob", "Dave"]
names.sort()
print("Names:", names)  # ['Alice', 'Bob', 'Charlie', 'Dave']

# Sort by length, then alphabetically
names.sort(key=lambda x: (len(x), x))
print("By length:", names)  # ['Bob', 'Alice', 'Dave', 'Charlie']

Section 2 — IComparable<T> — Natural Ordering

When your class has a natural ordering (e.g., students by GPA, products by price), implement IComparable<T>. This defines the default sort order — what happens when you call Sort() without any extra arguments.

using System;
using System.Collections.Generic;

// --- IComparable<T> — Natural Order ------------------------------------------

class Student : IComparable<Student>
{
    public string Name  { get; set; }
    public double GPA   { get; set; }
    public int    Year  { get; set; }

    /// Natural order: sort by GPA descending (highest first)
    /// If GPAs are equal, sort by Name ascending
    public int CompareTo(Student other)
    {
        if (other == null) return 1;  // this > null

        // Primary: GPA descending (note: other.GPA.CompareTo for descending)
        int gpaCmp = other.GPA.CompareTo(this.GPA);
        if (gpaCmp != 0) return gpaCmp;

        // Tiebreaker: Name ascending
        return string.Compare(this.Name, other.Name, StringComparison.Ordinal);
    }

    public override string ToString() => $"{Name} (GPA:{GPA}, Year:{Year})";
}

class Program
{
    static void Main()
    {
        var students = new List<Student>
        {
            new Student { Name="Charlie", GPA=3.5, Year=2 },
            new Student { Name="Alice",   GPA=3.9, Year=3 },
            new Student { Name="Bob",     GPA=3.9, Year=1 },
            new Student { Name="Diana",   GPA=3.2, Year=4 }
        };

        students.Sort();  // uses IComparable<Student>.CompareTo

        foreach (var s in students)
            Console.WriteLine(s);
    }
}

/* Output:
   Alice (GPA:3.9, Year:3)    ? GPA 3.9, Alice before Bob alphabetically
   Bob (GPA:3.9, Year:1)      ? GPA 3.9
   Charlie (GPA:3.5, Year:2)
   Diana (GPA:3.2, Year:4)
*/
from dataclasses import dataclass, field
from functools import total_ordering
from typing import List

# --- Python — Natural Ordering with total_ordering ---------------------------
# @total_ordering fills in all comparison methods from __eq__ and one other

@total_ordering
@dataclass
class Student:
    name: str
    gpa:  float
    year: int

    def __eq__(self, other):
        return (self.gpa, self.name) == (other.gpa, other.name)

    def __lt__(self, other):
        # Descending GPA: if self.gpa > other.gpa, self comes first (is "less than" in sort order)
        if self.gpa != other.gpa:
            return self.gpa > other.gpa  # higher GPA = "smaller" (sorts first)
        return self.name < other.name   # same GPA: alphabetical

    def __repr__(self):
        return f"{self.name} (GPA:{self.gpa}, Year:{self.year})"


students = [
    Student("Charlie", 3.5, 2),
    Student("Alice",   3.9, 3),
    Student("Bob",     3.9, 1),
    Student("Diana",   3.2, 4),
]
students.sort()   # uses __lt__ via total_ordering
for s in students: print(s)
# Alice (GPA:3.9, Year:3)
# Bob (GPA:3.9, Year:1)
# Charlie (GPA:3.5, Year:2)
# Diana (GPA:3.2, Year:4)

Section 3 — IComparer<T> — Custom Sort Strategies

Sometimes you need multiple different sort orders for the same type. For example: sort students by GPA, or by name, or by year — all without changing the class itself. Use IComparer<T> for this.

using System;
using System.Collections.Generic;

// --- IComparer<T> — Multiple Sort Strategies ----------------------------------

class Product
{
    public string Name  { get; set; }
    public decimal Price { get; set; }
    public int    Stock { get; set; }
    public override string ToString() => $"{Name,10} ${Price:F2} Stock:{Stock}";
}

// Comparer 1: Sort by price ascending
class PriceComparer : IComparer<Product>
{
    public int Compare(Product x, Product y) => x.Price.CompareTo(y.Price);
}

// Comparer 2: Sort by stock descending (highest stock first)
class StockDescComparer : IComparer<Product>
{
    public int Compare(Product x, Product y) => y.Stock.CompareTo(x.Stock);
}

// Comparer 3: Sort by name length, then alphabetically
class NameLengthComparer : IComparer<Product>
{
    public int Compare(Product x, Product y)
    {
        int lenCmp = x.Name.Length.CompareTo(y.Name.Length);
        return lenCmp != 0 ? lenCmp : string.Compare(x.Name, y.Name);
    }
}

class Program
{
    static void Main()
    {
        var products = new List<Product>
        {
            new Product { Name="Widget",  Price=9.99m,  Stock=50 },
            new Product { Name="Gadget",  Price=24.99m, Stock=10 },
            new Product { Name="Gizmo",   Price=4.99m,  Stock=200},
            new Product { Name="Doohickey",Price=14.99m,Stock=75 },
        };

        Console.WriteLine("- By Price:");
        products.Sort(new PriceComparer());
        products.ForEach(p => Console.WriteLine(p));

        Console.WriteLine("\n- By Stock (descending):");
        products.Sort(new StockDescComparer());
        products.ForEach(p => Console.WriteLine(p));
    }
}
from dataclasses import dataclass
from typing import List

@dataclass
class Product:
    name:  str
    price: float
    stock: int
    def __repr__(self): return f"{self.name:10} ${self.price:.2f} Stock:{self.stock}"

    Product("Widget",   9.99,  50),
    Product("Gadget",   24.99, 10),
    Product("Gizmo",    4.99,  200),
    Product("Doohickey",14.99, 75),

print("By price:")
for p in by_price: print(p)

# Sort by stock descending

print("\nBy stock (descending):")
for p in by_stock_desc: print(p)

# Multi-key: first by length of name, then alphabetically
by_name = sorted(products, key=lambda p: (len(p.name), p.name))
print("\nBy name length then alpha:")
for p in by_name: print(p)

Section 4 — LINQ Sorting (C#)

LINQ provides a fluent, readable API for sorting collections. It uses deferred execution and returns IEnumerable<T> rather than modifying in place — very useful in functional pipelines.

using System;
using System.Collections.Generic;
using System.Linq;

record Employee(string Name, string Department, int Salary, int Years);

class LinqSortingExamples
{
    static void Main()
    {
        var employees = new List<Employee>
        {
            new("Alice",   "Engineering", 95000, 5),
            new("Bob",     "Marketing",   72000, 3),
            new("Charlie", "Engineering", 88000, 8),
            new("Diana",   "Marketing",   85000, 6),
            new("Eve",     "Engineering", 95000, 2),
        };

        // -- Single key: sort by salary descending --
        var bySalary = employees.OrderByDescending(e => e.Salary);
        Console.WriteLine("By Salary (desc):");
        foreach (var e in bySalary)
            Console.WriteLine($"  {e.Name,-10} ${e.Salary}");

        // -- Multi-key: by department, then by salary descending --
        var byDeptThenSalary = employees
            .OrderBy(e => e.Department)
            .ThenByDescending(e => e.Salary)
            .ThenBy(e => e.Name);

        Console.WriteLine("\nBy Dept ? Salary? ? Name?:");
        foreach (var e in byDeptThenSalary)
            Console.WriteLine($"  {e.Department,-12} {e.Name,-10} ${e.Salary}");

        // -- LINQ + Where + OrderBy pipeline --
        var seniorEngineers = employees
            .Where(e => e.Department == "Engineering" && e.Years >= 4)
            .OrderBy(e => e.Years)
            .Select(e => $"{e.Name} ({e.Years}yr)");

        Console.WriteLine("\nSenior engineers by seniority:");
        Console.WriteLine(string.Join(", ", seniorEngineers));
        // Alice (5yr), Charlie (8yr)
    }
}
from dataclasses import dataclass

@dataclass
class Employee:
    name:       str
    department: str
    salary:     int
    years:      int

employees = [
    Employee("Alice",   "Engineering", 95000, 5),
    Employee("Bob",     "Marketing",   72000, 3),
    Employee("Charlie", "Engineering", 88000, 8),
    Employee("Diana",   "Marketing",   85000, 6),
    Employee("Eve",     "Engineering", 95000, 2),
]

# Single key: salary descending
by_salary = sorted(employees, key=lambda e: e.salary, reverse=True)
print("By Salary (desc):")
for e in by_salary: print(f"  {e.name:10} ${e.salary}")

# Multi-key: department asc, then salary desc
# Python tip: for descending numeric key, negate it
by_dept_salary = sorted(employees, key=lambda e: (e.department, -e.salary, e.name))
print("\nBy Dept ? Salary? ? Name?:")
for e in by_dept_salary:
    print(f"  {e.department:12} {e.name:10} ${e.salary}")

# Filter + sort pipeline
senior_engineers = sorted(
    (e for e in employees if e.department=="Engineering" and e.years >= 4),
    key=lambda e: e.years
)
print("\nSenior engineers:", [f"{e.name}({e.years}yr)" for e in senior_engineers])

Section 5 — Python functools.cmp_to_key

Python's sorted() uses a key function (returns a sortable value). But sometimes you inherit legacy code using a comparator (like Java's Comparator or C's qsort). The functools.cmp_to_key adapter converts a comparator to a key function.

from functools import cmp_to_key
from typing import List

# --- functools.cmp_to_key — Legacy Comparator ---------------------------------
# A comparator: return negative if a < b, 0 if equal, positive if a > b

# Example 1: Sort version strings like "1.10.2" correctly
# String sort: "1.10.2" < "1.9.0" (wrong!), numeric sort: "1.9.0" < "1.10.2" (correct!)
def version_cmp(a: str, b: str) -> int:
    parts_a = [int(x) for x in a.split(".")]
    parts_b = [int(x) for x in b.split(".")]
    for pa, pb in zip(parts_a, parts_b):
        if pa != pb: return pa - pb
    return len(parts_a) - len(parts_b)

versions = ["1.10.2", "1.9.0", "2.0.0", "1.10.1", "1.2.3"]
print(sorted(versions))                               # WRONG: lexicographic
print(sorted(versions, key=cmp_to_key(version_cmp))) # CORRECT: semantic version
# ['1.2.3', '1.9.0', '1.10.1', '1.10.2', '2.0.0']

# Example 2: LeetCode 179 — Largest Number
# Given [3,30,34,5,9], arrange to form the largest number: "9534330"
def largest_number(nums: List[int]) -> str:
    def cmp(a: str, b: str) -> int:
        if a+b > b+a: return -1   # a should come first
        if a+b < b+a: return  1   # b should come first
        return 0
    strs = sorted(map(str, nums), key=cmp_to_key(cmp))
    result = "".join(strs)
    return "0" if result[0]=="0" else result

print(largest_number([3,30,34,5,9]))  # "9534330"
print(largest_number([10,2]))          # "210"
using System;
using System.Collections.Generic;
using System.Linq;

// --- C# — Sorting with custom Comparison lambda (equivalent to cmp_to_key) -----
// LeetCode 179 — Largest Number

class LargestNumber
{
    static string Solve(int[] nums)
    {
        string[] strs = Array.ConvertAll(nums, n => n.ToString());
        // Comparator: prefer ab over ba if ab > ba lexicographically
        Array.Sort(strs, (a, b) => string.Compare(b + a, a + b, StringComparison.Ordinal));
        string result = string.Concat(strs);
        return result.TrimStart('0') == "" ? "0" : result.TrimStart('0');
    }

    static void Main()
    {
        Console.WriteLine(Solve(new[] { 3,30,34,5,9 }));  // 9534330
        Console.WriteLine(Solve(new[] { 10,2 }));           // 210
        Console.WriteLine(Solve(new[] { 0,0 }));            // 0
    }
}

Section 7 — Practice Questions

? Q1: Sort this list of tuples (name, age) first by age ascending, then by name descending: [("Alice", 30), ("Bob", 25), ("Charlie", 30), ("Diana", 25)]
? Multi-key sort with mixed directions
var people = new[] {("Alice",30),("Bob",25),("Charlie",30),("Diana",25)};
var sorted = people.OrderBy(p=>p.Item2).ThenByDescending(p=>p.Item1).ToList();
// Diana 25, Bob 25, Charlie 30, Alice 30
people = [("Alice",30),("Bob",25),("Charlie",30),("Diana",25)]
result = sorted(people, key=lambda x: (x[1], [-ord(c) for c in x[0]]))
# Simpler approach: two stable passes (sort by secondary first, then primary)
result2 = sorted(people, key=lambda x: x[0], reverse=True)  # name desc
result2 = sorted(result2, key=lambda x: x[1])              # age asc (stable!)
print(result2)  # [('Diana',25),('Bob',25),('Charlie',30),('Alice',30)]
? Q2: In C#, what is the difference between Array.Sort() and LINQ OrderBy() in terms of stability?
? Array.Sort = unstable, OrderBy = stable

Array.Sort() uses introspective sort which is unstable. If two elements compare equal, their original relative order is NOT preserved. LINQ OrderBy() is documented as a stable sort — equal elements maintain their original order. For List<T>.Sort(): also technically unstable. Best practice: use LINQ's OrderBy() when stability matters.
