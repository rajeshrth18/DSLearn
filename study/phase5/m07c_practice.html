Easy Problems (1–3)
#1Number of Islands
EasyGrid DFS/BFS

Given a 2D grid of '1' (land) and '0' (water), count the number of islands. An island is surrounded by water and formed by connecting adjacent lands horizontally or vertically.

Example: 
11110
11010
11000
00000  → 1 island
Approach: Scan the grid. When we find '1', increment count and DFS/BFS to mark all connected land as '0' (visited). Count = number of times we start a new flood-fill.
public int NumIslands(char[][] grid)
{
    int rows = grid.Length, cols = grid[0].Length, count = 0;
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
            if (grid[r][c] == '1') { count++; Sink(grid, r, c, rows, cols); }
    return count;
}
private void Sink(char[][] g, int r, int c, int rows, int cols)
{
    if (r < 0 || r >= rows || c < 0 || c >= cols || g[r][c] != '1') return;
    g[r][c] = '0'; // mark visited by sinking
    Sink(g, r+1, c, rows, cols); Sink(g, r-1, c, rows, cols);
    Sink(g, r, c+1, rows, cols); Sink(g, r, c-1, rows, cols);
}
// Time: O(rows×cols), Space: O(rows×cols) recursion stack
// Modifies input grid — clone first if that's a concern
#2Flood Fill
EasyGrid DFS

Given an image (2D int array), a starting pixel (sr,sc), and a new color, perform flood fill. Change the color of the starting pixel and all adjacent pixels of the same original color to the new color.

Approach: DFS from (sr,sc). Spread to neighbors that have the same original color. Edge case: if starting color already equals new color, return immediately to avoid infinite loop.
public int[][] FloodFill(int[][] image, int sr, int sc, int color)
{
    int orig = image[sr][sc];
    if (orig == color) return image; // avoid infinite loop
    Fill(image, sr, sc, orig, color);
    return image;
}
private void Fill(int[][] img, int r, int c, int orig, int newColor)
{
    if (r < 0 || r >= img.Length || c < 0 || c >= img[0].Length) return;
    if (img[r][c] != orig) return;
    img[r][c] = newColor;
    Fill(img, r+1, c, orig, newColor); Fill(img, r-1, c, orig, newColor);
    Fill(img, r, c+1, orig, newColor); Fill(img, r, c-1, orig, newColor);
}
// Time: O(rows×cols), Space: O(rows×cols) recursion stack
#3Find if Path Exists in Graph
EasyBFS/DFS or Union-Find

Given n nodes (0 to n-1), edges array, source, and destination — return true if a valid path exists from source to destination.

Approach (Union-Find): Union all edges. Check if source and destination have the same root — they're connected.
public bool ValidPath(int n, int[][] edges, int source, int destination)
{
    if (source == destination) return true;
    var uf = new UnionFind(n);
    foreach (var e in edges) uf.Union(e[0], e[1]);
    return uf.Connected(source, destination);
}
// Alternatively: BFS from source, check if we reach destination
// Union-Find is cleaner for pure connectivity queries
Medium Problems (4–11)
#4Clone Graph
MediumBFSHashMap

Given a reference to a node in a connected undirected graph, return a deep copy (clone) of the graph. Each node has a val and a list of neighbors.

Approach: BFS traversal. Use a HashMap from original node → cloned node. When visiting a neighbor: if not in map, create clone and enqueue. Always connect cloned neighbors.
public Node? CloneGraph(Node? node)
{
    if (node == null) return null;
    var clones = new Dictionary<Node, Node>();
    clones[node] = new Node(node.val);

    var q = new Queue<Node>();
    q.Enqueue(node);

    while (q.Count > 0)
    {
        var cur = q.Dequeue();
        foreach (var nb in cur.neighbors)
        {
            if (!clones.ContainsKey(nb))
            {
                clones[nb] = new Node(nb.val);
                q.Enqueue(nb);
            }
            clones[cur].neighbors.Add(clones[nb]); // wire cloned neighbors
        }
    }
    return clones[node];
}
// Time: O(V+E), Space: O(V+E)
// Key: create clone when FIRST SEEN, wire neighbors for every edge
#5Course Schedule (Can Finish?)
MediumDirected Cycle DetectionTopological Sort

Given numCourses and prerequisites array where [a,b] means "b must be taken before a", return true if you can finish all courses.

Example: numCourses=2, prerequisites=[[1,0]] → true. [[1,0],[0,1]] → false (cycle).
Approach: Build directed graph. Run Kahn's topological sort. If we can process all n nodes = no cycle = all courses finishable.
public bool CanFinish(int n, int[][] prereqs)
{
    List<int>[] adj = new List<int>[n];
    int[] inDeg = new int[n];
    for (int i = 0; i < n; i++) adj[i] = new();
    foreach (var p in prereqs)
    {
        adj[p[1]].Add(p[0]); // p[1] → p[0] (prereq → course)
        inDeg[p[0]]++;
    }
    var q = new Queue<int>();
    for (int i = 0; i < n; i++) if (inDeg[i] == 0) q.Enqueue(i);
    int processed = 0;
    while (q.Count > 0)
    {
        int c = q.Dequeue(); processed++;
        foreach (int next in adj[c])
            if (--inDeg[next] == 0) q.Enqueue(next);
    }
    return processed == n;
}
// Time: O(V+E), Space: O(V+E)
#6Course Schedule II (Return Order)
MediumTopological Sort

Same as #5 but return an ordering of courses you should take. Return empty if impossible.

Approach: Kahn's algorithm — collect processed nodes into result list. If result.Count == n, return it. Otherwise cycle exists, return empty.
public int[] FindOrder(int n, int[][] prereqs)
{
    List<int>[] adj = new List<int>[n];
    int[] inDeg = new int[n];
    for (int i = 0; i < n; i++) adj[i] = new();
    foreach (var p in prereqs) { adj[p[1]].Add(p[0]); inDeg[p[0]]++; }
    var q = new Queue<int>();
    for (int i = 0; i < n; i++) if (inDeg[i] == 0) q.Enqueue(i);
    var order = new List<int>();
    while (q.Count > 0)
    {
        int c = q.Dequeue(); order.Add(c);
        foreach (int next in adj[c])
            if (--inDeg[next] == 0) q.Enqueue(next);
    }
    return order.Count == n ? order.ToArray() : Array.Empty<int>();
}
// Time: O(V+E), Space: O(V+E)
#7Number of Connected Components
MediumUnion-Find

Given n nodes (0 to n-1) and an array of undirected edges, return the number of connected components.

Approach: Union-Find is perfect here. Start with n components. Each successful union reduces count by 1. Return final count.
public int CountComponents(int n, int[][] edges)
{
    var uf = new UnionFind(n);
    foreach (var e in edges) uf.Union(e[0], e[1]);
    return uf.Components;
}
// Time: O(E·α(n)) ≈ O(E), Space: O(n)
#8Pacific Atlantic Water Flow
MediumMulti-source BFS

Given m×n grid of heights, water can flow to adjacent cells with equal or lower height. Find all cells from which water can flow to both the Pacific (top/left) and Atlantic (bottom/right) ocean.

Approach: Reverse the flow direction — instead of water flowing DOWN from each cell, do BFS/DFS flowing UP from ocean borders. Pacific BFS from top/left edges. Atlantic BFS from bottom/right edges. Answer = cells reachable by both.
public IList<IList<int>> PacificAtlantic(int[][] heights)
{
    int rows=heights.Length, cols=heights[0].Length;
    bool[,] pac = BFSOcean(heights, rows, cols, true);
    bool[,] atl = BFSOcean(heights, rows, cols, false);
    var res = new List<IList<int>>();
    for (int r=0;r<rows;r++)
        for (int c=0;c<cols;c++)
            if (pac[r,c] && atl[r,c]) res.Add(new List<int>{r,c});
    return res;
}
private bool[,] BFSOcean(int[][] h, int R, int C, bool pacific)
{
    bool[,] vis = new bool[R,C];
    var q = new Queue<(int,int)>();
    int[][] dirs = {{0,1},{0,-1},{1,0},{-1,0}};
    // Seed border cells adjacent to the ocean
    for (int r=0;r<R;r++) { int c=pacific?0:C-1; vis[r,c]=true; q.Enqueue((r,c)); }
    for (int c=0;c<C;c++) { int r=pacific?0:R-1; if(!vis[r,c]){vis[r,c]=true;q.Enqueue((r,c));} }
    while (q.Count>0) {
        var(r,c)=q.Dequeue();
        foreach(var d in dirs) {
            int nr=r+d[0],nc=c+d[1];
            if(nr>=0&&nr<R&&nc>=0&&nc<C&&!vis[nr,nc]&&h[nr][nc]>=h[r][c])
            { vis[nr,nc]=true; q.Enqueue((nr,nc)); }
        }
    }
    return vis;
}
// Time: O(rows×cols), Space: O(rows×cols)
// KEY INSIGHT: flow UP (neighbor height >= current) from ocean borders
#9Rotting Oranges
MediumMulti-source BFS

Grid of 0 (empty), 1 (fresh), 2 (rotten). Every minute, rotten oranges infect adjacent fresh ones. Return minimum minutes to rot all oranges, or -1 if impossible.

Approach: Multi-source BFS — all rotten oranges expand simultaneously. Track count of fresh oranges. Return minutes when fresh=0, or -1 if any fresh remain.
public int OrangesRotting(int[][] grid)
{
    int rows=grid.Length, cols=grid[0].Length, fresh=0, min=0;
    int[][] dirs={{0,1},{0,-1},{1,0},{-1,0}};
    var q=new Queue<(int,int)>();
    for(int r=0;r<rows;r++) for(int c=0;c<cols;c++)
    {
        if(grid[r][c]==2) q.Enqueue((r,c));
        if(grid[r][c]==1) fresh++;
    }
    while(q.Count>0 && fresh>0)
    {
        min++; int sz=q.Count;
        for(int i=0;i<sz;i++) {
            var(r,c)=q.Dequeue();
            foreach(var d in dirs) {
                int nr=r+d[0],nc=c+d[1];
                if(nr>=0&&nr<rows&&nc>=0&&nc<cols&&grid[nr][nc]==1)
                { grid[nr][nc]=2; fresh--; q.Enqueue((nr,nc)); }
            }
        }
    }
    return fresh==0 ? min : -1;
}
#10Graph Valid Tree
MediumUnion-Find

Given n nodes (0 to n-1) and a list of undirected edges, return true if these edges form a valid tree. A valid tree = connected + no cycles (equivalently: n nodes and exactly n-1 edges with 1 component).

Approach: Two checks: 1) exactly n-1 edges (tree property), 2) all nodes connected (Union-Find: 1 component after all unions, and no edge creates a cycle).
public bool ValidTree(int n, int[][] edges)
{
    if (edges.Length != n - 1) return false; // tree needs exactly n-1 edges
    var uf = new UnionFind(n);
    foreach (var e in edges)
        if (!uf.Union(e[0], e[1])) return false; // cycle detected
    return uf.Components == 1; // all nodes connected
}
// Actually with n-1 edges and no cycle, it must be connected.
// The two checks are sufficient. Time: O(n·α(n)) ≈ O(n)
#11Walls and Gates (BFS Distance)
MediumMulti-source BFS

Given an m×n grid with -1 (wall), 0 (gate), INF (empty room). Fill each empty room with the distance to its nearest gate. INF = 2147483647.

Approach: Multi-source BFS starting from all gates (value 0) simultaneously. Expand outward incrementing distance. Only update cells that are INF.
public void WallsAndGates(int[][] rooms)
{
    int R=rooms.Length, C=rooms[0].Length, INF=int.MaxValue;
    int[][] dirs={{0,1},{0,-1},{1,0},{-1,0}};
    var q=new Queue<(int,int)>();
    // Seed all gates
    for(int r=0;r<R;r++) for(int c=0;c<C;c++)
        if(rooms[r][c]==0) q.Enqueue((r,c));
    while(q.Count>0) {
        var(r,c)=q.Dequeue();
        foreach(var d in dirs) {
            int nr=r+d[0],nc=c+d[1];
            if(nr>=0&&nr<R&&nc>=0&&nc<C&&rooms[nr][nc]==INF) {
                rooms[nr][nc]=rooms[r][c]+1;
                q.Enqueue((nr,nc));
            }
        }
    }
}
// Time: O(rows×cols), Space: O(rows×cols)
Hard Problems (12–15)
#12Word Ladder
HardBFSString Graph

Given beginWord, endWord, and a wordList — find the length of the shortest transformation sequence from beginWord to endWord, where each step changes exactly one letter and each intermediate word must exist in wordList.

Example: begin="hit", end="cog", list=["hot","dot","dog","lot","log","cog"] → 5 (hit→hot→dot→dog→cog)
Approach: BFS on "word graph" — two words are connected if they differ by exactly 1 character. Use a set for O(1) lookup. BFS gives shortest path.
public int LadderLength(string begin, string end, IList<string> wordList)
{
    var wordSet = new HashSet<string>(wordList);
    if (!wordSet.Contains(end)) return 0;
    var q = new Queue<(string w, int steps)>();
    q.Enqueue((begin, 1));
    wordSet.Remove(begin);
    while (q.Count > 0)
    {
        var (word, steps) = q.Dequeue();
        if (word == end) return steps;
        char[] arr = word.ToCharArray();
        for (int i = 0; i < arr.Length; i++)
        {
            char orig = arr[i];
            for (char c = 'a'; c <= 'z'; c++)
            {
                arr[i] = c;
                string next = new string(arr);
                if (wordSet.Contains(next))
                {
                    wordSet.Remove(next); // mark visited by removing from set
                    q.Enqueue((next, steps + 1));
                }
            }
            arr[i] = orig;
        }
    }
    return 0;
}
// Time: O(M² × N) where M=word length, N=wordList size
// Space: O(M² × N)
#13Alien Dictionary
HardTopological Sort

Given a sorted list of words in an alien language, determine the order of characters. Return any valid ordering of the alien alphabet, or "" if invalid.

Approach: Compare adjacent words to extract character ordering rules. Build directed graph of char→char. Topological sort. If cycle → invalid (return ""). If result contains fewer chars than unique chars → cycle.
public string AlienOrder(string[] words)
{
    var adj = new Dictionary<char, HashSet<char>>();
    var inDeg = new Dictionary<char, int>();
    foreach (var w in words) foreach (char c in w) { adj[c]=new(); inDeg[c]=0; }

    for (int i=0;i<words.Length-1;i++)
    {
        string w1=words[i], w2=words[i+1];
        int minLen=Math.Min(w1.Length,w2.Length);
        if(w1.Length>w2.Length && w1.StartsWith(w2)) return ""; // invalid: "ab" before "a"
        for(int j=0;j<minLen;j++)
        {
            if(w1[j]!=w2[j]) { if(!adj[w1[j]].Contains(w2[j])){adj[w1[j]].Add(w2[j]);inDeg[w2[j]]++;} break; }
        }
    }
    // Kahn's topo sort on character graph
    var q=new Queue<char>();
    foreach(var (c,d) in inDeg) if(d==0) q.Enqueue(c);
    var sb=new System.Text.StringBuilder();
    while(q.Count>0) {
        char c=q.Dequeue(); sb.Append(c);
        foreach(char nb in adj[c]) if(--inDeg[nb]==0) q.Enqueue(nb);
    }
    return sb.Length==inDeg.Count ? sb.ToString() : "";
}
// Time: O(C + U + min(n,U)) where C=total chars in all words, U=unique chars
#14Network Delay Time
HardDijkstra

Given directed weighted edges times[i]=[u,v,w] (signal travel time), a source k, and n nodes — return the minimum time for all n nodes to receive the signal, or -1 if impossible.

Approach: Dijkstra from source k. The answer = max of all shortest distances (last node to receive signal). If any node is unreachable (dist = ∞), return -1.
public int NetworkDelayTime(int[][] times, int n, int k)
{
    var g = new List<(int to, int w)>[n + 1];
    for (int i = 0; i <= n; i++) g[i] = new();
    foreach (var t in times) g[t[0]].Add((t[1], t[2]));

    int[] dist = new int[n + 1]; Array.Fill(dist, int.MaxValue); dist[k] = 0;
    var pq = new PriorityQueue<int, int>();
    pq.Enqueue(k, 0);

    while (pq.Count > 0) {
        pq.TryDequeue(out int node, out int d);
        if (d > dist[node]) continue;
        foreach (var (nb, w) in g[node]) {
            if (dist[node] + w < dist[nb]) {
                dist[nb] = dist[node] + w;
                pq.Enqueue(nb, dist[nb]);
            }
        }
    }
    int maxDist = dist.Skip(1).Max();
    return maxDist == int.MaxValue ? -1 : maxDist;
}
// Time: O((V+E) log V), Space: O(V+E)
#15Cheapest Flights Within K Stops
HardBellman-Ford

Given n cities, flights[i]=[from,to,price], src, dst, and k — find cheapest price from src to dst with at most k stops. Return -1 if impossible.

Note: K stops = K+1 edges in path.
Approach: Bellman-Ford with exactly K+1 iterations. Copy dist array before each iteration to prevent using updates from the same round (otherwise we'd allow more than 1 edge per step).
public int FindCheapestPrice(int n, int[][] flights, int src, int dst, int k)
{
    int[] dist = new int[n]; Array.Fill(dist, int.MaxValue); dist[src] = 0;

    for (int i = 0; i <= k; i++) // at most k+1 edges = k stops
    {
        int[] temp = (int[])dist.Clone(); // snapshot before this round
        foreach (var f in flights)
        {
            int u=f[0], v=f[1], w=f[2];
            if (dist[u] != int.MaxValue && dist[u] + w < temp[v])
                temp[v] = dist[u] + w;
        }
        dist = temp;
    }
    return dist[dst] == int.MaxValue ? -1 : dist[dst];
}
// Time: O(k × E), Space: O(n)
// KEY: clone dist before each iteration — ensures we only extend paths by 1 edge per round
// This is the K-constrained Bellman-Ford variant
