1Network Delay Time (LeetCode #743)
MediumDijkstra
You have a network of n nodes, labeled from 1 to n. Given a list of travel times as directed edges times[i] = [u, v, w] (from u to v taking w milliseconds), and a source node k, return the minimum time for all nodes to receive the signal. If it's impossible, return -1.
Input: times = [[2,1,1],[2,3,1],[3,4,1]], n = 4, k = 2 Output: 2 Explanation: Start at 2. 2?1 takes 1ms. 2?3 takes 1ms, then 3?4 takes 1ms. All nodes reached by time 2 (max over all nodes of dist[node]).
Approach: Dijkstra from k, return max dist. Build adjacency list. Run Dijkstra from k. Return max(dist.values()). If any node is unreachable (dist = 8), return -1. The answer is the maximum shortest-path distance from k to any node — the time when the last node receives the signal.
Time: O((V+E) log V) | Space: O(V+E)
C# 
public int NetworkDelayTime(int[][] times, int n, int k)
{
    var adj = new List<(int v, int w)>[n + 1];
    for (int i = 0; i <= n; i++) adj[i] = new();
    foreach (var t in times) adj[t[0]].Add((t[1], t[2]));

    int[] dist = new int[n + 1];
    Array.Fill(dist, int.MaxValue);
    dist[k] = 0;

    var pq = new PriorityQueue<int, int>();
    pq.Enqueue(k, 0);

    while (pq.Count > 0)
    {
        pq.TryDequeue(out int u, out int d);
        if (d > dist[u]) continue;
        foreach (var (v, w) in adj[u])
            if (dist[u] + w < dist[v])
            {
                dist[v] = dist[u] + w;
                pq.Enqueue(v, dist[v]);
            }
    }
    int maxDist = 0;
    for (int i = 1; i <= n; i++)
    {
        if (dist[i] == int.MaxValue) return -1;
        maxDist = Math.Max(maxDist, dist[i]);
    }
    return maxDist;
}
Python 
import heapq
from collections import defaultdict

def networkDelayTime(times, n, k):
    adj = defaultdict(list)
    for u, v, w in times: adj[u].append((v, w))
    dist = {k: 0}
    pq = [(0, k)]
    while pq:
        d, u = heapq.heappop(pq)
        if d > dist.get(u, float('inf')): continue
        for v, w in adj[u]:
            nd = d + w
            if nd < dist.get(v, float('inf')):
                dist[v] = nd
                heapq.heappush(pq, (nd, v))
    return max(dist.values()) if len(dist) == n else -1
2Path With Minimum Effort (LeetCode #1631)
MediumDijkstra
You're hiking on a grid where heights[r][c] is the height of cell (r,c). "Effort" of a route = maximum absolute difference in heights of consecutive cells. Find the path from top-left (0,0) to bottom-right (rows-1, cols-1) that minimizes effort.
Input: heights = [[1,2,2],[3,8,2],[5,3,5]] Output: 2 Explanation: Path (0,0)?(0,1)?(0,2)?(1,2)?(2,2): efforts=[1,0,6,3], max=6. Too high. Path (0,0)?(1,0)?(2,0)?(2,1)?(2,2): efforts=[2,2,2,2], max=2. ?
Approach: Modified Dijkstra minimizing max-edge instead of sum. Edge weight from cell (r,c) to neighbor (nr,nc) = abs(heights[r][c] - heights[nr][nc]). dist[r][c] = minimum possible maximum-effort to reach (r,c). Update: dist[nr][nc] = min(dist[nr][nc], max(dist[r][c], edgeWeight)). Use a min-heap on effort.
Time: O(R×C × log(R×C)) | Space: O(R×C)
C# 
public int MinimumEffortPath(int[][] heights)
{
    int rows = heights.Length, cols = heights[0].Length;
    int[][] dist = Enumerable.Range(0, rows)
        .Select(_ => Enumerable.Repeat(int.MaxValue, cols).ToArray())
        .ToArray();
    dist[0][0] = 0;
    var pq = new PriorityQueue<(int r, int c), int>();
    pq.Enqueue((0,0), 0);
    int[] dr = {0,0,1,-1}, dc = {1,-1,0,0};
    while (pq.Count > 0) {
        pq.TryDequeue(out var pos, out int eff);
        var (r, c) = pos;
        if (r == rows-1 && c == cols-1) return eff;
        if (eff > dist[r][c]) continue;
        for (int d = 0; d < 4; d++) {
            int nr = r+dr[d], nc = c+dc[d];
            if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
            int newEff = Math.Max(eff, Math.Abs(heights[r][c]-heights[nr][nc]));
            if (newEff < dist[nr][nc]) { dist[nr][nc] = newEff; pq.Enqueue((nr,nc), newEff); }
        }
    }
    return dist[rows-1][cols-1];
}
Python 
import heapq

def minimumEffortPath(heights):
    rows, cols = len(heights), len(heights[0])
    dist = [[float('inf')] * cols for _ in range(rows)]
    dist[0][0] = 0
    pq = [(0, 0, 0)]  # (effort, r, c)
    while pq:
        eff, r, c = heapq.heappop(pq)
        if r == rows-1 and c == cols-1: return eff
        if eff > dist[r][c]: continue
        for nr, nc in [(r+1,c),(r-1,c),(r,c+1),(r,c-1)]:
            if 0 <= nr < rows and 0 <= nc < cols:
                new_eff = max(eff, abs(heights[r][c] - heights[nr][nc]))
                if new_eff < dist[nr][nc]:
                    dist[nr][nc] = new_eff
                    heapq.heappush(pq, (new_eff, nr, nc))
    return dist[rows-1][cols-1]
3Cheapest Flights Within K Stops (LeetCode #787)
MediumBellman-Ford / DP
There are n cities connected by flights. Each flight has source, destination, and cost. Given src, dst, and k, find the cheapest price from src to dst with at most k stops. Return -1 if no such route exists.
Input: n=4, flights=[[0,1,100],[1,2,100],[2,0,100],[1,3,600],[2,3,200]], src=0, dst=3, k=1 Output: 700 Explanation: 0?1?3 costs 700 (1 stop). 0?1?2?3 costs 400 but has 2 stops (too many).
Approach: Bellman-Ford with exactly k+1 passes. Run at most k+1 edge relaxations. This is Bellman-Ford limited to k+1 edges (k stops = k+1 edges). Key: copy the dist array before each pass to prevent cascading updates within the same pass (which would use more than 1 step per pass). This is also solvable with Dijkstra adding stops to the state, or DP.
Time: O(k × E) | Space: O(n)
C# 
public int FindCheapestPrice(int n, int[][] flights, int src, int dst, int k)
{
    int[] prices = new int[n];
    Array.Fill(prices, int.MaxValue);
    prices[src] = 0;
    for (int i = 0; i <= k; i++)          // k stops = k+1 edges
    {
        int[] tmp = (int[])prices.Clone();  // snapshot to prevent cascading
        foreach (var f in flights)
        {
            int u = f[0], v = f[1], w = f[2];
            if (prices[u] == int.MaxValue) continue;
            if (prices[u] + w < tmp[v]) tmp[v] = prices[u] + w;
        }
        prices = tmp;
    }
    return prices[dst] == int.MaxValue ? -1 : prices[dst];
}
Python 
def findCheapestPrice(n, flights, src, dst, k):
    prices = [float('inf')] * n
    prices[src] = 0
    for _ in range(k + 1):            # k stops = k+1 edges
        tmp = prices[:]               # snapshot!
        for u, v, w in flights:
            if prices[u] != float('inf') and prices[u] + w < tmp[v]:
                tmp[v] = prices[u] + w
        prices = tmp
    return prices[dst] if prices[dst] != float('inf') else -1
4Find the City With Fewest Reachable Neighbors (LeetCode #1334)
MediumFloyd-Warshall
Given n cities, edges with distances, and a threshold, return the city that has the fewest number of other cities reachable with a distance = threshold. If there's a tie, return the city with the greatest number (highest index).
Input: n=4, edges=[[0,1,3],[1,2,1],[1,3,4],[2,3,1]], distanceThreshold=4 Output: 3 City 0: reach {1,2,3} (dist 3,4,4) ? 3 cities (4 is threshold, counts) City 3: reach {1,2} (dist 4,2) ? 2 cities ? fewest (tie with city 0 would pick 3)
Approach: Floyd-Warshall for all-pairs shortest paths, then count. With n = 100, O(n³) Floyd-Warshall is fine. After computing all-pairs shortest paths, for each city count how many others are reachable within the threshold. Return the city with minimum count (highest index on tie).
Time: O(n³) | Space: O(n²)
C# 
public int FindTheCity(int n, int[][] edges, int thr)
{
    int[,] dist = new int[n, n];
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++)
            dist[i,j] = i == j ? 0 : 100001;
    foreach (var e in edges) { dist[e[0],e[1]] = e[2]; dist[e[1],e[0]] = e[2]; }
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                dist[i,j] = Math.Min(dist[i,j], dist[i,k] + dist[k,j]);
    int res = 0, minCnt = n;
    for (int i = 0; i < n; i++) {
        int cnt = 0;
        for (int j = 0; j < n; j++)
            if (i != j && dist[i,j] <= thr) cnt++;
        if (cnt <= minCnt) { minCnt = cnt; res = i; }  // <= picks highest index on tie
    }
    return res;
}
Python 
def findTheCity(n, edges, distanceThreshold):
    dist = [[float('inf')] * n for _ in range(n)]
    for i in range(n): dist[i][i] = 0
    for u, v, w in edges: dist[u][v] = dist[v][u] = w
    for k in range(n):
        for i in range(n):
            for j in range(n):
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
    res, min_cnt = 0, n
    for i in range(n):
        cnt = sum(1 for j in range(n) if i != j and dist[i][j] <= distanceThreshold)
        if cnt <= min_cnt: min_cnt, res = cnt, i
    return res

5Min Cost to Connect All Points (LeetCode #1584)
MediumMST / Prim's
Given an array of points in a 2D plane, connect all points with the minimum total Manhattan distance cost. Return the minimum cost. You can connect any two points directly.
Input: points = [[0,0],[2,2],[3,10],[5,2],[7,0]] Output: 20 Explanation: MST uses edges: (0,0)-(2,2)=4, (2,2)-(5,2)=3, (5,2)-(7,0)=4, (2,2)-(3,10)=9. Total = 20.
Approach: Prim's MST on complete graph. All pairs are connected with Manhattan distance weight. Run Prim's: maintain a minCost[] array (minimum edge cost to reach each unvisited vertex from the current MST). At each step, greedily add the cheapest connection. O(n²) time — better than Kruskal's O(n² log n) edge sort for this dense complete graph.
Time: O(n²) | Space: O(n)
C# 
public int MinCostConnectPoints(int[][] pts)
{
    int n = pts.Length;
    int[] minCost = new int[n];
    Array.Fill(minCost, int.MaxValue);
    minCost[0] = 0;
    bool[] inMST = new bool[n];
    int total = 0;
    for (int iter = 0; iter < n; iter++)
    {
        // Pick vertex with min cost not yet in MST
        int u = -1;
        for (int i = 0; i < n; i++)
            if (!inMST[i] && (u == -1 || minCost[i] < minCost[u])) u = i;
        inMST[u] = true;
        total += minCost[u];
        // Update costs of remaining vertices
        for (int v = 0; v < n; v++)
            if (!inMST[v]) {
                int d = Math.Abs(pts[u][0]-pts[v][0]) + Math.Abs(pts[u][1]-pts[v][1]);
                minCost[v] = Math.Min(minCost[v], d);
            }
    }
    return total;
}
Python 
def minCostConnectPoints(points):
    n = len(points)
    min_cost = [float('inf')] * n; min_cost[0] = 0
    in_mst = [False] * n; total = 0
    for _ in range(n):
        u = min(range(n), key=lambda x: min_cost[x] if not in_mst[x] else float('inf'))
        in_mst[u] = True; total += min_cost[u]
        for v in range(n):
            if not in_mst[v]:
                d = abs(points[u][0]-points[v][0]) + abs(points[u][1]-points[v][1])
                min_cost[v] = min(min_cost[v], d)
    return total
6Number of Connected Components in Undirected Graph (LeetCode #323)
MediumUnion-Find
Given n nodes labeled 0 to n-1 and a list of undirected edges, return the number of connected components.
Input: n=5, edges=[[0,1],[1,2],[3,4]] Output: 2 Components: {0,1,2} and {3,4}.
Approach: Union-Find. Start with n components. For each edge, Union(u,v). If they were in different components (Union returns true), decrement count. Final count = number of connected components. Alternatively, use BFS/DFS counting unvisited start vertices. Union-Find is faster O(a(n)) per edge vs O(V+E) for BFS/DFS but both are optimal here.
Time: O(n + E × a(n)) ˜ O(n + E) | Space: O(n)
C# 
public int CountComponents(int n, int[][] edges)
{
    int[] parent = Enumerable.Range(0, n).ToArray();
    int[] rank = new int[n];
    int Find(int x) => parent[x] == x ? x : parent[x] = Find(parent[x]);
    int components = n;
    foreach (var e in edges)
    {
        int px = Find(e[0]), py = Find(e[1]);
        if (px == py) continue;
        if (rank[px] < rank[py]) (px, py) = (py, px);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        components--;
    }
    return components;
}
Python 
def countComponents(n, edges):
    parent = list(range(n)); rank = [0] * n; count = n
    def find(x):
        if parent[x] != x: parent[x] = find(parent[x])
        return parent[x]
    for u, v in edges:
        pu, pv = find(u), find(v)
        if pu == pv: continue
        if rank[pu] < rank[pv]: pu, pv = pv, pu
        parent[pv] = pu
        if rank[pu] == rank[pv]: rank[pu] += 1
        count -= 1
    return count
7Redundant Connection (LeetCode #684)
MediumUnion-Find
A tree of n nodes (labeled 1..n) has had one extra edge added, forming exactly one cycle. Find the redundant edge that was added last (if multiple valid, return the one appearing latest in the input).
Input: edges = [[1,2],[1,3],[2,3]] Output: [2,3] Adding [2,3] creates a cycle 1-2-3-1. It appears last among valid cycle edges.
Approach: Process edges with Union-Find. First edge that connects same component = cycle edge. Process edges in order. For each edge (u,v), try Union(u,v). If Union returns false (already connected), this edge is the redundant one — return it. Since we process in order, the first such edge we encounter is exactly the one "added last" to create the cycle.
Time: O(n × a(n)) | Space: O(n)
C# 
public int[] FindRedundantConnection(int[][] edges)
{
    int n = edges.Length;
    int[] parent = Enumerable.Range(0, n+1).ToArray();
    int[] rank = new int[n+1];
    int Find(int x) => parent[x] == x ? x : parent[x] = Find(parent[x]);
    bool Union(int x, int y) {
        int px = Find(x), py = Find(y);
        if (px == py) return false;
        if (rank[px] < rank[py]) (px, py) = (py, px);
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
    foreach (var e in edges)
        if (!Union(e[0], e[1])) return e;
    return new int[0];
}
Python 
def findRedundantConnection(edges):
    n = len(edges)
    parent = list(range(n+1)); rank = [0] * (n+1)
    def find(x):
        if parent[x] != x: parent[x] = find(parent[x])
        return parent[x]
    def union(x, y):
        px, py = find(x), find(y)
        if px == py: return False
        if rank[px] < rank[py]: px, py = py, px
        parent[py] = px
        if rank[px] == rank[py]: rank[px] += 1
        return True
    for u, v in edges:
        if not union(u, v): return [u, v]
8Swim in Rising Water (LeetCode #778)
HardDijkstra / Binary Search
You have an n×n grid where grid[r][c] is the elevation at cell (r,c). At time t, you can swim from any cell to adjacent cells if t = elevation of both cells. Find the minimum time to swim from (0,0) to (n-1,n-1).
Input: grid = [[0,2],[1,3]] Output: 3 At t=3: (0,0)?(0,1) requires max(0,2)=2 = 3 ?, (0,1)?(1,1) requires max(2,3)=3 = 3 ?.
Approach: Dijkstra where edge weight = max(current_time, grid[nr][nc]). The "time" to reach (nr,nc) = max(time_to_reach_current, grid[nr][nc]) — you must wait until both the current cell and next cell are swimmable. This is exactly "minimum bottleneck path" — identical in structure to "Path With Minimum Effort" (P2) but with max(dist, elevation) instead of max(dist, abs_diff). Use Dijkstra with a min-heap on time.
Time: O(n² × log(n²)) | Space: O(n²)
C# 
public int SwimInWater(int[][] grid)
{
    int n = grid.Length;
    int[][] dist = Enumerable.Range(0,n)
        .Select(_ => Enumerable.Repeat(int.MaxValue, n).ToArray()).ToArray();
    dist[0][0] = grid[0][0];
    var pq = new PriorityQueue<(int,int),int>();
    pq.Enqueue((0,0), grid[0][0]);
    int[] dr={0,0,1,-1},dc={1,-1,0,0};
    while(pq.Count>0) {
        pq.TryDequeue(out var pos, out int t);
        var(r,c)=pos;
        if(r==n-1&&c==n-1) return t;
        if(t>dist[r][c]) continue;
        for(int d=0;d<4;d++) {
            int nr=r+dr[d],nc=c+dc[d];
            if(nr<0||nr>=n||nc<0||nc>=n) continue;
            int nt=Math.Max(t,grid[nr][nc]);
            if(nt<dist[nr][nc]){dist[nr][nc]=nt;pq.Enqueue((nr,nc),nt);}
        }
    }
    return dist[n-1][n-1];
}

import heapq

def swimInWater(grid):
    n = len(grid)
    dist = [[float('inf')]*n for _ in range(n)]
    dist[0][0] = grid[0][0]
    pq = [(grid[0][0], 0, 0)]
    while pq:
        t, r, c = heapq.heappop(pq)
        if r == n-1 and c == n-1: return t
        if t > dist[r][c]: continue
        for nr, nc in [(r+1,c),(r-1,c),(r,c+1),(r,c-1)]:
            if 0 <= nr < n and 0 <= nc < n:
                nt = max(t, grid[nr][nc])
                if nt < dist[nr][nc]:
                    dist[nr][nc] = nt
                    heapq.heappush(pq, (nt, nr, nc))

9Course Schedule II (LeetCode #210)
MediumTopological Sort
Given numCourses and prerequisites [a,b] (must take b before a), return a valid ordering to take all courses. If impossible (cycle), return an empty array.
Input: numCourses=4, prerequisites=[[1,0],[2,0],[3,1],[3,2]] Output: [0,2,1,3] or [0,1,2,3] (any valid topological order)
Approach: Kahn's algorithm (BFS topological sort). Build graph: edge b?a for each [a,b] prerequisite. Compute in-degrees. Start with in-degree 0 courses. Process in BFS order, decrement neighbors' in-degrees. If result length == numCourses, return result. Else cycle exists, return [].
Time: O(V+E) | Space: O(V+E)
C# 
public int[] FindOrder(int numCourses, int[][] prereqs)
{
    var adj = new List<int>[numCourses];
    int[] inDeg = new int[numCourses];
    for (int i = 0; i < numCourses; i++) adj[i] = new();
    foreach (var p in prereqs) { adj[p[1]].Add(p[0]); inDeg[p[0]]++; }
    var q = new Queue<int>();
    for (int i = 0; i < numCourses; i++) if (inDeg[i] == 0) q.Enqueue(i);
    var res = new List<int>();
    while (q.Count > 0) {
        int u = q.Dequeue(); res.Add(u);
        foreach (int v in adj[u]) if (--inDeg[v] == 0) q.Enqueue(v);
    }
    return res.Count == numCourses ? res.ToArray() : new int[0];
}
Python 
from collections import defaultdict, deque

def findOrder(numCourses, prerequisites):
    adj = defaultdict(list); in_deg = [0] * numCourses
    for a, b in prerequisites: adj[b].append(a); in_deg[a] += 1
    q = deque(i for i in range(numCourses) if in_deg[i] == 0)
    res = []
    while q:
        u = q.popleft(); res.append(u)
        for v in adj[u]:
            in_deg[v] -= 1
            if in_deg[v] == 0: q.append(v)
    return res if len(res) == numCourses else []
10Alien Dictionary (LeetCode #269)
HardTopological Sort
Given a sorted list of words from an alien language (sorted by alien alphabet order), determine the order of letters in the alien alphabet. Return any valid ordering, or "" if the input is invalid.
Input: words = ["wrt","wrf","er","ett","rftt"] Output: "wertf" From wrt vs wrf: t before f. From wrf vs er: w before e. From er vs ett: r before t. From ett vs rftt: e before r. Order: w?e?r?t?f
Approach: Build ordering graph from word comparisons, then topological sort. Compare adjacent words character by character. At the first differing character, word[i][j] comes before word[i+1][j] in the alphabet — add directed edge. Special case: if word[i] is a prefix of word[i+1] but longer, it's invalid (e.g., ["abc","ab"] is impossible). Run Kahn's on the character ordering graph. If all characters are ordered = return result. If cycle = return "".
Time: O(C) where C = total chars in all words | Space: O(1) — only 26 letters
C# 
public string AlienOrder(string[] words)
{
    var chars = new HashSet<char>(string.Concat(words));
    var adj   = new Dictionary<char, List<char>>();
    var inDeg = new Dictionary<char, int>();
    foreach (char c in chars) { adj[c] = new(); inDeg[c] = 0; }

    for (int i = 0; i < words.Length-1; i++) {
        string w1=words[i], w2=words[i+1];
        if (w1.Length > w2.Length && w1.StartsWith(w2)) return "";
        for (int j=0; j<Math.Min(w1.Length,w2.Length); j++) {
            if (w1[j]!=w2[j]) { adj[w1[j]].Add(w2[j]); inDeg[w2[j]]++; break; }
        }
    }
    var q = new Queue<char>(chars.Where(c => inDeg[c] == 0));
    var sb = new System.Text.StringBuilder();
    while (q.Count > 0) {
        char u = q.Dequeue(); sb.Append(u);
        foreach (char v in adj[u]) if (--inDeg[v] == 0) q.Enqueue(v);
    }
    return sb.Length == chars.Count ? sb.ToString() : "";
}

def alienOrder(words):
    adj    = {c: [] for w in words for c in w}
    in_deg = {c: 0 for c in adj}
    for i in range(len(words)-1):

        if len(w1) > len(w2) and w1.startswith(w2): return ""
        for a, b in zip(w1, w2):
            if a != b: adj[a].append(b); in_deg[b] += 1; break
    q = deque(c for c in in_deg if in_deg[c] == 0)
    res = []
    while q:
        u = q.popleft(); res.append(u)
        for v in adj[u]:
            in_deg[v] -= 1
            if in_deg[v] == 0: q.append(v)
    return "".join(res) if len(res) == len(in_deg) else ""
11Safe States in a Graph (LeetCode #802)
MediumTopological Sort / DFS
In a directed graph, a "safe" node is one from which every path leads to a terminal node (a node with no outgoing edges). Return all safe nodes in sorted order.
Input: graph = [[1,2],[2,3],[5],[0],[5],[],[]] Output: [2,4,5,6] Node 5,6 are terminals. Node 4 only goes to 5 (safe). Node 2 only goes to 5 (safe). Etc.
Approach: Reverse the graph, then topological sort from terminal nodes. A node is safe if it cannot reach a cycle. Reverse all edges. Terminal nodes in the original graph (out-degree 0) become source nodes in the reversed graph. Run Kahn's on the reversed graph. Only nodes in the topological order (reachable from sources in reversed graph = reachable from safe nodes in original) are safe.
Time: O(V+E) | Space: O(V+E)
C# 
public IList<int> EventualSafeNodes(int[][] graph)
{
    int n = graph.Length;
    var radj  = new List<int>[n];  // reversed graph
    int[] outDeg = new int[n];     // original out-degree ? in-degree in reversed
    for (int i=0;i<n;i++) radj[i]=new();
    for (int u=0;u<n;u++)
        foreach(int v in graph[u]) { radj[v].Add(u); outDeg[u]++; }
    var q = new Queue<int>();
    for(int i=0;i<n;i++) if(outDeg[i]==0) q.Enqueue(i);
    var safe = new HashSet<int>();
    while(q.Count>0) {
        int u=q.Dequeue(); safe.Add(u);
        foreach(int v in radj[u]) if(--outDeg[v]==0) q.Enqueue(v);
    }
    var res = safe.OrderBy(x=>x).ToList();
    return res;
}

from collections import defaultdict, deque

def eventualSafeNodes(graph):
    n = len(graph)
    radj = defaultdict(list)
    out_deg = [0] * n
    for u, nbrs in enumerate(graph):
        for v in nbrs: radj[v].append(u); out_deg[u] += 1
    q = deque(i for i in range(n) if out_deg[i] == 0)
    safe = set()
    while q:
        u = q.popleft(); safe.add(u)
        for v in radj[u]:
            out_deg[v] -= 1
            if out_deg[v] == 0: q.append(v)
    return sorted(safe)
12Critical Connections in a Network (LeetCode #1192)
HardBridges / DFS
Given a network of n servers and connections, find all critical connections — edges whose removal disconnects the network. These are also called "bridges." Return all such connections.
Input: n=4, connections=[[0,1],[1,2],[2,0],[1,3]] Output: [[1,3]] Removing edge 1-3 disconnects server 3 from the network. The cycle 0-1-2-0 means no edge in the cycle is a bridge.
Approach: Tarjan's Bridge Finding Algorithm — DFS with discovery times and low-link values. For each edge (u,v), track: disc[u] = discovery time of u, low[u] = minimum discovery time reachable from the subtree of u (via back edges). An edge (u,v) is a bridge if low[v] > disc[u] — meaning v's subtree cannot reach u or anything above u without going through the edge (u,v). This runs in O(V+E).
Time: O(V+E) | Space: O(V+E)
C# 
public IList<IList<int>> CriticalConnections(int n, IList<IList<int>> conns)
{
    var adj  = new List<int>[n];
    for(int i=0;i<n;i++) adj[i]=new();
    foreach(var c in conns){adj[c[0]].Add(c[1]);adj[c[1]].Add(c[0]);}
    int[] disc=new int[n], low=new int[n];
    Array.Fill(disc,-1);
    int timer=0;
    var res=new List<IList<int>>();
    void Dfs(int u,int parent){
        disc[u]=low[u]=timer++;
        foreach(int v in adj[u]){
            if(v==parent) continue;
            if(disc[v]==-1){
                Dfs(v,u);
                low[u]=Math.Min(low[u],low[v]);
                if(low[v]>disc[u]) res.Add(new List<int>{u,v});  // bridge!
            }else low[u]=Math.Min(low[u],disc[v]);
        }
    }
    for(int i=0;i<n;i++) if(disc[i]==-1) Dfs(i,-1);
    return res;
}

from collections import defaultdict

def criticalConnections(n, connections):
    adj = defaultdict(list)
    for u, v in connections: adj[u].append(v); adj[v].append(u)
    disc = [-1] * n; low = [0] * n; timer = [0]; bridges = []
    def dfs(u, parent):
        disc[u] = low[u] = timer[0]; timer[0] += 1
        for v in adj[u]:
            if v == parent: continue
            if disc[v] == -1:
                dfs(v, u); low[u] = min(low[u], low[v])
                if low[v] > disc[u]: bridges.append([u, v])
            else: low[u] = min(low[u], disc[v])
    for i in range(n):
        if disc[i] == -1: dfs(i, -1)
    return bridges
