Section 1 — What is a Suffix Array?

A Suffix Array (SA) is an array of integers giving the starting positions of all suffixes of a string in lexicographically sorted order. It is a space-efficient alternative to suffix trees and supports powerful string queries.

String s = "banana$" ($ is sentinel, smallest char) Index: 0123456 All suffixes: 0: banana$ 1: anana$ 2: nana$ 3: ana$ 4: na$ 5: a$ 6: $ Sorted lexicographically: 6: $ ← SA[0] = 6 5: a$ ← SA[1] = 5 3: ana$ ← SA[2] = 3 1: anana$ ← SA[3] = 1 0: banana$ ← SA[4] = 0 4: na$ ← SA[5] = 4 2: nana$ ← SA[6] = 2 SA = [6, 5, 3, 1, 0, 4, 2]

The LCP (Longest Common Prefix) array stores the LCP between adjacent suffixes in the sorted order. LCP[i] = LCP length between suffix SA[i-1] and suffix SA[i]. LCP[0] is undefined.

Section 2 — Building the Suffix Array

2.1 — O(n log n) via Prefix Doubling

Sort suffixes by their first 1 characters, then 2, then 4, ... doubling each time. Uses radix sort for O(n log n) total.

C#
static int[] BuildSuffixArray(string s)
{
    int n = s.Length;
    int[] sa = Enumerable.Range(0, n).ToArray();
    int[] rank = s.Select(c => (int)c).ToArray();
    int[] tmp = new int[n];

    for (int gap = 1; gap < n; gap *= 2)
    {
        int g = gap;
        int[] r = rank;  // capture for closure

        Array.Sort(sa, (a, b) =>
        {
            if (r[a] != r[b]) return r[a] - r[b];
            int ra = a + g < n ? r[a + g] : -1;
            int rb = b + g < n ? r[b + g] : -1;
            return ra - rb;
        });

        tmp[sa[0]] = 0;
        for (int i = 1; i < n; i++)
        {
            tmp[sa[i]] = tmp[sa[i - 1]];
            int prev = sa[i - 1], curr = sa[i];
            int rp2 = prev + g < n ? r[prev + g] : -1;
            int rc2 = curr + g < n ? r[curr + g] : -1;
            if (r[curr] != r[prev] || rc2 != rp2) tmp[sa[i]]++;
        }
        rank = (int[])tmp.Clone();
    }
    return sa;
}
Python
def build_suffix_array(s: str) -> list[int]:
    n = len(s)
    sa = sorted(range(n), key=lambda i: s[i:])
    return sa

# Simple O(n² log n) version for clarity
# For competitive programming, use SA-IS or DC3 for O(n)

def build_sa_efficient(s: str) -> list[int]:
    """O(n log² n) prefix doubling."""
    n = len(s)
    sa = list(range(n))
    rank = [ord(c) for c in s]

    gap = 1
    while gap < n:
        key = lambda i: (rank[i], rank[i + gap] if i + gap < n else -1)
        sa.sort(key=key)
        tmp = [0] * n
        tmp[sa[0]] = 0
        for i in range(1, n):
            tmp[sa[i]] = tmp[sa[i-1]]
            if key(sa[i]) != key(sa[i-1]):
                tmp[sa[i]] += 1
        rank = tmp
        gap *= 2
    return sa

sa = build_suffix_array("banana$")
print(sa)  # [6, 5, 3, 1, 0, 4, 2]
Section 3 — LCP Array (Kasai's Algorithm)

Given the suffix array, Kasai's algorithm builds the LCP array in O(n) using the key insight:

C#
// Kasai's O(n) LCP array given suffix array sa
static int[] BuildLcp(string s, int[] sa)
{
    int n = s.Length;
    int[] rank = new int[n], lcp = new int[n];
    for (int i = 0; i < n; i++) rank[sa[i]] = i;

    int k = 0;
    for (int i = 0; i < n; i++) {
        if (rank[i] == 0) { k = 0; continue; }
        int j = sa[rank[i] - 1];  // previous suffix in sorted order
        while (i + k < n && j + k < n && s[i + k] == s[j + k]) k++;
        lcp[rank[i]] = k;
        if (k > 0) k--;  // key: LCP[rank[i+1]] >= k-1
    }
    return lcp;
}
Python
def build_lcp_array(s: str, sa: list[int]) -> list[int]:
    n = len(s)
    rank = [0] * n
    for i in range(n):
        rank[sa[i]] = i

    lcp = [0] * n
    k = 0
    for i in range(n):
        if rank[i] == 0:
            k = 0
            continue
        j = sa[rank[i] - 1]    # previous suffix in sorted order
        while i + k < n and j + k < n and s[i+k] == s[j+k]:
            k += 1
        lcp[rank[i]] = k
        if k:
            k -= 1   # key: LCP[rank[i+1]] >= k-1
    return lcp
Section 4 — Applications

4.1 — Longest Repeated Substring

The maximum value in the LCP array is the length of the longest repeated substring.

C#
static string LongestRepeatedSubstring(string s)
{
    string t = s + '$';
    var sa  = BuildSuffixArray(t);
    var lcp = BuildLcp(t, sa);
    int maxLen = 0, idx = 0;
    for (int i = 0; i < lcp.Length; i++)
        if (lcp[i] > maxLen) { maxLen = lcp[i]; idx = i; }
    return s.Substring(sa[idx], maxLen);
}
Console.WriteLine(LongestRepeatedSubstring("banana"));  // "ana"
Console.WriteLine(LongestRepeatedSubstring("abcabc")); // "abc"
Python
def longest_repeated_substring(s: str) -> str:
    sa = build_suffix_array(s + '$')
    lcp = build_lcp_array(s + '$', sa)
    max_lcp = max(lcp)
    idx = lcp.index(max_lcp)
    return s[sa[idx] : sa[idx] + max_lcp]

print(longest_repeated_substring("banana"))  # "ana"
print(longest_repeated_substring("abcabc"))  # "abc"

4.2 — Count Distinct Substrings

C#
static long CountDistinctSubstrings(string s)
{
    string t = s + '$';
    int n = s.Length;
    var sa  = BuildSuffixArray(t);
    var lcp = BuildLcp(t, sa);
    return (long)n*(n+1)/2 - lcp.Sum();
}
Console.WriteLine(CountDistinctSubstrings("aab"));  // 5
Python
def count_distinct_substrings(s: str) -> int:
    n = len(s)
    sa = build_suffix_array(s + '$')
    lcp = build_lcp_array(s + '$', sa)
    total = n * (n + 1) // 2
    return total - sum(lcp)

print(count_distinct_substrings("aab"))   # 5 (a, a, b, aa, ab, aab) → unique: 5
📌 Key Takeaways:
  • SA construction: O(n log n) with prefix doubling; O(n) with SA-IS
  • LCP construction: O(n) with Kasai's algorithm using SA and inverse SA
  • Applications: longest repeated substring, count distinct substrings, string matching, bioinformatics
  • In Python, sorted(range(n), key=lambda i: s[i:]) gives a simple O(n² log n) SA
