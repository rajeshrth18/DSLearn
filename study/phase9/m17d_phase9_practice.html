Problem 1 of 10
#137 — Single Number II Medium

Every element appears 3 times except one element which appears exactly once. Find the single element in O(n) time and O(1) space.

nums = [2,2,3,2] → 3

Track bits that have appeared once (ones) and twice (twos). Update both with each new number so that thrice-seen bits vanish.
C# 
public int SingleNumber(int[] nums) {
    int ones = 0, twos = 0;
    foreach (int n in nums) {
        ones = (ones ^ n) & ~twos;
        twos = (twos ^ n) & ~ones;
    }
    return ones; // bits seen exactly once
}
Python 
def singleNumber(nums):
    ones = twos = 0
    for n in nums:
        ones = (ones ^ n) & ~twos
        twos = (twos ^ n) & ~ones
    return ones
Problem 2 of 10
#338 — Counting Bits Easy

Return an array ans of length n+1 where ans[i] = number of 1s in binary representation of i. O(n) time and space, no built-in popcount per element.

i >> 1 has the same bits as i except the last bit. So dp[i] = dp[i >> 1] + (i & 1).
C# 
public int[] CountBits(int n) {
    var dp = new int[n + 1];
    for (int i = 1; i <= n; i++) dp[i] = dp[i >> 1] + (i & 1);
    return dp;
}
Python 
def countBits(n):
    dp = [0] * (n+1)
    for i in range(1, n+1): dp[i] = dp[i>>1] + (i&1)
    return dp
Problem 3 of 10
#371 — Sum of Two Integers Medium

Calculate a + b without using + or - operators.

XOR = sum without carry. AND shifted left = carry. Repeat until carry = 0. In Python, mask to 32 bits to handle two's complement negatives.
C# 
public int GetSum(int a, int b) {
    while (b != 0) { int c = (a & b) << 1; a ^= b; b = c; }
    return a;
}
Python 
def getSum(a, b):
    MASK = 0xFFFFFFFF; MAX = 0x7FFFFFFF
    while b & MASK:
        carry = ((a & b) << 1) & MASK
        a = (a ^ b) & MASK; b = carry
    return a if a <= MAX else ~(a ^ MASK)

#96 — Unique Binary Search Trees Medium

Given n, return the number of structurally unique BSTs which have exactly n nodes. (Answer = Catalan number Cn.)

For each root r in [1..n], left subtree uses r-1 nodes, right uses n-r. dp[n] = Σ dp[i] * dp[n-1-i] for i in 0..n-1.
C# 
public int NumTrees(int n) {
    var dp = new long[n + 1]; dp[0] = dp[1] = 1;
    for (int i = 2; i <= n; i++)
        for (int j = 0; j < i; j++) dp[i] += dp[j] * dp[i - 1 - j];
    return (int)dp[n];
}
Python 
def numTrees(n):
    dp = [0] * (n+1); dp[0] = dp[1] = 1
    for i in range(2, n+1):
        for j in range(i): dp[i] += dp[j] * dp[i-1-j]
    return dp[n]
Problem 5 of 10
#50 — Pow(x, n) Medium

Implement pow(x, n) where n can be negative. Must be O(log n).

Binary exponentiation: if n is even, pow(x,n) = pow(x*x, n/2). If odd, x * pow(x, n-1). Handle n<0: pow(1/x, -n).
C# 
public double MyPow(double x, int n) {
    long N = n;
    if (N < 0) { x = 1 / x; N = -N; }
    double res = 1;
    while (N > 0) {
        if ((N & 1) == 1) res *= x;
        x *= x; N >>= 1;
    }
    return res;
}
Python 
def myPow(x, n):
    if n < 0: x, n = 1/x, -n
    res = 1
    while n:
        if n & 1: res *= x
        x *= x; n >>= 1
    return res

#1201 — Ugly Number III Hard

Given integers a, b, c, find the n-th positive integer divisible by a, b, or c. (Constraints: a,b,c ≤ 10^9, n ≤ 10^9)

Binary search on X. Count ugly numbers ≤ X using inclusion-exclusion: X/a + X/b + X/c - X/lcm(a,b) - X/lcm(b,c) - X/lcm(a,c) + X/lcm(a,b,c). lcm(a,b) = a/gcd(a,b)*b.
C# 
public int NthUglyNumber(int n, long a, long b, long c) {
    long Gcd(long x, long y) => y==0?x:Gcd(y,x%y);
    long Lcm(long x, long y) => x/Gcd(x,y)*y;
    long ab=Lcm(a,b), bc=Lcm(b,c), ac=Lcm(a,c), abc=Lcm(ab,c);
    long Count(long x) => x/a+x/b+x/c-x/ab-x/bc-x/ac+x/abc;
    long lo=1, hi=2_000_000_000L;
    while(lo<hi){long mid=(lo+hi)/2;if(Count(mid)>=n)hi=mid;elselo=mid+1;}
    return (int)lo;
}
Python 
from math import gcd

def nthUglyNumber(n, a, b, c):
    def lcm(x, y): return x // gcd(x, y) * y
    ab = lcm(a, b); bc = lcm(b, c); ac = lcm(a, c)
    abc = lcm(ab, c)
    def count(x):
        return x//a + x//b + x//c - x//ab - x//bc - x//ac + x//abc
    lo, hi = 1, 2*10**9
    while lo < hi:
        mid = (lo + hi) // 2
        if count(mid) >= n: hi = mid
        else: lo = mid + 1
    return lo

Problem 7 of 10
#587 — Erect the Fence Hard

Given coordinates of trees, find all trees on the convex hull perimeter (including collinear points on edges). Return the convex hull tree indices.

Monotone chain convex hull. Include collinear points on hull edges: when building, pop only when cross product is strictly NEGATIVE (not <= 0 as usual). Use a set to track all hull points.
C# 
public int[][] OuterTrees(int[][] trees) {
    Array.Sort(trees,(a,b)=>a[0]==b[0]?a[1]-b[1]:a[0]-b[0]);
    int n=trees.Length;
    long Cross(int[] O,int[] A,int[] B)=>
        (long)(A[0]-O[0])*(B[1]-O[1])-(long)(A[1]-O[1])*(B[0]-O[0]);
    var h=new List<int[]>();
    foreach(var p in trees){while(h.Count>=2&&Cross(h[^2],h[^1],p)<0)h.RemoveAt(h.Count-1);h.Add(p);}
    int low=h.Count;
    for(int i=n-2;i>=0;i--){while(h.Count>low&&Cross(h[^2],h[^1],trees[i])<0)h.RemoveAt(h.Count-1);h.Add(trees[i]);}
    return h.DistinctBy(p => (p[0],p[1])).ToArray();
}
Python 
def outerTrees(trees):
    pts = sorted(map(tuple, trees))
    def cross(O, A, B):
        return (A[0]-O[0])*(B[1]-O[1]) - (A[1]-O[1])*(B[0]-O[0])
    hull = []
    for p in pts + pts[-2::-1]:  # lower + upper
        while len(hull) >= 2 and cross(hull[-2], hull[-1], p) < 0:  # strict < keeps collinear
            hull.pop()
        hull.append(p)
    return list(map(list, set(hull)))
Problem 8 of 10
#149 — Max Points on a Line Hard

Given n points, find the maximum number of points that lie on the same straight line. Return the count.

For each anchor point P[i], hash slopes as reduced fractions (dy/gcd, dx/gcd) to group collinear points. Handle vertical lines, duplicates separately. O(n²) total.
C# 
public int MaxPoints(int[][] pts) {
    int n=pts.Length; if(n<=2) return n; int ans=1;
    long Gcd(long x,long y)=>y==0?x:Gcd(y,x%y);
    for(int i=0;i<n;i++) {
        var cnt=new Dictionary<(long,long),int>(); int dup=1;
        for(int j=i+1;j<n;j++) {
            long dx=pts[j][0]-pts[i][0], dy=pts[j][1]-pts[i][1];
            if(dx==0&&dy==0){dup++;continue;}
            long g=Gcd(Math.Abs(dx),Math.Abs(dy)); dx/=g; dy/=g;
            if(dx<0){dx=-dx;dy=-dy;}else if(dx==0)dy=Math.Abs(dy);
            cnt[(dx,dy)]=cnt.GetValueOrDefault((dx,dy))+1;
        }
        ans=Math.Max(ans,(cnt.Count>0?cnt.Values.Max():0)+dup);
    }
    return ans;
}
Python 
from math import gcd
from collections import defaultdict

def maxPoints(points):
    n = len(points)
    if n <= 2: return n
    ans = 0
    for i in range(n):
        cnt = defaultdict(lambda: 0)
        dup = 1
        for j in range(i+1, n):
            dx = points[j][0] - points[i][0]
            dy = points[j][1] - points[i][1]
            if dx == 0 and dy == 0: dup += 1; continue
            g = gcd(abs(dx), abs(dy))
            dx //= g; dy //= g
            if dx < 0: dx, dy = -dx, -dy  # normalize sign
            elif dx == 0: dy = abs(dy)
            cnt[(dx,dy)] += 1
        best = max(cnt.values(), default=0)
        ans = max(ans, best + dup)
    return ans

#847 — Shortest Path Visiting All Nodes Hard

An undirected connected graph of n nodes. Return the length of the shortest path that visits every node. You may start and stop at any node, revisit nodes and reuse edges.

BFS with state = (currentNode, visitedMask). Start BFS from ALL nodes simultaneously (all nodes with their single-node mask). The first time a state with full mask is reached, that's the answer.
C# 
public int ShortestPathLength(int[][] graph) {
    int n=graph.Length, full=(1<<n)-1;
    var vis=new HashSet<(int,int)>();
    var q=new Queue<(int nd,int mask,int dist)>();
    for(int i=0;i<n;i++){vis.Add((i,1<<i));q.Enqueue((i,1<<i,0));}
    while(q.Count>0) {
        var(node,mask,dist)=q.Dequeue();
        foreach(var nb in graph[node]) {
            int nm=mask|(1<<nb);
            if(nm==full) return dist+1;
            if(vis.Add((nb,nm))) q.Enqueue((nb,nm,dist+1));
        }
    }
    return -1;
}
Python 
from collections import deque

def shortestPathLength(graph):
    n = len(graph); full = (1<<n)-1
    if full == 0: return 0
    vis = set()
    q = deque()
    for i in range(n):
        state = (i, 1<<i)
        vis.add(state); q.append((i, 1<<i, 0))
    while q:
        node, mask, dist = q.popleft()
        for nb in graph[node]:
            nm = mask | (1<<nb)
            if nm == full: return dist+1
            if (nb, nm) not in vis:
                vis.add((nb, nm)); q.append((nb, nm, dist+1))
    return -1

#1044 — Longest Duplicate Substring Hard

Given a string s, return the longest substring that occurs at least twice. If no such substring exists, return empty string.

Binary search on length L. For each L, check if any substring of length L appears twice using rolling hash (Rabin-Karp). If a duplicate exists, try longer; else shorter.
C# 
public string LongestDupSubstring(string s) {
    int n=s.Length; const long MOD=1_000_000_007, BASE=31;
    string Check(int L) {
        long pw=1; for(int i=0;i<L-1;i++)pw=pw*BASE%MOD;
        long h=0; for(int i=0;i<L;i++)h=(h*BASE+(s[i]-'a'+1))%MOD;
        var seen=new Dictionary<long,List<int>>{{h,new{0}}};
        for(int i=1;i<=n-L;i++){
            h=(h-(s[i-1]-'a'+1)*pw%MOD+MOD)*BASE%MOD;
            h=(h+(s[i+L-1]-'a'+1))%MOD;
            if(seen.TryGetValue(h,out var lst))
                foreach(var j in lst)
                    if(s.Substring(i,L)==s.Substring(j,L))return s.Substring(i,L);
            seen.TryAdd(h,new());seen[h].Add(i);
        }
        return null;
    }
    int lo=1,hi=n-1; string ans="";
    while(lo<=hi){int mid=(lo+hi)/2;var r=Check(mid);if(r!=null){ans=r;lo=mid+1;}else hi=mid-1;}
    return ans;
}
Python 
def longestDupSubstring(s):
    n = len(s); A = [ord(c) - ord('a') for c in s]
    MOD, BASE = (1<<61)-1, 31

    def has_dup(L):
        h = 0; pw = pow(BASE, L, MOD)
        seen = {}
        for i in range(L): h = (h * BASE + A[i]) % MOD
        seen[h] = [0]
        for i in range(1, n-L+1):
            h = (h * BASE - A[i-1] * pw + A[i+L-1]) % MOD
            if h in seen:
                cur = s[i:i+L]
                if any(s[j:j+L] == cur for j in seen[h]): return cur
                seen[h].append(i)
            else: seen[h] = [i]
        return ''

    lo, hi, ans = 1, n-1, ''
    while lo <= hi:
        mid = (lo + hi) // 2
        cand = has_dup(mid)
        if cand: ans = cand; lo = mid + 1
        else: hi = mid - 1
    return ans
