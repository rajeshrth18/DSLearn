Section 1 — Permutations (LC 46)

Generate all n! permutations of a list of distinct integers.

Approach: At each step, choose an unused element. Track used elements with a boolean array or by swapping in-place.

Permutations of [1, 2, 3]: [] +-- [1] ¦ +-- [1,2] ? [1,2,3] ? ¦ +-- [1,3] ? [1,3,2] ? +-- [2] ¦ +-- [2,1] ? [2,1,3] ? ¦ +-- [2,3] ? [2,3,1] ? +-- [3] +-- [3,1] ? [3,1,2] ? +-- [3,2] ? [3,2,1] ? ? 6 = 3! results
C#
public IList<IList<int>> Permute(int[] nums)
{
    var results = new List<IList<int>>();
    bool[] used = new bool[nums.Length];
    var path = new List<int>();

    void Backtrack()
    {
        if (path.Count == nums.Length) { results.Add(new List<int>(path)); return; }
        for (int i = 0; i < nums.Length; i++)
        {
            if (used[i]) continue;
            used[i] = true;  path.Add(nums[i]);
            Backtrack();
            used[i] = false; path.RemoveAt(path.Count - 1);
        }
    }
    Backtrack();
    return results;
}
Python
def permute(nums: list[int]) -> list[list[int]]:
    results, path, used = [], [], [False] * len(nums)

    def backtrack():
        if len(path) == len(nums):
            results.append(path[:])
            return
        for i, n in enumerate(nums):
            if used[i]: continue
            used[i] = True;  path.append(n)
            backtrack()
            used[i] = False; path.pop()

    backtrack()
    return results

print(permute([1, 2, 3]))  # 6 permutations
Section 2 — Subsets / Power Set (LC 78)

Generate all 2^n subsets of a list of distinct integers.

Approach: At each index, choose to include or skip the element. Or iterate and extend current subsets.

C#
public IList<IList<int>> Subsets(int[] nums)
{
    var results = new List<IList<int>>();
    var path    = new List<int>();

    void Backtrack(int start)
    {
        results.Add(new List<int>(path));  // add subset at every node
        for (int i = start; i < nums.Length; i++)
        {
            path.Add(nums[i]);
            Backtrack(i + 1);
            path.RemoveAt(path.Count - 1);
        }
    }
    Backtrack(0);
    return results;
}
Python
def subsets(nums: list[int]) -> list[list[int]]:
    results, path = [], []

    def backtrack(start: int):
        results.append(path[:])   # record subset at every level
        for i in range(start, len(nums)):
            path.append(nums[i])
            backtrack(i + 1)
            path.pop()

    backtrack(0)
    return results

print(subsets([1, 2, 3]))  # [[], [1], [1,2], [1,2,3], [1,3], [2], [2,3], [3]]
Section 3 — Combination Sum (LC 39)

Find all unique combinations in an array whose sum equals target.

C#
public static IList<IList<int>> CombinationSum(int[] candidates, int target)
{
    var results = new List<IList<int>>(); var path = new List<int>();
    Array.Sort(candidates);
    void Bt(int start, int rem) {
        if (rem == 0) { results.Add(new List<int>(path)); return; }
        for (int i = start; i < candidates.Length; i++) {
            if (candidates[i] > rem) break;   // pruning
            path.Add(candidates[i]);
            Bt(i, rem - candidates[i]);          // i not i+1 ? allow reuse
            path.RemoveAt(path.Count - 1);
        }
    }
    Bt(0, target); return results;
}
Console.WriteLine(string.Join(", ", CombinationSum(new[]{2,3,6,7}, 7)
    .Select(g => "["+string.Join(",",g)+"]"))); // [2,2,3], [7]
Python
def combination_sum(candidates: list[int], target: int) -> list[list[int]]:
    results, path = [], []
    candidates.sort()

    def backtrack(start: int, remaining: int):
        if remaining == 0:
            results.append(path[:])
            return
        for i in range(start, len(candidates)):
            if candidates[i] > remaining: break   # pruning
            path.append(candidates[i])
            backtrack(i, remaining - candidates[i])   # i not i+1 ? reuse
            path.pop()

    backtrack(0, target)
    return results

print(combination_sum([2,3,6,7], 7))  # [[2,2,3],[7]]
Section 4 — Letter Combinations of a Phone Number (LC 17)
C#
public IList<string> LetterCombinations(string digits)
{
    if (string.IsNullOrEmpty(digits)) return new List<string>();
    var phone = new Dictionary<char, string>
    {
        ['2']="abc", ['3']="def", ['4']="ghi", ['5']="jkl",
        ['6']="mno", ['7']="pqrs", ['8']="tuv", ['9']="wxyz"
    };
    var results = new List<string>();
    var sb = new System.Text.StringBuilder();

    void Backtrack(int idx)
    {
        if (idx == digits.Length) { results.Add(sb.ToString()); return; }
        foreach (char c in phone[digits[idx]])
        {
            sb.Append(c);
            Backtrack(idx + 1);
            sb.Remove(sb.Length - 1, 1);
        }
    }
    Backtrack(0);
    return results;
}
Python
def letter_combinations(digits: str) -> list[str]:
    if not digits: return []
    phone = {'2':"abc",'3':"def",'4':"ghi",'5':"jkl",
             '6':"mno",'7':"pqrs",'8':"tuv",'9':"wxyz"}
    results, path = [], []

    def backtrack(idx: int):
        if idx == len(digits):
            results.append("".join(path))
            return
        for c in phone[digits[idx]]:
            path.append(c)
            backtrack(idx + 1)
            path.pop()

    backtrack(0)
    return results

print(letter_combinations("23"))
# ["ad","ae","af","bd","be","bf","cd","ce","cf"]
?? Pattern Summary for Interview:
  • Permutations: All orderings of elements — use used[] array, no start index
  • Subsets/Combinations: No repeated use, order doesn't matter — use start index, pass i+1
  • Combination Sum (reuse): Same as above but pass i instead of i+1
  • Sort + pruning: Sort candidates first so you can break early when candidate > remaining
