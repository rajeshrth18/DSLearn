#1192 — Critical Connections in a Network

There are n servers (0 to n-1) connected by undirected edges. A critical connection is one whose removal makes some servers unreachable. Return all critical connections.

Run Tarjan's bridge-finding algorithm. Edge (u,v) is a bridge if low[v] > disc[u]. The "low" value is the earliest discovery time reachable from v's subtree.
C# 
public IList<IList<int>> CriticalConnections(int n, IList<IList<int>> connections)
{
    var adj = Enumerable.Range(0, n).Select(_ => new List<int>()).ToArray();
    foreach (var c in connections) { adj[c[0]].Add(c[1]); adj[c[1]].Add(c[0]); }
    int[] disc = new int[n], low = new int[n]; Array.Fill(disc, -1);
    int t = 0; var res = new List<IList<int>>();
    void Dfs(int u, int p) {
        disc[u] = low[u] = t++;
        foreach (int v in adj[u]) {
            if (disc[v] == -1) { Dfs(v, u); low[u] = Math.Min(low[u], low[v]);
                if (low[v] > disc[u]) res.Add(new[] { u, v }); }
            else if (v != p) low[u] = Math.Min(low[u], disc[v]);
        }
    }
    for (int i = 0; i < n; i++) if (disc[i] == -1) Dfs(i, -1);
    return res;
}
Python

            elif v!=p: low[u]=min(low[u],disc[v])
    for i in range(n):
        if disc[i]==-1: dfs(i,-1)
    return res

#785 — Is Graph Bipartite?

Given an undirected graph, return true if it is bipartite (2-colorable). The graph may be disconnected.

Try to 2-color the graph using BFS. Assign alternating colors (0/1) to each node. If you encounter a neighbor with the same color as the current node, it's not bipartite. Run for every unvisited node to handle disconnected components.
C# 
public bool IsBipartite(int[][] graph)
{
    int n = graph.Length;
    int[] color = new int[n]; Array.Fill(color, -1);
    for (int i = 0; i < n; i++)
    {
        if (color[i] != -1) continue;
        var q = new Queue<int>(); q.Enqueue(i); color[i] = 0;
        while (q.Count > 0) {
            int u = q.Dequeue();
            foreach (int v in graph[u]) {
                if (color[v] == -1) { color[v] = 1 - color[u]; q.Enqueue(v); }
                else if (color[v] == color[u]) return false;
            }
        }
    }
    return true;
}
Python 
from collections import deque
def isBipartite(graph):
    n = len(graph); color = [-1] * n
    for i in range(n):
        if color[i] != -1: continue
        q = deque([i]); color[i] = 0
        while q:
            u = q.popleft()
            for v in graph[u]:
                if color[v] == -1: color[v] = 1-color[u]; q.append(v)
                elif color[v] == color[u]: return False
    return True

#207 — Course Schedule

Given numCourses and a list of [course, prerequisite] pairs, return true if all courses can be finished (i.e., no cycle in the prerequisite graph).

Build a directed graph: prerequisite → course. Detect cycles using Kahn's algorithm (BFS topological sort) or DFS with 3-color marking (white/gray/black). If a cycle exists, return false.
C# 
public bool CanFinish(int n, int[][] pre)
{
    var adj = Enumerable.Range(0, n).Select(_ => new List<int>()).ToArray();
    int[] indeg = new int[n];
    foreach (var p in pre) { adj[p[1]].Add(p[0]); indeg[p[0]]++; }
    var q = new Queue<int>();
    for (int i = 0; i < n; i++) if (indeg[i] == 0) q.Enqueue(i);
    int count = 0;
    while (q.Count > 0) {
        int u = q.Dequeue(); count++;
        foreach (int v in adj[u]) if (--indeg[v] == 0) q.Enqueue(v);
    }
    return count == n;
}
Python 
def canFinish(numCourses, prerequisites):
    adj = defaultdict(list); indeg = [0] * numCourses
    for c, p in prerequisites: adj[p].append(c); indeg[c] += 1
    q = deque(i for i in range(numCourses) if indeg[i]==0); count = 0
    while q:
        u = q.popleft(); count += 1
        for v in adj[u]:
            indeg[v] -= 1
            if indeg[v] == 0: q.append(v)
    return count == numCourses

#332 — Reconstruct Itinerary

Given a list of airline tickets [from, to], reconstruct the itinerary starting from "JFK". Use all tickets exactly once. If multiple valid itineraries, return the one with lexicographic smallest order.

This is an Euler path problem on a directed graph. Sort adjacency lists alphabetically. Use Hierholzer's algorithm (DFS with stack). When a node has no more neighbors, prepend it to the result. C# 
public IList<string> FindItinerary(IList<IList<string>> tickets) {
    var adj = new Dictionary<string, LinkedList<string>>();
    foreach (var t in tickets.OrderBy(t => t[1])) {
        if (!adj.ContainsKey(t[0])) adj[t[0]] = new();
        adj[t[0]].AddLast(t[1]);
    }
    var res = new LinkedList<string>();
    void Dfs(string u) {
        while (adj.TryGetValue(u, out var nb) && nb.Count > 0)
            { var nx=nb.First!.Value; nb.RemoveFirst(); Dfs(nx); }
        res.AddFirst(u);
    }
    Dfs("JFK"); return res.ToList();
}
// ["JFK","MUC","LHR","SFO","SJC"]
Python 
def findItinerary(tickets):
    adj = defaultdict(list)
    for u, v in sorted(tickets, reverse=True): adj[u].append(v)
    res = []; stack = ["JFK"]
    while stack:
        while adj[stack[-1]]: stack.append(adj[stack[-1]].pop())
        res.append(stack.pop())
    return res[::-1]

#753 — Cracking the Safe

A safe's combination is a sequence of n digits (0–k-1). The safe unlocks if any n consecutive digits match the combination. Find the shortest string that contains ALL possible combinations as substrings.

This is a De Bruijn sequence — an Euler circuit problem. Build a graph where nodes are (n-1)-digit strings. Add an edge from node s to s[1:]+d for each digit d (representing the n-digit password s+d). Find an Euler circuit in this graph (all nodes have equal in/out degree). C# 
public string CrackSafe(int n, int k) {
    if (n == 1) return string.Concat(Enumerable.Range(0, k));
    var adj = new Dictionary<string, Stack<string>>();
    // Build De Bruijn graph: node = (n-1)-digit string
    var nodes = GetAllStrings(n-1, k);
    foreach (var node in nodes) {
        adj[node] = new Stack<string>();
        for (int d = k-1; d >= 0; d--) adj[node].Push(node[1..]+d);
    }
    string start = new('0', n-1);
    var result = new List<char>(); var stack = new Stack<string>();
    stack.Push(start);
    while (stack.Count > 0) {
        var top = stack.Peek();
        if (adj.TryGetValue(top, out var nb) && nb.Count > 0)
            stack.Push(nb.Pop());
        else { result.Add(stack.Pop()[^1]); }
    }
    result.Reverse();
    return start + new string(result.ToArray());
}
Python 
def crackSafe(n, k):
    if n == 1: return "".join(str(i) for i in range(k))
    adj = defaultdict(list)
    for node in ("".join(p) for p in __import__('itertools').product(map(str, range(k)), repeat=n-1)):
        for d in range(k): adj[node].append(node[1:] + str(d))
    # Hierholzer's on this De Bruijn graph
    start = "0" * (n-1); result = []; stack = [start]
    while stack:
        top = stack[-1]
        if adj[top]: stack.append(adj[top].pop())
        else: result.append(stack.pop()[-1])  # take last digit
    return start + "".join(reversed(result))

#1557 — Minimum Number of Vertices to Reach All Nodes

Given a directed acyclic graph, find the smallest set of vertices from which all other nodes are reachable. Each vertex appears at most once in the answer.

Since the graph is a DAG, any node that can be reached from another node doesn't need to be in the set. C# 
public IList<int> FindSmallestSetOfVertices(int n, IList<IList<int>> edges) {
    var hasIncoming = new HashSet<int>();
    foreach (var e in edges) hasIncoming.Add(e[1]);
    return Enumerable.Range(0, n).Where(i => !hasIncoming.Contains(i)).ToList();
}
// O(V+E) — nodes with in-degree=0 can't be reached from anywhere else
Python 
def findSmallestSetOfVertices(n, edges):
    has_incoming = set()
    for _, v in edges: has_incoming.add(v)
    return [i for i in range(n) if i not in has_incoming]

#802 — Find Eventual Safe States

A node is "safe" if every path starting from it eventually leads to a terminal node (no cycle). Return all safe nodes in sorted order.

A node is safe if it's NOT part of a cycle and doesn't lead to a cycle. Use 3-color DFS (0=unvisited, 1=in-progress, 2=safe). A node is safe if ALL its neighbors are safe. Nodes involved in cycles will remain color=1 (never reach 2).
C# 
public IList<int> EventualSafeNodes(int[][] graph)
{
    int n = graph.Length;
    int[] color = new int[n];  // 0=unvisited,1=visiting,2=safe
    bool IsSafe(int u) {
        if (color[u] > 0) return color[u] == 2;
        color[u] = 1;
        foreach (int v in graph[u]) if (!IsSafe(v)) return false;
        color[u] = 2; return true;
    }
    return Enumerable.Range(0, n).Where(IsSafe).ToList();
}
Python 
def eventualSafeNodes(graph):
    n = len(graph); color = [0] * n
    def is_safe(u):
        if color[u] > 0: return color[u] == 2
        color[u] = 1
        for v in graph[u]:
            if not is_safe(v): return False
        color[u] = 2; return True
    return [i for i in range(n) if is_safe(i)]

#1568 — Minimum Number of Days to Disconnect Island

Given a grid of 0s (water) and 1s (land), return the minimum number of days to disconnect the island (make it so there is NOT exactly 1 connected component of land). You can set one land cell to water per day.

The answer is always 0, 1, or 2. Check: 0 days = already disconnected. 1 day = an articulation point exists (removing it disconnects). 2 days = always possible by removing a corner cell. C# 
public int MinDays(int[][] grid) {
    int m=grid.Length, n=grid[0].Length;
    int CountIslands() {
        bool[][] vis = Enumerable.Range(0,m).Select(_=>new bool[n]).ToArray(); int cnt=0;
        void Dfs(int r,int c) {
            vis[r][c]=true;
            foreach (var (dr,dc) in new[]{(1,0),(-1,0),(0,1),(0,-1)}) {
                int nr=r+dr,nc=c+dc;
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&grid[nr][nc]==1&&!vis[nr][nc]) Dfs(nr,nc);
            }
        }
        for(int r=0;r<m;r++) for(int c=0;c<n;c++) if(grid[r][c]==1&&!vis[r][c]){Dfs(r,c);cnt++;}
        return cnt;
    }
    if(CountIslands()!=1) return 0;
    for(int r=0;r<m;r++) for(int c=0;c<n;c++) if(grid[r][c]==1){
        grid[r][c]=0; if(CountIslands()!=1) return 1; grid[r][c]=1;
    }
    return 2;
}

def minDays(grid):
    m, n = len(grid), len(grid[0])
    def count_islands(g):
        vis = [[False]*n for _ in range(m)]; cnt = 0
        def dfs(r,c):
            vis[r][c] = True
            for dr,dc in [(1,0),(-1,0),(0,1),(0,-1)]:
                nr,nc=r+dr,c+dc
                if 0<=nrand 0<=ncand g[nr][nc]==1 and not vis[nr][nc]: dfs(nr,nc)
        for r in range(m):
            for c in range(n):
                if g[r][c]==1 and not vis[r][c]: dfs(r,c); cnt+=1
        return cnt

            if grid[r][c] == 1:
                grid[r][c] = 0

                grid[r][c] = 1
    return 2  # always possible in 2 days

#743 — Network Delay Time

Given a directed graph with weighted edges (times), a source node k, and n nodes, find the time for all nodes to receive a signal from k. Return -1 if impossible.

Single-source shortest paths from k. Use Dijkstra with a min-heap. The answer is the maximum of all shortest distances from k. C# 
public int NetworkDelayTime(int[][] times, int n, int k) {
    var adj = new Dictionary<int, List<(int,int)>>();
    foreach (var t in times) {
        if(!adj.ContainsKey(t[0])) adj[t[0]]=new();
        adj[t[0]].Add((t[1],t[2]));
    }
    var dist = Enumerable.Range(1,n).ToDictionary(i=>i,_=>int.MaxValue);
    dist[k]=0;
    var pq = new PriorityQueue<(int,int),int>(); pq.Enqueue((0,k),0);
    while(pq.Count>0) {
        var(d,u)=pq.Dequeue(); if(d>dist[u]) continue;
        foreach(var(v,w) in adj.GetValueOrDefault(u,new())) {
            int nd=d+w; if(nd<dist[v]){dist[v]=nd; pq.Enqueue((nd,v),nd);}
        }
    }
    int ans=dist.Values().Max();
    return ans==int.MaxValue?-1:ans;
}
Python 
import heapq
def networkDelayTime(times, n, k):
    adj = defaultdict(list)
    for u,v,w in times: adj[u].append((v,w))
    dist = {i: float('inf') for i in range(1,n+1)}; dist[k] = 0
    heap = [(0, k)]
    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue
        for v, w in adj[u]:
            if dist[u] + w < dist[v]: dist[v] = dist[u]+w; heapq.heappush(heap,(dist[v],v))
    ans = max(dist.values())
    return ans if ans < float('inf') else -1

#2097 — Valid Arrangement of Pairs

Given pairs [start, end], arrange them so that each pair's end equals the next pair's start. Return any valid arrangement.

Each pair is a directed edge. We need an Euler path: a path using every edge exactly once. Find the start node (out-degree − in-degree = 1, or any node if it's a circuit). Run Hierholzer's algorithm. C# 
public int[][] ValidArrangement(int[][] pairs) {
    var adj    = new Dictionary<int, Stack<int>>();
    var indeg  = new Dictionary<int,int>();
    var outdeg = new Dictionary<int,int>();
    foreach(var p in pairs) {
        if(!adj.ContainsKey(p[0])) adj[p[0]]=new();
        adj[p[0]].Push(p[1]);
        outdeg[p[0]]=outdeg.GetValueOrDefault(p[0])+1;
        indeg[p[1]]=indeg.GetValueOrDefault(p[1])+1;
    }
    int start=pairs[0][0];
    foreach(var node in adj.Keys)
        if(outdeg.GetValueOrDefault(node)-indeg.GetValueOrDefault(node)==1){start=node;break;}
    var res=new List<int>(); var stk=new Stack<int>(); stk.Push(start);
    while(stk.Count>0){
        while(adj.TryGetValue(stk.Peek(),out var nb)&&nb.Count>0) stk.Push(nb.Pop());
        res.Add(stk.Pop());
    }
    res.Reverse();
    return Enumerable.Range(0,res.Count-1).Select(i=>new[]{res[i],res[i+1]}).ToArray();
}
Python 
def validArrangement(pairs):
    adj = defaultdict(list)
    indeg = defaultdict(int); outdeg = defaultdict(int)
    for u, v in pairs:
        adj[u].append(v); outdeg[u] += 1; indeg[v] += 1
    # Find start: out-in=1, else any node
    start = pairs[0][0]
    for node in adj:
        if outdeg[node] - indeg[node] == 1: start = node; break
    result = []; stack = [start]
    while stack:
        while adj[stack[-1]]: stack.append(adj[stack[-1]].pop())
        result.append(stack.pop())
    result.reverse()
    return [[result[i], result[i+1]] for i in range(len(result)-1)]
