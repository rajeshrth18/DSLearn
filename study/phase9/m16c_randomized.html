Section 1 — QuickSelect: Kth Largest Element (LC 215)

QuickSelect finds the k-th smallest/largest in O(n) average time using partitioning — without full sorting (which is O(n log n)).

It's the selection version of QuickSort: after each partition, only recurse into the side that contains the target index.

Find 2nd largest in [3, 2, 1, 5, 6, 4]: After 1 partition around pivot=4 (for example): [3, 2, 1, _4_, 5, 6] pivot is at index 3 We need index 4 (0-based for 2nd largest = n-k=4) Recurse only on right side [5, 6] ? answer is 5 ? Average: O(n). Worst case: O(n˛) — use random pivot to avoid.
C#
using System;

public class KthLargest
{
    private static Random _rng = new Random();

    public static int FindKthLargest(int[] nums, int k)
    {
        // Target index in 0-based sorted order (0 = largest)
        int target = nums.Length - k;
        return QuickSelect(nums, 0, nums.Length - 1, target);
    }

    static int QuickSelect(int[] nums, int lo, int hi, int k)
    {
        if (lo == hi) return nums[lo];

        // Random pivot to avoid worst case
        int pivotIdx = _rng.Next(lo, hi + 1);
        Swap(nums, pivotIdx, hi);
        int p = Partition(nums, lo, hi);

        if      (p == k) return nums[p];
        else if (p < k)  return QuickSelect(nums, p + 1, hi, k);
        else             return QuickSelect(nums, lo, p - 1, k);
    }

    static int Partition(int[] nums, int lo, int hi)
    {
        int pivot = nums[hi], wall = lo;
        for (int i = lo; i < hi; i++)
            if (nums[i] <= pivot) Swap(nums, i, wall++);
        Swap(nums, wall, hi);
        return wall;
    }

    static void Swap(int[] a, int i, int j) => (a[i], a[j]) = (a[j], a[i]);
}

// KthLargest.FindKthLargest([3,2,1,5,6,4], 2) ? 5
Python
import random

def find_kth_largest(nums: list[int], k: int) -> int:
    target = len(nums) - k  # index in sorted-ascending order

    def quickselect(lo: int, hi: int) -> int:
        if lo == hi: return nums[lo]
        pivot_idx = random.randint(lo, hi)
        nums[pivot_idx], nums[hi] = nums[hi], nums[pivot_idx]
        p = partition(lo, hi)
        if   p == target: return nums[p]
        elif p < target:  return quickselect(p + 1, hi)
        else:             return quickselect(lo, p - 1)

    def partition(lo: int, hi: int) -> int:
        pivot, wall = nums[hi], lo
        for i in range(lo, hi):
            if nums[i] <= pivot:
                nums[i], nums[wall] = nums[wall], nums[i]
                wall += 1
        nums[wall], nums[hi] = nums[hi], nums[wall]
        return wall

    return quickselect(0, len(nums) - 1)

print(find_kth_largest([3,2,1,5,6,4], 2))  # 5
Section 2 — Reservoir Sampling (LC 382)

Randomly sample k items from a stream of unknown length n, ensuring each element has equal probability k/n — using O(k) space.

Reservoir Sampling — k=1 (pick 1 random element): i=0: reservoir = nums[0] i=1: with prob 1/2 ? replace reservoir with nums[1] i=2: with prob 1/3 ? replace reservoir with nums[2] i=3: with prob 1/4 ? replace reservoir with nums[3] ... After n items, each element has prob 1/n of being selected. Proof by induction: element i survives all future steps with probability (1/i+1) * (i+1/i+2) * ... * (n-1/n) = 1/n
C#
public class Solution  // LC 382 — Linked List Random Node
{
    private ListNode _head;
    private Random _rng = new Random();

    public Solution(ListNode head) { _head = head; }

    public int GetRandom()
    {
        int result = _head.val, i = 2;
        var cur = _head.next;
        while (cur != null)
        {
            if (_rng.Next(i) == 0) result = cur.val;  // prob 1/i
            cur = cur.next; i++;
        }
        return result;
    }
}

// General k-sample reservoir:
static List<int> ReservoirSample(IEnumerable<int> stream, int k)
{
    var res = new List<int>();
    var rng = new Random();
    int i = 0;
    foreach (var x in stream)
    {
        i++;
        if (res.Count < k)           res.Add(x);
        else if (rng.Next(i) < k)  res[rng.Next(k)] = x;
    }
    return res;
}
Python
import random

def reservoir_sample(stream, k: int) -> list:
    """Sample k items uniformly at random from stream (O(k) space)."""
    reservoir = []
    for i, item in enumerate(stream):
        if i < k:
            reservoir.append(item)
        else:
            j = random.randint(0, i)
            if j < k:
                reservoir[j] = item
    return reservoir
Section 3 — Fisher-Yates Shuffle (LC 384)

Produce a uniformly random permutation in O(n) time and O(1) extra space. Each of the n! orderings is equally likely.

C#
public class ShuffleSolution  // LC 384
{
    private int[] _orig, _nums;
    private Random _rng = new Random();

    public ShuffleSolution(int[] nums)
    {
        _orig = (int[])nums.Clone();
        _nums = nums;
    }

    public int[] Reset() { Array.Copy(_orig, _nums, _orig.Length); return _nums; }

    public int[] Shuffle()
    {
        // Fisher-Yates: swap each element with a random later position
        for (int i = _nums.Length - 1; i > 0; i--)
        {
            int j = _rng.Next(i + 1);  // 0 <= j <= i
            (_nums[i], _nums[j]) = (_nums[j], _nums[i]);
        }
        return _nums;
    }
}
Python
import random

class Solution:  # LC 384
    def __init__(self, nums: list[int]):
        self.orig = nums[:]
        self.nums = nums

    def reset(self) -> list[int]:
        self.nums[:] = self.orig[:]
        return self.nums

    def shuffle(self) -> list[int]:
        n = len(self.nums)
        for i in range(n - 1, 0, -1):
            j = random.randint(0, i)  # 0 <= j <= i
            self.nums[i], self.nums[j] = self.nums[j], self.nums[i]
        return self.nums

# random.shuffle() uses the same algorithm internally
