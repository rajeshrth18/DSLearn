Section 1 — The Multi-Pattern Problem

Given a text of length n and k patterns of total length m, find all occurrences of all patterns in the text.

Running KMP separately for each pattern takes O(n·k + m). Aho-Corasick does it in O(n + m + z) where z is the total number of matches — regardless of k. It's KMP generalized to a trie.

Patterns: {"he", "she", "his", "hers"} Build Trie: root / \ h s / \ \ e i h | | \ [he] s e | \ [his] [she] (failure links guide search when mismatch occurs) Key insight: KMP's failure function = Aho-Corasick's failure links on a single-path trie. Aho-Corasick extends this to a trie of all patterns.

2.2 — Step 2: Build Failure Links via BFS

Failure links are the trie equivalent of KMP's failure function. For each node u, fail[u] = the longest proper suffix of the string represented by u that is also a prefix of some pattern in the trie.

C#
public class AhoCorasick
{
    private List<Dictionary<char, int>> _goto = new();
    private List<int> _fail = new();
    private List<List<string>> _output = new();

    public AhoCorasick() { NewNode(); }  // root = 0

    private int NewNode()
    {
        _goto.Add(new()); _fail.Add(0); _output.Add(new());
        return _goto.Count - 1;
    }

    public void Add(string pattern)
    {
        int cur = 0;
        foreach (char c in pattern)
        {
            if (!_goto[cur].ContainsKey(c)) _goto[cur][c] = NewNode();
            cur = _goto[cur][c];
        }
        _output[cur].Add(pattern);
    }

    public void Build()
    {
        var q = new Queue<int>();
        foreach (var (c, child) in _goto[0]) q.Enqueue(child);
        while (q.Count > 0)
        {
            int u = q.Dequeue();
            _output[u].AddRange(_output[_fail[u]]);  // dictionary link
            foreach (var (c, v) in _goto[u])
            {
                int f = _fail[u];
                while (f != 0 && !_goto[f].ContainsKey(c)) f = _fail[f];
                _fail[v] = _goto[f].GetValueOrDefault(c, 0);
                if (_fail[v] == v) _fail[v] = 0;
                q.Enqueue(v);
            }
        }
    }

    public Dictionary<string, List<int>> Search(string text)
    {
        var res = new Dictionary<string, List<int>>();
        int cur = 0;
        for (int i = 0; i < text.Length; i++)
        {
            char c = text[i];
            while (cur != 0 && !_goto[cur].ContainsKey(c)) cur = _fail[cur];
            cur = _goto[cur].GetValueOrDefault(c, 0);
            foreach (string p in _output[cur])
            {
                if (!res.ContainsKey(p)) res[p] = new();
                res[p].Add(i - p.Length + 1);
            }
        }
        return res;
    }
}

// Usage
var ac = new AhoCorasick();
foreach (var p in new[] { "he", "she", "his", "hers" }) ac.Add(p);
ac.Build();
var found = ac.Search("ushers");
// she: [1], he: [2], hers: [2]
Python
from collections import deque, defaultdict

class AhoCorasick:
    def __init__(self):
        # Each node: children dict, fail link, output list
        self.goto = [{}]        # goto[node][char] = next_node
        self.fail = [0]         # failure link
        self.output = [[]]      # patterns ending at this node
        self.size = 1

    def _new_node(self):
        self.goto.append({})
        self.fail.append(0)
        self.output.append([])
        self.size += 1
        return self.size - 1

    def add_pattern(self, pattern: str):
        cur = 0
        for ch in pattern:
            if ch not in self.goto[cur]:
                self.goto[cur][ch] = self._new_node()
            cur = self.goto[cur][ch]
        self.output[cur].append(pattern)

    def build(self):
        """BFS to set failure links."""
        q = deque()
        # Root's children: fail = 0 (root itself)
        for ch, child in self.goto[0].items():
            self.fail[child] = 0
            q.append(child)

        while q:
            u = q.popleft()
            # Merge outputs via fail chain
            self.output[u] += self.output[self.fail[u]]

            for ch, v in self.goto[u].items():
                f = self.fail[u]
                while f != 0 and ch not in self.goto[f]:
                    f = self.fail[f]
                self.fail[v] = self.goto[f].get(ch, 0)
                if self.fail[v] == v:
                    self.fail[v] = 0
                q.append(v)

    def search(self, text: str) -> dict:
        """Returns {pattern: [start_indices]} for all matches."""
        results = defaultdict(list)
        cur = 0
        for i, ch in enumerate(text):
            while cur != 0 and ch not in self.goto[cur]:
                cur = self.fail[cur]
            cur = self.goto[cur].get(ch, 0)
            for pattern in self.output[cur]:
                results[pattern].append(i - len(pattern) + 1)
        return dict(results)

# Usage
ac = AhoCorasick()
for p in ["he", "she", "his", "hers"]:
    ac.add_pattern(p)
ac.build()

print(ac.search("ushers"))
# {'she': [1], 'he': [2], 'hers': [2]}
?? Key Takeaways:
  • Aho-Corasick = KMP failure function extended to a trie of multiple patterns
  • Build: O(total pattern length m); Search: O(n + z) where z = total matches
  • Failure links built via BFS from root — children of root always have fail = root
  • Output links (dictionary links) propagate outputs through the fail chain
  • Used in antivirus engines, DNA search, network intrusion detection
