#28 — Find the Index of First Occurrence in a String

Given strings haystack and needle, return the index of the first occurrence of needle in haystack, or -1 if not present.

Classic KMP application. Build the failure function (LPS array) from needle, then scan haystack. When a mismatch occurs at position j in needle, fall back to lps[j-1] instead of restarting.
C# 
public int StrStr(string haystack, string needle)
{
    if (needle.Length == 0) return 0;
    int n = haystack.Length, m = needle.Length;
    int[] lps = new int[m];
    int len = 0, k = 1;
    while (k < m) {
        if (needle[k] == needle[len]) lps[k++] = ++len;
        else if (len > 0) len = lps[len - 1];
        else lps[k++] = 0;
    }
    int i = 0, j = 0;
    while (i < n) {
        if (haystack[i] == needle[j]) { i++; j++; }
        if (j == m) return i - m;
        else if (i < n && haystack[i] != needle[j])
            j = j > 0 ? lps[j - 1] : (i++ >= 0 ? 0 : 0);
    }
    return -1;
}

Easy Z-Function KMP

#459 — Repeated Substring Pattern

Given a string s, return true if it can be constructed by taking a substring and appending multiple copies of it.

Two approaches: (1) KMP: build lps array; if lps[n-1] > 0 and n % (n - lps[n-1]) == 0, return true. (2) String trick: s is repeated iff s appears in (s+s)[1:-1].
C# 
public bool RepeatedSubstringPattern(string s)
{
    // KMP approach: check if period divides n
    int n = s.Length;
    int[] lps = new int[n];
    int len = 0, i = 1;
    while (i < n) {
        if (s[i] == s[len]) lps[i++] = ++len;
        else if (len > 0) len = lps[len - 1];
        else lps[i++] = 0;
    }
    int period = n - lps[n - 1];
    return lps[n - 1] > 0 && n % period == 0;
}
Python

    return s in (s+s)[1:-1]  # elegant O(n) trick

#1392 — Longest Happy Prefix

A "happy prefix" is a prefix of string s that is also a suffix (but not the whole string). Return the longest such prefix, or "" if none.

The KMP failure function (LPS array) directly gives this answer. lps[n-1] = length of the longest proper prefix that is also a suffix. C# 
public string LongestPrefix(string s) {
    int n = s.Length; int[] lps = new int[n];
    int ln = 0, i = 1;
    while (i < n) {
        if (s[i] == s[ln]) { lps[i++] = ++ln; }
        else if (ln > 0)  { ln = lps[ln-1]; }
        else               { i++; }
    }
    return s.Substring(0, lps[n-1]);  // "" if lps[n-1]==0
}

#438 — Find All Anagrams in a String

Given strings s and p, return all start indices of p's anagrams in s.

Use a sliding window of size len(p). Maintain a frequency count of the window. An anagram exists when the window's frequency matches p's frequency. Track a "matches" counter to avoid full frequency comparison each step.
C# 
public IList<int> FindAnagrams(string s, string p)
{
    var res = new List<int>();
    if (s.Length < p.Length) return res;
    int[] need = new int[26], win = new int[26];
    foreach (char c in p) need[c - 'a']++;
    int m = p.Length;
    for (int i = 0; i < m; i++) win[s[i] - 'a']++;
    if (ArrayEqual(win, need)) res.Add(0);
    for (int i = m; i < s.Length; i++) {
        win[s[i] - 'a']++;
        win[s[i - m] - 'a']--;
        if (ArrayEqual(win, need)) res.Add(i - m + 1);
    }
    return res;
}
bool ArrayEqual(int[] a, int[] b) { for (int i=0;i<26;i++) if(a[i]!=b[i]) return false; return true; }
Python 
from collections import Counter
def findAnagrams(s, p):
    m = len(p); need = Counter(p); win = Counter(s[:m]); res = []
    if win == need: res.append(0)
    for i in range(m, len(s)):
        win[s[i]] += 1; win[s[i-m]] -= 1
        if win[s[i-m]] == 0: del win[s[i-m]]
        if win == need: res.append(i-m+1)
    return res
Medium Manacher Palindrome

#5 — Longest Palindromic Substring

Given string s, return the longest palindromic substring.

Manacher's algorithm finds all palindromes in O(n) using a "mirror" trick similar to KMP/Z-function. For competitive use: expand-around-center is O(n²) but simpler and often fast enough. O(n) Manacher: transform s to "#a#b#a#", C# (Manacher O(n)) 
public string LongestPalindrome(string s) {
    // Transform: "#a#b#a#"
    var sb = new System.Text.StringBuilder("#");
    foreach (char c in s) { sb.Append(c); sb.Append('#'); }
    string t = sb.ToString(); int n = t.Length;
    int[] p = new int[n]; int c = 0, r = 0;
    for (int i = 0; i < n; i++) {
        if (i < r) p[i] = Math.Min(r-i, p[2*c-i]);
        while (i-p[i]-1 >= 0 && i+p[i]+1 < n && t[i-p[i]-1]==t[i+p[i]+1]) p[i]++;
        if (i+p[i] > r) { c=i; r=i+p[i]; }
    }
    int maxLen=p.Max(), center=Array.IndexOf(p, maxLen);
    int start=(center-maxLen)/2;
    return s.Substring(start, maxLen);
}
Python (Manacher O(n)) 
def longestPalindrome(s):
    # Transform: insert '#' between chars and at boundaries
    t = '#' + '#'.join(s) + '#'
    n = len(t); p = [0] * n
    c = r = 0  # center and right boundary of rightmost palindrome
    for i in range(n):
        mirror = 2*c - i
        if i < r: p[i] = min(r-i, p[mirror])
        # Try to expand palindrome centered at i
        while i-p[i]-1 >= 0 and i+p[i]+1 < n and t[i-p[i]-1]==t[i+p[i]+1]:
            p[i] += 1
        if i+p[i] > r: c, r = i, i+p[i]
    # Find the maximum palindrome length
    max_len, center = max((p[i], i) for i in range(n))
    start = (center - max_len) // 2
    return s[start : start + max_len]
Hard Palindrome DP

#132 — Palindrome Partitioning II

Given string s, return the minimum number of cuts needed to partition s into palindromes.

DP: dp[i] = min cuts for s[0..i]. For each i, try all j ≤ i where s[j..i] is a palindrome: dp[i] = min(dp[j-1] + 1). Pre-compute isPalin[j][i] in O(n²). Start with dp[i] = i (max cuts = one char each). C# 
public int MinCut(string s) {
    int n = s.Length;
    bool[,] isP = new bool[n, n];
    for (int i=0; i<n; i++) isP[i,i] = true;
    for (int len=2; len<=n; len++)
        for (int i=0; i<=n-len; i++) {
            int j=i+len-1;
            isP[i,j] = s[i]==s[j] && (len==2 || isP[i+1,j-1]);
        }
    int[] dp = Enumerable.Range(0,n).ToArray();
    for (int i=1; i<n; i++) {
        if(isP[0,i]) { dp[i]=0; continue; }
        for (int j=1; j<=i; j++)
            if(isP[j,i]) dp[i] = Math.Min(dp[i], dp[j-1]+1);
    }
    return dp[n-1];
}
Python 
def minCut(s):
    n = len(s)
    # is_p[i][j] = True if s[i..j] is palindrome
    is_p = [[False]*n for _ in range(n)]
    for i in range(n): is_p[i][i] = True
    for length in range(2, n+1):
        for i in range(n-length+1):
            j = i+length-1
            is_p[i][j] = s[i]==s[j] and (length==2 or is_p[i+1][j-1])
    dp = list(range(n))  # dp[i] = min cuts for s[0..i]
    for i in range(1, n):
        if is_p[0][i]: dp[i] = 0; continue
        for j in range(1, i+1):
            if is_p[j][i]: dp[i] = min(dp[i], dp[j-1]+1)
    return dp[n-1]

#212 — Word Search II

Given a board of characters and a list of words, return all words found in the board (adjacent horizontal/vertical cells). Each cell used at most once per word.

Build a Trie from all words. DFS/backtrack from each cell, following the trie. When you reach a word-end node, add it to results and mark that trie node (to avoid duplicates). Prune branches where the current prefix matches no trie node. C# 
class TN { public Dictionary<char,TN> ch=new(); public string word; }
public IList<string> FindWords(char[][] board, string[] words) {
    var root=new TN();
    foreach(var w in words) {
        var n=root;
        foreach(char c in w){if(!n.ch.ContainsKey(c))n.ch[c]=new TN();n=n.ch[c];}
        n.word=w;
    }
    int m=board.Length,nc=board[0].Length; var res=new List<string>();
    void Dfs(int r,int c,TN node) {
        char ch=board[r][c]; if(ch=='#'||!node.ch.ContainsKey(ch)) return;
        var nxt=node.ch[ch];
        if(nxt.word!=null){res.Add(nxt.word);nxt.word=null;}
        board[r][c]='#';
        foreach(var(dr,dc) in new[]{(1,0),(-1,0),(0,1),(0,-1)}) {
            int nr=r+dr,ncc=c+dc;
            if(nr>=0&&nr<m&&ncc>=0&&ncc<nc) Dfs(nr,ncc,nxt);
        }
        board[r][c]=ch;
        if(nxt.ch.Count==0&&nxt.word==null) node.ch.Remove(ch);
    }
    for(int r=0;r<m;r++) for(int c=0;c<nc;c++) Dfs(r,c,root);
    return res;
}
Python 
def findWords(board, words):
    # Build trie
    trie = {}
    for w in words:
        node = trie
        for c in w: node = node.setdefault(c, {})
        node['$'] = w  # store word at terminal node

    m, n, res = len(board), len(board[0]), []

    def dfs(r, c, node):
        ch = board[r][c]
        if ch not in node: return
        nxt = node[ch]
        if '$' in nxt: res.append(nxt.pop('$'))  # found, pop to avoid dup
        board[r][c] = '#'  # mark visited
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
            nr, nc = r+dr, c+dc
            if 0<=nr<m and 0<=nc<n: dfs(nr, nc, nxt)
        board[r][c] = ch  # restore
        if not nxt: del node[ch]  # prune dead trie nodes

    for r in range(m):
        for c in range(n): dfs(r, c, trie)
    return res

#1044 — Longest Duplicate Substring

Given string s, return the longest substring that appears at least twice. If none, return "".

Binary search on the length L. For each L, use Rabin-Karp rolling hash to check if any substring of length L appears twice (store hashes in a set). If duplicate found, try longer; else shorter. C# 
public string LongestDupSubstring(string s) {
    int n=s.Length; const long MOD=1_000_000_007, BASE=131;
    string Check(int L) {
        long pw=1; for(int i=0;i<L-1;i++) pw=pw*BASE%MOD;
        long h=0; for(int i=0;i<L;i++) h=(h*BASE+s[i])%MOD;
        var seen=new Dictionary<long,List<int>>{{h,new{0}}};
        for(int i=1;i<=n-L;i++) {
            h=(h-s[i-1]*pw%MOD+MOD)*BASE%MOD; h=(h+s[i+L-1])%MOD;
            if(seen.TryGetValue(h,out var lst))
                foreach(var j in lst)
                    if(s.Substring(i,L)==s.Substring(j,L)) return s.Substring(i,L);
            seen.TryAdd(h,new()); seen[h].Add(i);
        }
        return null;
    }
    int lo=1,hi=n-1; string ans="";
    while(lo<=hi) { int mid=(lo+hi)/2; var r=Check(mid); if(r!=null){ans=r;lo=mid+1;}else hi=mid-1; }
    return ans;
}
Python 
def longestDupSubstring(s):
    n = len(s); MOD = (1<<61)-1; BASE = 131

    def check(L):
        h = 0; power = pow(BASE, L, MOD)
        seen = {}
        for i in range(L): h = (h * BASE + ord(s[i])) % MOD
        seen[h] = [0]
        for i in range(1, n-L+1):
            h = (h * BASE - ord(s[i-1]) * power + ord(s[i+L-1])) % MOD
            if h in seen:
                for j in seen[h]:
                    if s[i:i+L] == s[j:j+L]: return i  # verify
                seen[h].append(i)
            else: seen[h] = [i]
        return -1

    lo, hi, ans = 1, n-1, 0; start = 0
    while lo <= hi:
        mid = (lo+hi)//2; idx = check(mid)
        if idx >= 0: ans = mid; start = idx; lo = mid+1
        else: hi = mid-1
    return s[start:start+ans]

#76 — Minimum Window Substring

Given strings s and t, return the minimum window substring of s that contains all characters of t. If none, return "".

Sliding window: expand right until all chars of t are covered. Then shrink from left while still covered. Track the minimum window. Use a `need` counter dict and a `formed` count (how many chars are currently satisfied) to check coverage in O(1).
C# 
public string MinWindow(string s, string t)
{
    var need = new Dictionary<char, int>();
    foreach (char c in t) { need[c] = need.GetValueOrDefault(c) + 1; }
    int have = 0, required = need.Count, l = 0, minLen = int.MaxValue, minL = 0;
    var win = new Dictionary<char, int>();
    for (int r = 0; r < s.Length; r++) {
        char c = s[r]; win[c] = win.GetValueOrDefault(c) + 1;
        if (need.ContainsKey(c) && win[c] == need[c]) have++;
        while (have == required) {
            if (r - l + 1 < minLen) { minLen = r - l + 1; minL = l; }
            char lc = s[l++]; win[lc]--;
            if (need.ContainsKey(lc) && win[lc] < need[lc]) have--;
        }
    }
    return minLen == int.MaxValue ? "" : s.Substring(minL, minLen);
}
Python 
from collections import Counter
def minWindow(s, t):
    need = Counter(t); win = {}; have = required = len(need)
    l = 0; ans = ""
    for r, c in enumerate(s):
        win[c] = win.get(c,0) + 1
        if c in need and win[c] == need[c]: have -= 1
        while have == 0:
            if not ans or r-l+1 < len(ans): ans = s[l:r+1]
            lc = s[l]; win[lc] -= 1; l += 1
            if lc in need and win[lc] < need[lc]: have += 1
    return ans

#214 — Shortest Palindrome

Given string s, find the shortest palindrome you can form by adding characters in front of s.

We need to find the longest palindromic prefix of s. Once we know it (length k), we prepend the reverse of s[k:] to s. To find the longest palindromic prefix: form the string s + "#" + reverse(s), compute KMP failure function. C# 
public string ShortestPalindrome(string s) {
    if (s.Length == 0) return s;
    string combined = s + "#" + new string(s.Reverse().ToArray());
    int n = combined.Length; int[] lps = new int[n];
    int ln=0, i=1;
    while (i < n) {
        if (combined[i]==combined[ln]) { lps[i++]=++ln; }
        else if (ln > 0)              { ln=lps[ln-1]; }
        else                           { i++; }
    }
    int k=lps[n-1];  // length of longest palindromic prefix
    return new string(s.Substring(k).Reverse().ToArray()) + s;
}
// "aacecaaa" → k=7, prepend reverse("a") → "aaacecaaa"
Python 
def shortestPalindrome(s):
    if not s: return s
    combined = s + '#' + s[::-1]  # '#' prevents cross-match
    n = len(combined); lps = [0]*n; ln = 0; i = 1
    while i < n:
        if combined[i] == combined[ln]: lps[i] = ln+1; ln+=1; i+=1
        elif ln: ln = lps[ln-1]
        else: i += 1
    k = lps[-1]  # length of longest palindromic prefix
    return s[k:][::-1] + s  # prepend reverse of non-palindromic suffix
