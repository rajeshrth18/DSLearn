Problem 1 — String Algorithms
Repeated Substring Pattern Easy
LC 459  |  KMP Application

Given a string s, check if it can be constructed by taking a substring of it and appending multiple copies of the substring together.

Hint: Use KMP's LPS array. If the last value in LPS is L and (n % (n - L)) == 0, return true. Alternatively: check if s appears in (s+s)[1:-1].
C#
public bool RepeatedSubstringPattern(string s) {
    string doubled = (s + s).Substring(1, 2*s.Length-2);
    return doubled.Contains(s);
}
// "abcabc" ? true (abc repeated)
Python
def repeated_substring_pattern(s: str) -> bool:
    return s in (s + s)[1:-1]
# True for "abcabc" ? abc repeated twice
Problem 2 — String Algorithms
Find All Anagrams in a String Medium
LC 438  |  Sliding Window + Hashing

Given strings s and p, return all start indices of p's anagrams in s.

Hint: Use a sliding window of length len(p). Maintain character count arrays and compare. O(n) time.
C#
public IList<int> FindAnagrams(string s, string p) {
    var res = new List<int>();
    if (s.Length < p.Length) return res;
    int[] need = new int[26], win = new int[26];
    int m = p.Length;
    foreach (char c in p) need[c-'a']++;
    for (int i = 0; i < m; i++) win[s[i]-'a']++;
    if (win.SequenceEqual(need)) res.Add(0);
    for (int i = m; i < s.Length; i++) {
        win[s[i]-'a']++; win[s[i-m]-'a']--;
        if (win.SequenceEqual(need)) res.Add(i-m+1);
    }
    return res;
}
Python
from collections import Counter

def find_anagrams(s: str, p: str) -> list[int]:
    need, window = Counter(p), Counter(s[:len(p)])
    result = [0] if need == window else []
    for i in range(len(p), len(s)):
        window[s[i]] += 1
        c = s[i - len(p)]
        window[c] -= 1
        if window[c] == 0: del window[c]
        if window == need: result.append(i - len(p) + 1)
    return result
Problem 3 — String Algorithms
Longest Palindromic Substring Medium
LC 5  |  Manacher's or Expand Around Center

Return the longest palindromic substring of s.

Hint: Expand around center for O(n˛) or use Manacher's algorithm for O(n). For each center (including between chars), expand while s[l]==s[r].
C#
public string LongestPalindrome(string s) {
    string best = "";
    for (int i = 0; i < s.Length; i++) {
        foreach (var (l0,r0) in new[]{(i,i),(i,i+1)}) {
            int l=l0, r=r0;
            while (l>=0 && r<s.Length && s[l]==s[r]) {l--;r++;}
            if (r-l-1 > best.Length) best = s.Substring(l+1, r-l-1);
        }
    }
    return best;
}
Python
def longest_palindrome(s: str) -> str:
    best = ""
    for i in range(len(s)):
        for l, r in [(i, i), (i, i+1)]:  # odd and even centers
            while l >= 0 and r < len(s) and s[l] == s[r]:
                l -= 1; r += 1
            if r - l - 1 > len(best):
                best = s[l+1:r]
    return best
Problem 4 — Bit Manipulation
Number of 1 Bits (Hamming Weight) Easy
LC 191  |  Brian Kernighan's Algorithm

Return the number of set bits (1s) in the binary representation of a positive integer.

Hint: n & (n-1) removes the lowest set bit. Count iterations until n == 0.
C#
public int HammingWeight(uint n) {
    int count = 0;
    while (n != 0) { n &= n - 1; count++; }  // clear lowest set bit
    return count;
}
Python
def hamming_weight(n: int) -> int:
    count = 0
    while n:
        n &= n - 1   # clear lowest set bit
        count += 1
    return count
Problem 5 — Bit Manipulation
Sum of Two Integers Without + or - Medium
LC 371  |  XOR + AND + Carry

Calculate the sum of two integers a and b, but you are not allowed to use the operator + or -.

Hint: XOR gives sum without carry. AND shifted left gives the carry. Repeat until no carry remains.
C#
public int GetSum(int a, int b)
{
    while (b != 0)
    {
        int carry = (a & b) << 1;
        a ^= b;
        b = carry;
    }
    return a;
}
Python
def get_sum(a: int, b: int) -> int:
    mask = 0xFFFFFFFF
    while b & mask:
        carry = ((a & b) << 1) & mask
        a = (a ^ b) & mask
        b = carry
    return a if a <= 0x7FFFFFFF else ~(a ^ mask)
Problem 6 — Bit Manipulation
Bitwise AND of Numbers Range Medium
LC 201  |  Common prefix of bits

Given two integers left and right that represent a range, return the bitwise AND of all numbers in this range.

Hint: The answer is the common bit prefix of left and right. Right-shift both until equal, then shift back. Each shifted-off bit becomes 0 because the range spans a power-of-2 boundary.
C#
public int RangeBitwiseAnd(int left, int right) {
    int shift = 0;
    while (left != right) { left >>= 1; right >>= 1; shift++; }
    return left << shift;
}
Python
def range_bitwise_and(left: int, right: int) -> int:
    shift = 0
    while left != right:
        left >>= 1
        right >>= 1
        shift += 1
    return left << shift
Problem 7 — Backtracking
Word Search Medium
LC 79  |  DFS + Backtracking on Grid

Given an m×n grid of characters and a string word, return true if the word exists in the grid. The word can be constructed from sequentially adjacent cells (horizontally or vertically). The same cell may not be used more than once.

Hint: DFS from each cell. Mark cell as visited (e.g., set to '#') before recursing, restore after. Prune if current char doesn't match.
C#
public bool Exist(char[][] board, string word) {
    int rows=board.Length, cols=board[0].Length;
    bool Dfs(int r, int c, int idx) {
        if (idx==word.Length) return true;
        if (r<0||r>=rows||c<0||c>=cols||board[r][c]!=word[idx]) return false;
        char tmp=board[r][c]; board[r][c]='#';
        bool found=Dfs(r+1,c,idx+1)||Dfs(r-1,c,idx+1)||Dfs(r,c+1,idx+1)||Dfs(r,c-1,idx+1);
        board[r][c]=tmp; return found;
    }
    for(int r=0;r<rows;r++) for(int c=0;c<cols;c++) if(Dfs(r,c,0)) return true;
    return false;
}
Python
def exist(board: list[list[str]], word: str) -> bool:
    rows, cols = len(board), len(board[0])

    def dfs(r: int, c: int, idx: int) -> bool:
        if idx == len(word): return True
        if not (0 <= r < rows and 0 <= c < cols): return False
        if board[r][c] != word[idx]: return False
        tmp, board[r][c] = board[r][c], '#'
        found = (dfs(r+1,c,idx+1) or dfs(r-1,c,idx+1) or
                 dfs(r,c+1,idx+1) or dfs(r,c-1,idx+1))
        board[r][c] = tmp
        return found

    return any(dfs(r, c, 0) for r in range(rows) for c in range(cols))
Problem 8 — Backtracking
Palindrome Partitioning Medium
LC 131  |  Backtracking + Palindrome Check

Given a string s, partition it such that every substring of the partition is a palindrome. Return all possible palindrome partitioning.

Hint: Backtrack: at each position, try all substrings starting here. If it's a palindrome, add it to current partition and recurse on the rest.
C#
public IList<IList<string>> Partition(string s) {
    var res = new List<IList<string>>(); var path = new List<string>();
    bool IsPal(int l,int r){while(l<r)if(s[l++]!=s[r--])return false;return true;}
    void Bt(int start) {
        if (start==s.Length){res.Add(new List<string>(path));return;}
        for(int end=start+1;end<=s.Length;end++) {
            if(IsPal(start,end-1)) {
                path.Add(s.Substring(start,end-start));
                Bt(end); path.RemoveAt(path.Count-1);
            }
        }
    }
    Bt(0); return res;
}
Python
def partition(s: str) -> list[list[str]]:
    results, path = [], []

    def is_pal(t: str) -> bool:
        return t == t[::-1]  # or t == t[::-1]

    def backtrack(start: int):
        if start == len(s):
            results.append(path[:])
            return
        for end in range(start + 1, len(s) + 1):
            sub = s[start:end]
            if is_pal(sub):
                path.append(sub)
                backtrack(end)
                path.pop()

    backtrack(0)
    return results

print(partition("aab"))  # [["a","a","b"],["aa","b"]]
Problem 9 — Backtracking
Generate Parentheses Medium
LC 22  |  Classic Backtracking

Given n pairs of parentheses, write a function to generate all combinations of well-formed parentheses.

Hint: Track open and close counts. Add '(' when open < n; add ')' when close < open. Stop when both equal n.
C#
public IList<string> GenerateParenthesis(int n)
{
    var results = new List<string>();
    void Bt(string cur, int open, int close)
    {
        if (cur.Length == n * 2) { results.Add(cur); return; }
        if (open < n)    Bt(cur + "(", open + 1, close);
        if (close < open) Bt(cur + ")", open, close + 1);
    }
    Bt("", 0, 0);
    return results;
}
Python
def generate_parentheses(n: int) -> list[str]:
    results = []
    def bt(cur: str, op: int, cl: int):
        if len(cur) == 2 * n: results.append(cur); return
        if op < n:  bt(cur + "(", op + 1, cl)
        if cl < op: bt(cur + ")", op, cl + 1)
    bt("", 0, 0)
    return results
Problem 10 — Mixed / Hard
Minimum Window Substring Hard
LC 76  |  Sliding Window + HashMap

Given strings s and t, return the minimum window in s which contains all characters of t. Return "" if no such window exists.

Hint: Use a sliding window. Expand right until all chars covered; shrink left while still covered. Track "have" vs "need" counts. O(n) time.
C#
public string MinWindow(string s, string t) {
    var need = new Dictionary<char,int>();
    foreach(char c in t) need[c]=need.GetValueOrDefault(c)+1;
    int have=0,required=need.Count,l=0,minLen=int.MaxValue,minL=0;
    var win=new Dictionary<char,int>();
    for(int r=0;r<s.Length;r++) {
        char c=s[r]; win[c]=win.GetValueOrDefault(c)+1;
        if(need.ContainsKey(c)&&win[c]==need[c]) have++;
        while(have==required) {
            if(r-l+1<minLen){minLen=r-l+1;minL=l;}
            char lc=s[l++]; win[lc]--;
            if(need.ContainsKey(lc)&&win[lc]<need[lc]) have--;
        }
    }
    return minLen==int.MaxValue?"":s.Substring(minL,minLen);
}
// MinWindow("ADOBECODEBANC","ABC") ? "BANC"
Python
from collections import Counter

def min_window(s: str, t: str) -> str:
    if not t: return ""
    need = Counter(t)
    have, required = {}, len(need)
    l, formed = 0, 0
    best = ""

    for r, c in enumerate(s):
        have[c] = have.get(c, 0) + 1
        if c in need and have[c] == need[c]:
            formed += 1
        while formed == required:
            if not best or r - l + 1 < len(best):
                best = s[l:r+1]
            lc = s[l]; have[lc] -= 1; l += 1
            if lc in need and have[lc] < need[lc]:
                formed -= 1
    return best

print(min_window("ADOBECODEBANC", "ABC"))  # "BANC"
