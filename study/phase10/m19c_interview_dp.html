#1 Climbing Stairs Easy LeetCode #70 · Fibonacci variant

Count the number of ways to climb n stairs (can take 1 or 2 steps at a time).

Time: O(n)Space: O(1)Pattern: 1D DP / Fibonacci
dp[i] = dp[i-1] + dp[i-2]. You can arrive at step i by taking 1 step from i-1 or 2 steps from i-2. Base: dp[0]=1, dp[1]=1.
C# 
public int ClimbStairs(int n) {
    int a=1, b=1;
    for(int i=0;i<n-1;i++) (a,b)=(b,a+b);
    return b;
}

    return b

#3 Coin Change Medium LeetCode #322 · Classic unbounded knapsack

Find the fewest coins to make up amount, given coins of different denominations (unlimited supply).

Time: O(n·amount)Space: O(amount)Pattern: BFS or 1D DP
dp[a] = min(dp[a - c] + 1) for each coin c = a. Initialize dp[0]=0, rest to infinity. BFS also works (treating amount as a graph node, each coin as an edge).
C# 
public int CoinChange(int[] coins, int amount) {
    int[] dp=new int[amount+1];
    Array.Fill(dp, amount+1); dp[0]=0;
    for(int a=1;a<=amount;a++)
        foreach(int c in coins)
            if(c<=a) dp[a]=Math.Min(dp[a],dp[a-c]+1);
    return dp[amount]>amount?-1:dp[amount];
}
Python 
def coinChange(coins, amount):
    dp = [float('inf')] * (amount+1); dp[0] = 0
    for a in range(1, amount+1):
        for c in coins:
            if c <= a: dp[a] = min(dp[a], dp[a-c]+1)
    return dp[amount] if dp[amount] != float('inf') else -1

public int NumDecodings(string s) {

        int two=int.Parse(s.Substring(i-1,2));
        if(two>=10&&two<=26) curr+=prev2;
        (prev2,prev1)=(prev1,curr);
    }
    return prev1;
}

#7 Longest Common Subsequence Medium LeetCode #1143 · Classic 2D DP

Find the length of the longest common subsequence of two strings.

Time: O(m·n)Space: O(m·n)Pattern: 2D DP
If chars match: dp[i][j] = dp[i-1][j-1] + 1. Else: dp[i][j] = max(dp[i-1][j], dp[i][j-1]).
C# 
public int LongestCommonSubsequence(string t1, string t2) {
    int m=t1.Length, n=t2.Length;
    int[,] dp=new int[m+1,n+1];
    for(int i=1;i<=m;i++)
        for(int j=1;j<=n;j++){
            if(t1[i-1]==t2[j-1]) dp[i,j]=dp[i-1,j-1]+1;
            else dp[i,j]=Math.Max(dp[i-1,j],dp[i,j-1]);
        }
    return dp[m,n];
}
Python 
def longestCommonSubsequence(text1, text2):

            else: dp[i][j]=max(dp[i-1][j],dp[i][j-1])
    return dp[m][n]
#8 0/1 Knapsack Medium Classic problem · LeetCode #416 (variant)

Given weights and values, fill a knapsack of capacity W to maximize value (each item used at most once).

Time: O(n·W)Space: O(W)Pattern: 2D DP (space-optimized 1D)
Iterate items. For each item, iterate capacity from W down to weight (backward to avoid reusing). dp[w] = max(dp[w], dp[w-wt] + val). Backward iteration ensures each item used at most once.
C# 
public int Knapsack01(int[] wts, int[] vals, int W) {
    int[] dp=new int[W+1];
    for(int i=0;i<wts.Length;i++)
        for(int w=W;w>=wts[i];w--) dp[w]=Math.Max(dp[w],dp[w-wts[i]]+vals[i]);
    return dp[W];
}

public bool CanPartition(int[] nums) {
    int s=nums.Sum(); if(s%2!=0) return false;
    int t=s/2; bool[] dp=new bool[t+1]; dp[0]=true;
    foreach(int n in nums)
        for(int w=t;w>=n;w--) dp[w]|=dp[w-n];
    return dp[t];
}
Python 
def knapsack01(weights, values, W):
    dp = [0] * (W+1)
    for wt, val in zip(weights, values):
        for w in range(W, wt-1, -1):  # backward!
            dp[w] = max(dp[w], dp[w-wt]+val)
    return dp[W]

def canPartition(nums):
    s = sum(nums)
    if s % 2: return False
    target = s // 2; dp = {0}
    for n in nums: dp |= {x+n for x in dp if x+n <= target}
    return target in dp
#9 Edit Distance Hard LeetCode #72 · Classic 2D DP

Find the minimum number of operations (insert, delete, replace) to convert word1 to word2.

Time: O(m·n)Space: O(m·n)Pattern: 2D DP
dp[i][j] = edit distance between word1[:i] and word2[:j]. If chars match: dp[i][j] = dp[i-1][j-1]. Else: 1 + min(dp[i-1][j-1] (replace), dp[i-1][j] (delete), dp[i][j-1] (insert)).
C# 
public int MinDistance(string w1, string w2) {
    int m=w1.Length, n=w2.Length;
    int[] dp=Enumerable.Range(0,n+1).ToArray();
    for(int i=1;i<=m;i++){
        int prev=dp[0]; dp[0]=i;
        for(int j=1;j<=n;j++){
            int tmp=dp[j];
            dp[j]=w1[i-1]==w2[j-1]?prev:1+Math.Min(prev,Math.Min(dp[j],dp[j-1]));
            prev=tmp;
        }
    }
    return dp[n];
}
Python 
def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = list(range(n+1))
    for i in range(1, m+1):
        prev = dp[0]; dp[0] = i
        for j in range(1, n+1):
            temp = dp[j]
            if word1[i-1]==word2[j-1]: dp[j] = prev
            else: dp[j] = 1 + min(prev, dp[j], dp[j-1])
            prev = temp
    return dp[n]

For each position, expand outward for both odd-length (center at i) and even-length (center between i and i+1) palindromes. Track the maximum. O(n˛) total, O(1) space.

        foreach(var p in new[]{Expand(i,i),Expand(i,i+1)})

Python 
def longestPalindrome(s):
    res = ''
    def expand(l, r):
        while l>=0 and r<len(s) and s[l]==s[r]: l-=1; r+=1
        return s[l+1:r]
    for i in range(len(s)):
        for p in [expand(i,i), expand(i,i+1)]:
            if len(p) > len(res): res = p
    return res

#11 Combination Sum Medium LeetCode #39

Find all unique combinations of candidates that sum to target (candidates can be used multiple times).

Time: O(n^(T/M))Space: O(T/M)Pattern: Backtracking
Backtrack. At each step, pick candidates[i] (start), recurse with reduced target, or move to next candidate (i+1) to avoid duplicates. Allows reuse of same element since we stay at same index.
C# 
public IList<IList<int>> CombinationSum(int[] cands, int target) {
    var res=new List<IList<int>>();
    void Bt(int start, List<int> cur, int rem){
        if(rem==0){res.Add(new List<int>(cur));return;}
        for(int i=start;i<cands.Length;i++)
            if(cands[i]<=rem){cur.Add(cands[i]);Bt(i,cur,rem-cands[i]);cur.RemoveAt(cur.Count-1);}
    }
    Bt(0,new(),target); return res;
}
Python 
def combinationSum(candidates, target):
    res = []
    def bt(start, cur, rem):
        if rem == 0: res.append(cur[:]); return
        for i in range(start, len(candidates)):
            if candidates[i] <= rem:
                cur.append(candidates[i]); bt(i, cur, rem-candidates[i]); cur.pop()
    bt(0, [], target); return res
#12 Word Search Medium LeetCode #79

Given an m×n board and a word, return true if the word exists in the grid (adjacent cells, no reuse).

Time: O(m·n·4^L)Space: O(L)Pattern: Backtracking DFS
DFS from each cell matching word[0]. Mark visited by modifying board temporarily (set to '#'), restore on backtrack. Return early if out of bounds, mismatch, or already visited.
C# 
public bool Exist(char[][] board, string word) {
    int m=board.Length, n=board[0].Length;
    bool Dfs(int r,int c,int i){
        if(i==word.Length) return true;
        if(r<0||r>=m||c<0||c>=n||board[r][c]!=word[i]) return false;
        char tmp=board[r][c]; board[r][c]='#';
        bool ok=Dfs(r+1,c,i+1)||Dfs(r-1,c,i+1)||Dfs(r,c+1,i+1)||Dfs(r,c-1,i+1);
        board[r][c]=tmp; return ok;
    }
    for(int r=0;r<m;r++) for(int c=0;c<n;c++) if(Dfs(r,c,0)) return true;
    return false;
}
Python 
def exist(board, word):
    m, n = len(board), len(board[0])
    def dfs(r, c, i):
        if i == len(word): return True
        if r<0 or r>=m or c<0 or c>=n or board[r][c]!=word[i]: return False
        tmp, board[r][c] = board[r][c], '#'
        found = any(dfs(r+dr,c+dc,i+1) for dr,dc in [(1,0),(-1,0),(0,1),(0,-1)])
        board[r][c] = tmp
        return found
    return any(dfs(r,c,0) for r in range(m) for c in range(n))
#13 LRU Cache Medium LeetCode #146 · Design

Design a data structure that follows the LRU (Least Recently Used) cache eviction policy. O(1) get and put.

Time: O(1) eachSpace: O(capacity)Pattern: OrderedDict / Doubly linked list + HashMap
Use Python's OrderedDict (or doubly-linked list + HashMap). On get: if key exists, move to end (most recent), return value. On put: insert/update at end, if over capacity evict first (least recent).
C# 
class LRUCache {
    readonly int _cap;
    readonly LinkedList<(int k,int v)> _list=new();
    readonly Dictionary<int,LinkedListNode<(int,int)>> _map=new();
    public LRUCache(int cap){_cap=cap;}
    public int Get(int key){
        if(!_map.TryGetValue(key,out var n)) return -1;
        _list.Remove(n); _list.AddFirst(n); return n.Value.v;
    }
    public void Put(int key,int val){
        if(_map.TryGetValue(key,out var n)){_list.Remove(n);_map.Remove(key);}
        var node=_list.AddFirst((key,val)); _map[key]=node;
        if(_list.Count>_cap){_map.Remove(_list.Last!.Value.k);_list.RemoveLast();}
    }
}
Python

#14 Merge K Sorted Lists Hard LeetCode #23

Merge k sorted linked lists into one sorted list.

Time: O(n log k)Space: O(k)Pattern: Min-heap
Use a min-heap of (val, index, node). Initialize with the first node of each list. Each step: pop minimum, add to result, push next node from that list. Or use divide & conquer (merge pairs recursively).
C# 
public ListNode MergeKLists(ListNode[] lists) {
    var dummy=new ListNode(0); var cur=dummy;
    var pq=new PriorityQueue<ListNode,int>();
    foreach(var l in lists) if(l!=null) pq.Enqueue(l,l.val);
    while(pq.Count>0){
        var n=pq.Dequeue(); cur.next=n; cur=n;
        if(n.next!=null) pq.Enqueue(n.next,n.next.val);
    }
    return dummy.next;
}
Python 
import heapq
def mergeKLists(lists):
    dummy = head = ListNode(0)
    heap = [(n.val, i, n) for i, n in enumerate(lists) if n]
    heapq.heapify(heap)
    while heap:
        val, i, node = heapq.heappop(heap)
        head.next = node; head = head.next
        if node.next: heapq.heappush(heap, (node.next.val, i, node.next))
    return dummy.next

def jump(nums):
    jumps = cur_end = cur_far = 0
    for i in range(len(nums)-1):

#16 Median of Two Sorted Arrays Hard LeetCode #4

Find the median of two sorted arrays in O(log(m+n)) time.

Time: O(log(min(m,n)))Space: O(1)Pattern: Binary search on partition
Binary search on the smaller array to find the correct partition point. At partition (i, j): maxLeft1 = minRight2 AND maxLeft2 = minRight1. This ensures exactly half the elements are on each side.
C# 
public double FindMedianSortedArrays(int[] a, int[] b) {
    if(a.Length>b.Length) return FindMedianSortedArrays(b,a);
    int m=a.Length, n=b.Length, lo=0, hi=m;
    while(lo<=hi){
        int i=(lo+hi)/2, j=(m+n+1)/2-i;
        int mL1=i>0?a[i-1]:int.MinValue, mR1=i<m?a[i]:int.MaxValue;
        int mL2=j>0?b[j-1]:int.MinValue, mR2=j<n?b[j]:int.MaxValue;
        if(mL1<=mR2&&mL2<=mR1){
            if((m+n)%2==1) return Math.Max(mL1,mL2);
            return (Math.Max(mL1,mL2)+(double)Math.Min(mR1,mR2))/2;
        } else if(mL1>mR2) hi=i-1; else lo=i+1;
    }
    return 0;
}

def findMedianSortedArrays(nums1, nums2):

#17 N-Queens Hard LeetCode #51

Place n queens on an n×n board so no two queens attack each other. Return all distinct solutions.

Time: O(n!)Space: O(n)Pattern: Backtracking with sets
Place queens row by row. Track occupied columns, diagonals (row-col), and anti-diagonals (row+col) in sets for O(1) conflict check. Backtrack if a column/diagonal is already occupied.
C# 
public IList<string[]> SolveNQueens(int n) {
    var res=new List<string[]>(); var board=new int[n];
    var cols=new HashSet<int>(), d1=new HashSet<int>(), d2=new HashSet<int>();
    void Bt(int row){
        if(row==n){res.Add(board.Select(c=>new string('.',c)+"Q"+new string('.',n-c-1)).ToArray());return;}
        for(int c=0;c<n;c++){
            if(cols.Contains(c)||d1.Contains(row-c)||d2.Contains(row+c)) continue;
            cols.Add(c);d1.Add(row-c);d2.Add(row+c);board[row]=c;
            Bt(row+1);
            cols.Remove(c);d1.Remove(row-c);d2.Remove(row+c);
        }
    }
    Bt(0); return res;
}
Python 
def solveNQueens(n):

#18 Find Median from Data Stream Hard LeetCode #295

Design a structure that supports adding numbers and finding the median in O(log n) / O(1).

Add: O(log n)FindMedian: O(1)Pattern: Two heaps
Use two heaps: max-heap for lower half, min-heap for upper half. Keep sizes balanced (differ by at most 1). Median = top of larger heap (odd total) or average of both tops (even total).
C# 
class MedianFinder {
    PriorityQueue<int,int> _small=new(), _large=new();
    public void AddNum(int num){
        _small.Enqueue(num,-num);
        _small.TryDequeue(out int top,out _); _large.Enqueue(top,top);
        if(_small.Count<_large.Count){_large.TryDequeue(out top,out _);_small.Enqueue(top,-top);}
    }
    public double FindMedian(){
        _small.TryPeek(out int s,out _); _large.TryPeek(out int l,out _);
        return _small.Count>_large.Count?s:(s+(double)l)/2;
    }
}
Python 
import heapq
class MedianFinder:
    def __init__(self): self.small = []; self.large = []  # max-heap, min-heap
    def addNum(self, num):
        heapq.heappush(self.small, -num)
        heapq.heappush(self.large, -heapq.heappop(self.small))
        if len(self.small) < len(self.large):
            heapq.heappush(self.small, -heapq.heappop(self.large))
    def findMedian(self):
        if len(self.small) > len(self.large): return -self.small[0]
        return (-self.small[0] + self.large[0]) / 2
#19 Regular Expression Matching Hard LeetCode #10

public bool IsMatch(string s, string p) {

    bool[,] dp=new bool[m+1,n+1]; dp[0,0]=true;
    for(int j=2;j<=n;j++) if(p[j-1]=='*') dp[0,j]=dp[0,j-2];

def isMatch(s, p):
    m, n = len(s), len(p)
    dp = [[False]*(n+1) for _ in range(m+1)]
    dp[0][0] = True
    for j in range(2, n+1):

class Codec {

            var n=new TreeNode(int.Parse(v)); n.left=Build(); n.right=Build(); return n;
        }

    }
}
