#1 Invert Binary Tree Easy LeetCode #226

Invert a binary tree (swap left and right children at every node).

Time: O(n)Space: O(h)Pattern: DFS / BFS
Recursively swap left and right children. Base case: null → return null. Recurse on both subtrees first, then swap.
C# 
public TreeNode InvertTree(TreeNode root) {
    if (root == null) return null;
    (root.left, root.right) = (InvertTree(root.right), InvertTree(root.left));
    return root;
}
Python 
def invertTree(root):
    if not root: return None
    root.left, root.right = invertTree(root.right), invertTree(root.left)
    return root

Find the maximum depth (number of nodes along the longest root-to-leaf path).

depth(node) = 1 + max(depth(left), depth(right)). Base case: null → 0.
C# 
public int MaxDepth(TreeNode root) =>

def maxDepth(root):

#3 Diameter of Binary Tree Easy LeetCode #543

Find the length of the longest path between any two nodes (the path may not pass through the root).

Time: O(n)Space: O(h)Pattern: DFS with global state
At each node, the diameter passing through it = depth(left) + depth(right). Track global max. Return max depth (not diameter) up the recursion.
C# 
public int DiameterOfBinaryTree(TreeNode root) {
    int best=0;
    int Dfs(TreeNode n){
        if(n==null) return 0;
        int L=Dfs(n.left), R=Dfs(n.right);
        best=Math.Max(best,L+R);
        return 1+Math.Max(L,R);
    }
    Dfs(root); return best;
}
Python 
def diameterOfBinaryTree(root):
    best = [0]
    def dfs(node):
        if not node: return 0
        L, R = dfs(node.left), dfs(node.right)
        best[0] = max(best[0], L+R)
        return 1 + max(L, R)
    dfs(root); return best[0]
#4 Balanced Binary Tree Easy LeetCode #110

Determine if a binary tree is height-balanced (heights of subtrees differ by at most 1).

Time: O(n)Space: O(h)Pattern: DFS, early termination
Return -1 (sentinel) from DFS if subtree is unbalanced. Otherwise return height. Check both subtrees: if either returns -1, propagate -1 up. Otherwise check if |left - right| > 1.
C# 
public bool IsBalanced(TreeNode root) {
    int Dfs(TreeNode n){
        if(n==null) return 0;
        int L=Dfs(n.left), R=Dfs(n.right);
        if(L==-1||R==-1||Math.Abs(L-R)>1) return -1;
        return 1+Math.Max(L,R);
    }
    return Dfs(root)!=-1;
}
Python 
def isBalanced(root):
    def dfs(node):
        if not node: return 0
        L, R = dfs(node.left), dfs(node.right)
        if L == -1 or R == -1 or abs(L-R) > 1: return -1
        return 1 + max(L, R)
    return dfs(root) != -1
#5 Same Tree Easy LeetCode #100

Given the roots of two binary trees, check if they are the same.

Time: O(n)Space: O(h)Pattern: DFS
Two trees are the same iff their roots have the same value and their left subtrees are the same and their right subtrees are the same.
C# 
public bool IsSameTree(TreeNode p, TreeNode q) {
    if(p==null&&q==null) return true;
    if(p==null||q==null||p.val!=q.val) return false;
    return IsSameTree(p.left,q.left)&&IsSameTree(p.right,q.right);
}
Python 
def isSameTree(p, q):
    if not p and not q: return True
    if not p or not q or p.val != q.val: return False
    return isSameTree(p.left,q.left) and isSameTree(p.right,q.right)

#6 Lowest Common Ancestor of BST Medium LeetCode #235

Find the lowest common ancestor (LCA) of two nodes in a BST.

Time: O(h)Space: O(1)Pattern: BST property
BST: if both p and q are less than current node, go left. If both greater, go right. Otherwise, current is the LCA (they split here, or one of them equals current).
C# 
public TreeNode LowestCommonAncestor(TreeNode root, TreeNode p, TreeNode q) {
    while(root!=null){
        if(p.val<root.val&&q.val<root.val) root=root.left;
        else if(p.val>root.val&&q.val>root.val) root=root.right;
        else return root;
    }
    return null;
}
Python 
def lowestCommonAncestor(root, p, q):
    while root:
        if p.val < root.val and q.val < root.val: root = root.left
        elif p.val > root.val and q.val > root.val: root = root.right
        else: return root
#7 Binary Tree Level Order Traversal Medium LeetCode #102

Return the level-order traversal of a binary tree (left to right, level by level).

Time: O(n)Space: O(n)Pattern: BFS with queue
BFS with a queue. At the start of each level, snapshot the current queue size. Process exactly that many nodes, add their children to the queue. Collect each level's values as a group.
C# 
public IList<IList<int>> LevelOrder(TreeNode root) {
    var res=new List<IList<int>>();
    if(root==null) return res;
    var q=new Queue<TreeNode>(); q.Enqueue(root);
    while(q.Count>0){
        var level=new List<int>(); int sz=q.Count;
        for(int i=0;i<sz;i++){
            var n=q.Dequeue(); level.Add(n.val);
            if(n.left!=null)q.Enqueue(n.left);
            if(n.right!=null)q.Enqueue(n.right);
        }
        res.Add(level);
    }
    return res;
}
Python 
from collections import deque
def levelOrder(root):
    if not root: return []
    q = deque([root]); res = []
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft(); level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res
#8 Validate Binary Search Tree Medium LeetCode #98

Determine if a binary tree is a valid BST.

Time: O(n)Space: O(h)Pattern: DFS with bounds
Pass min and max bounds down the recursion. Every node must satisfy min < node.val < max. Going left: update max to node.val. Going right: update min to node.val.
C# 
public bool IsValidBST(TreeNode root) {
    bool Valid(TreeNode n, long lo, long hi){
        if(n==null) return true;
        if(n.val<=lo||n.val>=hi) return false;
        return Valid(n.left,lo,n.val)&&Valid(n.right,n.val,hi);
    }
    return Valid(root,long.MinValue,long.MaxValue);
}
Python 
def isValidBST(root):
    def valid(node, lo, hi):
        if not node: return True
        if not lo < node.val < hi: return False
        return valid(node.left, lo, node.val) and valid(node.right, node.val, hi)
    return valid(root, float('-inf'), float('inf'))
#9 Kth Smallest Element in BST Medium LeetCode #230

Find the kth smallest element in a BST.

Time: O(h+k)Space: O(h)Pattern: Inorder traversal
Inorder traversal of BST gives elements in sorted order. Use an iterative inorder with a counter — stop when you've visited k nodes.
C# 
public int KthSmallest(TreeNode root, int k) {
    var stack=new Stack<TreeNode>(); var node=root;
    while(stack.Count>0||node!=null){
        while(node!=null){stack.Push(node);node=node.left;}
        node=stack.Pop(); if(--k==0) return node.val;
        node=node.right;
    }
    return -1;
}

#10 Binary Tree Right Side View Medium LeetCode #199

Given a binary tree, return the values of nodes visible when the tree is viewed from the right side.

Time: O(n)Space: O(n)Pattern: BFS, last in level
BFS level order. For each level, the rightmost node is the visible one. Collect the last element of each BFS level.
C# 
public IList<int> RightSideView(TreeNode root) {
    var res=new List<int>(); if(root==null) return res;
    var q=new Queue<TreeNode>(); q.Enqueue(root);
    while(q.Count>0){
        int sz=q.Count;
        for(int i=0;i<sz;i++){
            var n=q.Dequeue();
            if(i==sz-1) res.Add(n.val);
            if(n.left!=null)q.Enqueue(n.left);
            if(n.right!=null)q.Enqueue(n.right);
        }
    }
    return res;
}
Python 
from collections import deque
def rightSideView(root):
    if not root: return []
    q = deque([root]); res = []
    while q:
        for i in range(len(q)):
            node = q.popleft()
            if i == len(q): res.append(node.val)  # last in level
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
    return res

#11 Number of Islands Medium LeetCode #200 · Classic BFS/DFS

Count the number of islands (groups of connected '1's) in a 2D grid.

Time: O(m·n)Space: O(m·n)Pattern: DFS flood fill
For each unvisited '1', increment count and flood-fill (DFS or BFS) to mark all connected '1's as visited (change to '0' or use a visited set).
C# 
public int NumIslands(char[][] grid) {
    int count=0, m=grid.Length, n=grid[0].Length;
    void Dfs(int r, int c){
        if(r<0||r>=m||c<0||c>=n||grid[r][c]!='1') return;
        grid[r][c]='0';
        Dfs(r+1,c);Dfs(r-1,c);Dfs(r,c+1);Dfs(r,c-1);
    }
    for(int r=0;r<m;r++)
        for(int c=0;c<n;c++)
            if(grid[r][c]=='1'){Dfs(r,c);count++;}
    return count;
}
Python 
def numIslands(grid):
    count = 0
    def dfs(r, c):
        if r<0 or r>=len(grid) or c<0 or c>=len(grid[0]) or grid[r][c]!='1': return
        grid[r][c] = '0'
        for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]: dfs(r+dr, c+dc)
    for r in range(len(grid)):
        for c in range(len(grid[0])):
            if grid[r][c]=='1': dfs(r,c); count+=1
    return count
#12 Clone Graph Medium LeetCode #133

Deep clone an undirected graph given a reference to one of its nodes.

Time: O(V+E)Space: O(V)Pattern: BFS/DFS + HashMap
Use a hashmap from original node to cloned node. BFS/DFS the graph. When visiting a neighbor, if not cloned yet, create clone. Always use the clone from the map to avoid duplicates.
C# 
public Node CloneGraph(Node node) {
    if(node==null) return null;
    var clones=new Dictionary<Node,Node>();
    Node Dfs(Node n){
        if(clones.TryGetValue(n,out var c)) return c;
        var clone=new Node(n.val); clones[n]=clone;
        foreach(var nb in n.neighbors) clone.neighbors.Add(Dfs(nb));
        return clone;
    }
    return Dfs(node);
}
Python 
def cloneGraph(node):
    if not node: return None
    clones = {}
    def dfs(n):
        if n in clones: return clones[n]
        clone = Node(n.val); clones[n] = clone
        for nb in n.neighbors: clone.neighbors.append(dfs(nb))
        return clone
    return dfs(node)
#13 Course Schedule (Cycle Detection) Medium LeetCode #207

There are n courses. Given prerequisites, determine if it's possible to finish all courses.

Time: O(V+E)Space: O(V+E)Pattern: Topological Sort / Cycle Detection
Build a directed graph. Check for cycles using DFS with 3-color marking: white (unvisited), gray (in current path), black (fully processed). If you reach a gray node, there's a cycle.
C# 
public bool CanFinish(int n, int[][] prereqs) {
    var adj=new List<int>[n];
    for(int i=0;i<n;i++) adj[i]=new();
    foreach(var p in prereqs) adj[p[1]].Add(p[0]);
    int[] state=new int[n]; // 0=white,1=gray,2=black
    bool Dfs(int v){
        if(state[v]==1) return false;
        if(state[v]==2) return true;
        state[v]=1;
        foreach(int nb in adj[v]) if(!Dfs(nb)) return false;
        state[v]=2; return true;
    }
    for(int v=0;v<n;v++) if(!Dfs(v)) return false;
    return true;
}
Python 
from collections import defaultdict
def canFinish(numCourses, prerequisites):
    adj = defaultdict(list)
    for a, b in prerequisites: adj[b].append(a)
    state = [0] * numCourses  # 0=white, 1=gray, 2=black
    def dfs(v):
        if state[v] == 1: return False  # cycle!
        if state[v] == 2: return True
        state[v] = 1
        if not all(dfs(nb) for nb in adj[v]): return False
        state[v] = 2; return True
    return all(dfs(v) for v in range(numCourses))
#14 Pacific Atlantic Water Flow Medium LeetCode #417

Find all cells where water can flow to both the Pacific and Atlantic oceans.

Time: O(m·n)Space: O(m·n)Pattern: Multi-source BFS from both oceans
Reverse the problem: instead of flowing from cell to ocean (downhill), BFS/DFS from ocean edges uphill. Cells reachable from both Pacific edges and Atlantic edges are the answer.
C# 
public IList<IList<int>> PacificAtlantic(int[][] heights) {
    int m=heights.Length, n=heights[0].Length;
    int[] dr={0,0,1,-1}, dc={1,-1,0,0};
    HashSet<(int,int)> Bfs(IEnumerable<(int,int)> seeds){
        var vis=new HashSet<(int,int)>(seeds);
        var q=new Queue<(int,int)>(seeds);
        while(q.Count>0){
            var(r,c)=q.Dequeue();
            for(int d=0;d<4;d++){
                int nr=r+dr[d],nc=c+dc[d];
                if(nr>=0&&nr<m&&nc>=0&&nc<n&&vis.Add((nr,nc))&&heights[nr][nc]>=heights[r][c])
                    q.Enqueue((nr,nc));
            }
        }
        return vis;
    }
    var pac=Bfs(Enumerable.Range(0,m).Select(r=>(r,0)).Concat(Enumerable.Range(0,n).Select(c=>(0,c))));
    var atl=Bfs(Enumerable.Range(0,m).Select(r=>(r,n-1)).Concat(Enumerable.Range(0,n).Select(c=>(m-1,c))));
    return pac.Intersect(atl).Select(p=>(IList<int>)new[]{p.Item1,p.Item2}).ToList();
}
Python 
from collections import deque
def pacificAtlantic(heights):
    m, n = len(heights), len(heights[0])
    def bfs(seeds):
        vis = set(seeds); q = deque(seeds)
        while q:
            r, c = q.popleft()
            for dr, dc in [(1,0),(-1,0),(0,1),(0,-1)]:
                nr, nc = r+dr, c+dc
                if 0<=nr<m and 0<=nc<n and (nr,nc) not in vis and heights[nr][nc]>=heights[r][c]:
                    vis.add((nr,nc)); q.append((nr,nc))
        return vis
    pac = bfs([(r,0) for r in range(m)] + [(0,c) for c in range(n)])
    atl = bfs([(r,n-1) for r in range(m)] + [(m-1,c) for c in range(n)])
    return [list(p) for p in pac & atl]
#15 Word Ladder Hard LeetCode #127

Find the minimum number of steps to transform beginWord to endWord, changing one letter at a time, where each intermediate word must exist in the word list.

Time: O(M²·N)Space: O(M²·N)Pattern: BFS shortest path
BFS from beginWord. Neighbors of a word = all words differing by exactly one letter. Use pattern hashing: "hot" → ["*ot", "h*t", "ho*"]. Group words by pattern for fast neighbor lookup.
C# 
public int LadderLength(string beginWord, string endWord, IList<string> wordList) {
    var wordSet=new HashSet<string>(wordList);
    if(!wordSet.Contains(endWord)) return 0;
    var adj=new Dictionary<string,List<string>>();
    foreach(var w in wordSet)
        for(int i=0;i<w.Length;i++){
            var p=w[..i]+"*"+w[(i+1)..];
            if(!adj.ContainsKey(p)) adj[p]=new();
            adj[p].Add(w);
        }
    var q=new Queue<(string,int)>(); q.Enqueue((beginWord,1));
    var vis=new HashSet<string>{beginWord};
    while(q.Count>0){
        var(word,steps)=q.Dequeue();
        for(int i=0;i<word.Length;i++){
            var pat=word[..i]+"*"+word[(i+1)..];
            foreach(var nb in adj.GetValueOrDefault(pat,new())) {
                if(nb==endWord) return steps+1;
                if(vis.Add(nb)) q.Enqueue((nb,steps+1));
            }
        }
    }
    return 0;
}
Python 
from collections import deque, defaultdict
def ladderLength(beginWord, endWord, wordList):
    wordSet = set(wordList)
    if endWord not in wordSet: return 0
    adj = defaultdict(list)
    for w in wordSet:
        for i in range(len(w)):
            adj[w[:i]+'*'+w[i+1:]].append(w)
    q = deque([(beginWord, 1)]); vis = {beginWord}
    while q:
        word, steps = q.popleft()
        for i in range(len(word)):
            for nb in adj[word[:i]+'*'+word[i+1:]]:
                if nb == endWord: return steps+1
                if nb not in vis: vis.add(nb); q.append((nb, steps+1))
    return 0
