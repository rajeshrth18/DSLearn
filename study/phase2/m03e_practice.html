#1Valid Parentheses
Easy Stack String

Given a string s containing only '(', ')', '{', '}', '[', ']', determine if the input string is valid. A string is valid if every open bracket is closed by the same type of bracket in the correct order.

Examples:
  • "()" → true
  • "()[]{}" → true
  • "(]" → false
  • "([)]" → false
Constraints: 1 = s.length = 104. Time: O(n). Space: O(n).
Approach: Use a stack. Push opening brackets. When a closing bracket appears, pop and check if it matches. Stack must be empty at the end.
Time: O(n) — one pass through string &nbsp;|&nbsp; Space: O(n) — stack stores up to n/2 open brackets
C# 
public bool IsValid(string s)
{
    Stack<char> stack = new();
    foreach (char c in s)
    {
        if (c is '(' or '[' or '{')
        {
            stack.Push(c);
        }
        else
        {
            if (stack.Count == 0) return false;
            char top = stack.Pop();
            if (c == ')' && top != '(') return false;
            if (c == ']' && top != '[') return false;
            if (c == '}' && top != '{') return false;
        }
    }
    return stack.Count == 0;
}
Python 
def is_valid(s: str) -> bool:
    match = {')': '(', ']': '[', '}': '{'}
    stack = []
    for c in s:
        if c in '([{':
            stack.append(c)
        elif not stack or stack[-1] != match[c]:
            return False
        else:
            stack.pop()
    return not stack
#2Min Stack
Easy Stack Design

Design a stack that supports Push, Pop, Top, and retrieving the minimum element in O(1). Implement the MinStack class with: Push(val), Pop(), Top() (returns top element), and GetMin() (returns minimum element).

Example: 
MinStack minStack = new MinStack();
minStack.Push(-2); minStack.Push(0); minStack.Push(-3);
minStack.GetMin(); // returns -3
minStack.Pop();
minStack.Top();    // returns 0
minStack.GetMin(); // returns -2
Approach: Maintain two stacks: the main stack and a "min stack". The min stack's top always holds the current minimum. Only push to min stack when the new value is = current minimum; only pop from min stack when the popped value equals the current minimum.
Time: O(1) — all operations (Push, Pop, Top, GetMin) &nbsp;|&nbsp; Space: O(n) — two stacks
C# 
public class MinStack
{
    private Stack<int> main = new();
    private Stack<int> mins = new();

    public void Push(int val)
    {
        main.Push(val);
        if (mins.Count == 0 || val <= mins.Peek())
            mins.Push(val);
    }

    public void Pop()
    {
        int val = main.Pop();
        if (val == mins.Peek()) mins.Pop();
    }

    public int Top()    => main.Peek();
    public int GetMin() => mins.Peek();
}
Python 
class MinStack:
    def __init__(self):
        self.stack     = []
        self.min_stack = []

    def push(self, val: int):
        self.stack.append(val)
        mn = val if not self.min_stack else min(val, self.min_stack[-1])
        self.min_stack.append(mn)

    def pop(self):
        self.stack.pop(); self.min_stack.pop()

    def top(self) -> int:    return self.stack[-1]
    def get_min(self) -> int: return self.min_stack[-1]
#3Evaluate Reverse Polish Notation
Easy Stack Math

Evaluate the value of an expression in Reverse Polish Notation (postfix). Valid operators are +, -, *, / (integer division, truncate toward zero). Each operand or operator is a separate token.

Examples:
  • ["2","1","+","3","*"] → 9 — ((2+1)*3)
  • ["4","13","5","/","+"] → 6 — (4+(13/5))
Approach: Scan tokens left to right. Push numbers. When an operator appears, pop two numbers (b = right operand, a = left), compute result, push back.
Time: O(n) — one pass through tokens &nbsp;|&nbsp; Space: O(n) — stack holds at most n/2 numbers
C# 
public int EvalRPN(string[] tokens)
{
    Stack<int> stack = new();
    foreach (string t in tokens)
    {
        if (int.TryParse(t, out int num))
        {
            stack.Push(num);
        }
        else
        {
            int b = stack.Pop(), a = stack.Pop();
            stack.Push(t switch
            {
                "+" => a + b,
                "-" => a - b,
                "*" => a * b,
                "/" => a / b,  // C# integer division truncates toward zero
                _   => throw new ArgumentException()
            });
        }
    }
    return stack.Pop();
}
Python 
def eval_rpn(tokens: list) -> int:
    stack = []
    for t in tokens:
        if t not in '+-*/':
            stack.append(int(t))
        else:
            b, a = stack.pop(), stack.pop()
            if   t == '+': stack.append(a + b)
            elif t == '-': stack.append(a - b)
            elif t == '*': stack.append(a * b)
            else:          stack.append(int(a / b))  # truncate toward zero
    return stack[0]
#4Implement Queue Using Two Stacks
Easy Stack Queue Design

Implement a first-in first-out (FIFO) queue using only two stacks. The queue should support Push (adds to back), Pop (removes from front), Peek (views front), and Empty. All operations should be amortized O(1).

Approach — "Lazy Transfer": Use an "in" stack for pushes and an "out" stack for pops/peeks. When the out stack is empty, transfer everything from in to out (this reverses order, giving FIFO). Each element is transferred at most once — amortized O(1).
Time: O(1) amortized — each element transferred at most once &nbsp;|&nbsp; Space: O(n) — two stacks total
C# 
public class MyQueue
{
    private Stack<int> inStack  = new();
    private Stack<int> outStack = new();

    public void Push(int x) => inStack.Push(x);

    public int Pop()
    {
        Transfer();
        return outStack.Pop();
    }

    public int Peek()
    {
        Transfer();
        return outStack.Peek();
    }

    public bool Empty => inStack.Count == 0 && outStack.Count == 0;

    private void Transfer()
    {
        if (outStack.Count == 0)
            while (inStack.Count > 0)
                outStack.Push(inStack.Pop());
    }
}
Python 
class MyQueue:
    def __init__(self):
        self.in_stack  = []
        self.out_stack = []

    def push(self, x: int):  self.in_stack.append(x)

    def _transfer(self):
        if not self.out_stack:
            while self.in_stack:
                self.out_stack.append(self.in_stack.pop())

    def pop(self) -> int:
        self._transfer(); return self.out_stack.pop()

    def peek(self) -> int:
        self._transfer(); return self.out_stack[-1]

    def empty(self) -> bool:
        return not self.in_stack and not self.out_stack
#5Daily Temperatures
Easy Monotonic Stack Array

Given an array temperatures, for each day return how many days you have to wait until a warmer temperature. If there is no future warmer day, return 0 for that day.

Example: [73,74,75,71,69,72,76,73] → [1,1,4,2,1,1,0,0]
Approach — Monotonic Decreasing Stack: Maintain a stack of indices with decreasing temperatures. When a warmer day is found, pop all cooler indices and record the distance.
Time: O(n) — each index pushed and popped at most once &nbsp;|&nbsp; Space: O(n) — stack stores indices
C# 
public int[] DailyTemperatures(int[] temps)
{
    int[] result = new int[temps.Length];
    Stack<int> stack = new();

    for (int i = 0; i < temps.Length; i++)
    {
        while (stack.Count > 0 && temps[stack.Peek()] < temps[i])
        {
            int j = stack.Pop();
            result[j] = i - j;
        }
        stack.Push(i);
    }
    return result;
}
Python 
def daily_temperatures(temps: list) -> list:
    result = [0] * len(temps)
    stack  = []   # indices, decreasing temps
    for i, t in enumerate(temps):
        while stack and temps[stack[-1]] < t:
            j = stack.pop()
            result[j] = i - j
        stack.append(i)
    return result

#6Next Greater Element II (Circular)
Medium Monotonic Stack Circular Array

Given a circular integer array nums, return the next greater number for every element. The next greater number is the first greater number to the right (wrapping around). Return -1 if none exists.

Example: [1,2,1] → [2,-1,2]
Approach — Double Pass: Loop from 0 to 2n-1 using index % n. Only push indices in the first pass (i < n). When popping, the current element is the next greater for the popped index.
Time: O(n) — double pass, each element processed at most twice &nbsp;|&nbsp; Space: O(n) — stack stores indices
C# 
public int[] NextGreaterElements(int[] nums)
{
    int n = nums.Length;
    int[] result = new int[n];
    Array.Fill(result, -1);
    Stack<int> stack = new();

    for (int i = 0; i < 2 * n; i++)
    {
        int idx = i % n;
        while (stack.Count > 0 && nums[stack.Peek()] < nums[idx])
            result[stack.Pop()] = nums[idx];
        if (i < n) stack.Push(idx);
    }
    return result;
}
Python 
def next_greater_elements(nums: list) -> list:
    n, result, stack = len(nums), [-1] * len(nums), []
    for i in range(2 * n):
        while stack and nums[stack[-1]] < nums[i % n]:
            result[stack.pop()] = nums[i % n]
        if i < n: stack.append(i)
    return result
#7Number of Visible People in a Queue
Medium Monotonic Stack Array

There are n people standing in a queue and their heights are given by array heights. For each person i, count how many people to the right they can see. Person i can see person j (j > i) if all people between them have heights strictly less than min(heights[i], heights[j]).

Example: [10,6,8,5,11,9] → [3,1,2,1,1,0]
Approach — Scan Right to Left: Process from right to left. Maintain a decreasing stack. For each person, count how many they can see (those popped because current is taller) + possibly 1 more (the one that stopped the popping).
Time: O(n) — right-to-left scan, each height pushed and popped once &nbsp;|&nbsp; Space: O(n) — decreasing stack
C# 
public int[] CanSeePersonsCount(int[] heights)
{
    int n = heights.Length;
    int[] result = new int[n];
    Stack<int> stack = new();  // Decreasing heights (values, not indices)

    for (int i = n - 1; i >= 0; i--)
    {
        int count = 0;
        // Count all shorter people visible before they're blocked
        while (stack.Count > 0 && stack.Peek() < heights[i])
        {
            stack.Pop();
            count++;
        }
        // If stack is not empty, the top (taller) person is also visible
        if (stack.Count > 0) count++;
        result[i] = count;
        stack.Push(heights[i]);
    }
    return result;
}
Python 
def can_see_persons_count(heights: list) -> list:
    n, result, stack = len(heights), [0] * len(heights), []
    for i in range(n - 1, -1, -1):   # right to left
        count = 0
        while stack and heights[stack[-1]] < heights[i]:
            stack.pop(); count += 1
        if stack: count += 1    # can see the blocking tall person
        result[i] = count; stack.append(i)
    return result
#8Decode String
Medium Stack String

Given an encoded string, return its decoded string. The encoding rule is k[encoded_string] — the encoded_string inside brackets is repeated exactly k times.

Examples:
  • "3[a]2[bc]" → "aaabcbc"
  • "3[a2[c]]" → "accaccacc"
  • "2[abc]3[cd]ef" → "abcabccdcdcdef"
Approach — Two Stacks: One stack for repeat counts, one for string contexts. When '[' is found, push current state and start fresh. When ']' is found, pop the count and context, and build the repeated string.
Time: O(n × k) — k = max repeat count, string built character by character &nbsp;|&nbsp; Space: O(n) — two stacks
C# 
public string DecodeString(string s)
{
    Stack<int>    countStack  = new();
    Stack<StringBuilder> strStack = new();
    StringBuilder current = new();
    int k = 0;

    foreach (char c in s)
    {
        if (char.IsDigit(c))
        {
            k = k * 10 + (c - '0');  // Handle multi-digit numbers
        }
        else if (c == '[')
        {
            countStack.Push(k);
            strStack.Push(current);
            current = new StringBuilder();
            k = 0;
        }
        else if (c == ']')
        {
            int repeat = countStack.Pop();
            StringBuilder prev = strStack.Pop();
            string seg = current.ToString();
            for (int i = 0; i < repeat; i++) prev.Append(seg);
            current = prev;
        }
        else
        {
            current.Append(c);
        }
    }
    return current.ToString();
}
Python 
def decode_string(s: str) -> str:
    count_stack, str_stack = [], []
    current, k = "", 0
    for c in s:
        if c.isdigit():
            k = k * 10 + int(c)
        elif c == '[':
            count_stack.append(k); str_stack.append(current)
            current = ""; k = 0
        elif c == ']':
            repeat = count_stack.pop()
            current = str_stack.pop() + current * repeat
        else:
            current += c
    return current
#9Asteroid Collision
Medium Stack Simulation

An array of integers represents asteroids. The absolute value is the asteroid's size; the sign indicates direction (positive = right, negative = left). Asteroids move at equal speed. If two asteroids meet, the smaller one explodes. Equal-size ones both explode. Moving in the same direction never collide. Return the state after all collisions.

Examples:
  • [5,10,-5] → [5,10] — -5 collides with 10 and explodes
  • [8,-8] → [] — both explode
  • [10,2,-5] → [10] — -5 destroys 2, then 10 destroys -5
Approach: Stack holds surviving asteroids. Only a collision occurs when stack top is positive AND current asteroid is negative. Handle: current destroyed (top larger), both destroyed (equal), or top destroyed (current larger — keep processing).
Time: O(n) — each asteroid pushed and popped at most once &nbsp;|&nbsp; Space: O(n) — stack for surviving asteroids
C# 
public int[] AsteroidCollision(int[] asteroids)
{
    Stack<int> stack = new();

    foreach (int ast in asteroids)
    {
        bool destroyed = false;

        while (stack.Count > 0 && stack.Peek() > 0 && ast < 0)
        {
            int top = stack.Peek();
            if (top < -ast)        { stack.Pop(); continue; }  // Top destroyed
            else if (top == -ast)  { stack.Pop(); }              // Both destroyed
            destroyed = true;  // Current asteroid destroyed (top >= -ast)
            break;
        }

        if (!destroyed) stack.Push(ast);
    }

    return stack.Reverse().ToArray();
}
Python 
def asteroid_collision(asteroids: list) -> list:
    stack = []
    for a in asteroids:
        alive = True
        while alive and a < 0 and stack and stack[-1] > 0:
            if   stack[-1] < -a: stack.pop()         # right wins
            elif stack[-1] == -a: stack.pop(); alive = False
            else: alive = False                       # left wins
        if alive: stack.append(a)
    return stack
#10Binary Tree Level Order Traversal (BFS)
Medium Queue BFS Tree

Given the root of a binary tree, return its level-order traversal as a list of lists of values (from left to right, level by level).

Example: Tree [3,9,20,null,null,15,7] → [[3],[9,20],[15,7]]
Approach — BFS with Queue: Enqueue root. Each iteration, process exactly queue.Count nodes (the whole current level), then enqueue their children. This naturally groups nodes by level.
Time: O(n) — visits every node exactly once &nbsp;|&nbsp; Space: O(w) — w = max width of tree (queue holds one full level)
C# 
public IList<IList<int>> LevelOrder(TreeNode? root)
{
    var result = new List<IList<int>>();
    if (root == null) return result;

    Queue<TreeNode> q = new();
    q.Enqueue(root);

    while (q.Count > 0)
    {
        int size = q.Count;
        var level = new List<int>();
        for (int i = 0; i < size; i++)
        {
            var node = q.Dequeue();
            level.Add(node.Val);
            if (node.Left  != null) q.Enqueue(node.Left);
            if (node.Right != null) q.Enqueue(node.Right);
        }
        result.Add(level);
    }
    return result;
}
Python 
from collections import deque

def level_order(root) -> list:
    if not root: return []
    result, q = [], deque([root])
    while q:
        level = []
        for _ in range(len(q)):
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        result.append(level)
    return result
#11Rotting Oranges
Medium Queue BFS Grid

You have a grid of cells: 0 = empty, 1 = fresh orange, 2 = rotten orange. Every minute, any fresh orange adjacent (4-directionally) to a rotten orange also becomes rotten. Return the minimum number of minutes until no fresh orange remains. Return -1 if impossible.

Example: [[2,1,1],[1,1,0],[0,1,1]] → 4
Approach — Multi-Source BFS: Seed the BFS queue with ALL initially rotten oranges. Each BFS "level" represents one minute of rotting. After BFS completes, check if any fresh oranges remain.
Time: O(m×n) — BFS visits each cell at most once &nbsp;|&nbsp; Space: O(m×n) — queue can hold entire grid
C# 
public int OrangesRotting(int[][] grid)
{
    int rows = grid.Length, cols = grid[0].Length;
    Queue<(int r, int c)> queue = new();
    int fresh = 0;

    // Seed all initially rotten oranges + count fresh
    for (int r = 0; r < rows; r++)
        for (int c = 0; c < cols; c++)
        {
            if (grid[r][c] == 2) queue.Enqueue((r, c));
            if (grid[r][c] == 1) fresh++;
        }

    if (fresh == 0) return 0;

    int[][] dirs = [[0,1],[0,-1],[1,0],[-1,0]];
    int minutes = 0;

    while (queue.Count > 0 && fresh > 0)
    {
        int size = queue.Count;
        minutes++;
        for (int i = 0; i < size; i++)
        {
            var (r, c) = queue.Dequeue();
            foreach (var d in dirs)
            {
                int nr = r + d[0], nc = c + d[1];
                if (nr < 0 || nr >= rows || nc < 0 || nc >= cols) continue;
                if (grid[nr][nc] != 1) continue;
                grid[nr][nc] = 2;
                fresh--;
                queue.Enqueue((nr, nc));
            }
        }
    }
    return fresh == 0 ? minutes : -1;
}
Python 
from collections import deque

def oranges_rotting(grid: list) -> int:
    rows, cols = len(grid), len(grid[0])
    q, fresh = deque(), 0
    for r in range(rows):
        for c in range(cols):
            if   grid[r][c] == 2: q.append((r, c))
            elif grid[r][c] == 1: fresh += 1
    minutes = 0
    while q and fresh:
        for _ in range(len(q)):
            r, c = q.popleft()
            for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < rows and 0 <= nc < cols and grid[nr][nc] == 1:
                    grid[nr][nc] = 2; q.append((nr, nc)); fresh -= 1
        minutes += 1
    return minutes if fresh == 0 else -1
#12Task Scheduler
Medium Priority Queue Greedy

Given tasks with a cooldown interval n, schedule them to minimize total execution time. Identical tasks must be at least n intervals apart. Return total intervals needed.

Example: tasks=["A","A","A","B","B","B"], n=2 → 8 (A B _ A B _ A B)
Approach — Greedy + Max-Heap + Queue: Always schedule the most frequent available task (max-heap). After scheduling, the task enters a cooldown queue until it becomes available again. Each tick either runs a task or idles.
Time: O(n log k) — n tasks, heap ops on k unique task types &nbsp;|&nbsp; Space: O(k) — heap and cooldown queue
C# 
public int LeastInterval(char[] tasks, int n)
{
    int[] freq = new int[26];
    foreach (char t in tasks) freq[t - 'A']++;

    var maxHeap = new PriorityQueue<int, int>(
        Comparer<int>.Create((a, b) => b - a));
    foreach (int f in freq) if (f > 0) maxHeap.Enqueue(f, f);

    var cooldown = new Queue<(int, int)>();  // (remaining, availableAt)
    int time = 0;

    while (maxHeap.Count > 0 || cooldown.Count > 0)
    {
        time++;
        if (cooldown.Count > 0 && cooldown.Peek().Item2 == time)
        {
            var (cnt, _) = cooldown.Dequeue();
            maxHeap.Enqueue(cnt, cnt);
        }
        if (maxHeap.Count > 0)
        {
            int rem = maxHeap.Dequeue() - 1;
            if (rem > 0) cooldown.Enqueue((rem, time + n + 1));
        }
    }
    return time;
}
Python 
import heapq
from collections import Counter, deque

def least_interval(tasks: list, n: int) -> int:
    freq     = Counter(tasks)
    max_heap = [-c for c in freq.values()]
    heapq.heapify(max_heap)
    q, time = deque(), 0
    while max_heap or q:
        time += 1
        if max_heap:
            cnt = heapq.heappop(max_heap) + 1
            if cnt: q.append((cnt, time + n))
        if q and q[0][1] == time:
            heapq.heappush(max_heap, q.popleft()[0])
    return time

#13Largest Rectangle in Histogram
Hard Monotonic Stack Array

Given an array of integers heights representing histogram bar heights (each of width 1), return the area of the largest rectangle in the histogram.

Example: [2,1,5,6,2,3] → 10 (bars at index 2 and 3, height=5 spanning width=2)
Approach — Monotonic Increasing Stack: For each bar, when we find a shorter bar, we pop — the popped bar defines the maximum height for a rectangle. The width is from the new top to the current index. A virtual bar of height 0 at the end flushes all remaining bars.
Time: O(n) — each bar pushed and popped at most once &nbsp;|&nbsp; Space: O(n) — stack stores indices
C# 
public int LargestRectangleArea(int[] heights)
{
    int n = heights.Length;
    int maxArea = 0;
    Stack<int> stack = new();

    for (int i = 0; i <= n; i++)
    {
        int h = (i == n) ? 0 : heights[i];
        while (stack.Count > 0 && heights[stack.Peek()] > h)
        {
            int height = heights[stack.Pop()];
            int left   = stack.Count == 0 ? -1 : stack.Peek();
            int width  = i - left - 1;
            maxArea = Math.Max(maxArea, height * width);
        }
        stack.Push(i);
    }
    return maxArea;
}
Python 
def largest_rectangle_area(heights: list) -> int:
    stack, result = [], 0
    for i in range(len(heights) + 1):
        h = heights[i] if i < len(heights) else 0
        while stack and heights[stack[-1]] > h:
            height = heights[stack.pop()]
            width  = i if not stack else i - stack[-1] - 1
            result = max(result, height * width)
        stack.append(i)
    return result
#14Trapping Rain Water
Hard Monotonic Stack Two Pointers Array

Given n non-negative integers representing an elevation map where each bar has width 1, compute how much water can be trapped after raining.

Example: [0,1,0,2,1,0,1,3,2,1,2,1] → 6
Two Approaches shown:
  • Stack approach — O(n) time, O(n) space. Natural extension of histogram pattern.
  • Two-pointer approach — O(n) time, O(1) space. More optimal for space.
C# 
// Approach 1: Monotonic Stack — O(n) time, O(n) space
public int TrapStack(int[] height)
{
    int water = 0;
    Stack<int> stack = new();

    for (int i = 0; i < height.Length; i++)
    {
        while (stack.Count > 0 && height[stack.Peek()] < height[i])
        {
            int bottom = stack.Pop();
            if (stack.Count == 0) break;
            int left = stack.Peek();
            int h = Math.Min(height[left], height[i]) - height[bottom];
            water += h * (i - left - 1);
        }
        stack.Push(i);
    }
    return water;
}

// Approach 2: Two Pointers — O(n) time, O(1) space
public int TrapTwoPointers(int[] height)
{
    int l = 0, r = height.Length - 1;
    int leftMax = 0, rightMax = 0, water = 0;

    while (l < r)
    {
        if (height[l] < height[r])
        {
            leftMax = Math.Max(leftMax, height[l]);
            water += leftMax - height[l];
            l++;
        }
        else
        {
            rightMax = Math.Max(rightMax, height[r]);
            water += rightMax - height[r];
            r--;
        }
    }
    return water;
}
// [0,1,0,2,1,0,1,3,2,1,2,1] ? 6
// [4,2,0,3,2,5]              ? 9

def trap_stack(height: list) -> int:
    stack, water = [], 0
    for i in range(len(height)):
        while stack and height[stack[-1]] < height[i]:
            bottom = height[stack.pop()]
            if not stack: break
            bounded = min(height[stack[-1]], height[i]) - bottom
            water  += bounded * (i - stack[-1] - 1)
        stack.append(i)
    return water
#15Sliding Window Maximum
Hard Monotonic Deque Sliding Window

Given an integer array nums and an integer k, find the maximum value in every contiguous subarray of size k. Return an array of these maxima.

Example: [1,3,-1,-3,5,3,6,7], k=3 → [3,3,5,5,6,7]
Approach — Monotonic Deque (Decreasing): The deque stores indices. Its front is always the index of the maximum in the current window. When adding a new element, remove from back all indices with smaller values (they can never be max while current element is in the window). Remove from front indices that fall outside the window.
Time: O(n) — each element added and removed from deque at most once &nbsp;|&nbsp; Space: O(k) — deque holds at most k indices
C# 
public int[] MaxSlidingWindow(int[] nums, int k)
{
    int n = nums.Length;
    int[] result = new int[n - k + 1];
    LinkedList<int> dq = new();  // Decreasing deque of indices

    for (int i = 0; i < n; i++)
    {
        // Evict indices outside the window
        while (dq.Count > 0 && dq.First!.Value < i - k + 1)
            dq.RemoveFirst();

        // Remove smaller values from back (useless)
        while (dq.Count > 0 && nums[dq.Last!.Value] < nums[i])
            dq.RemoveLast();

        dq.AddLast(i);

        // Record maximum once window is full
        if (i >= k - 1)
            result[i - k + 1] = nums[dq.First!.Value];
    }
    return result;
}
// [1,3,-1,-3,5,3,6,7], k=3 ? [3,3,5,5,6,7]
// [1], k=1                  ? [1]
// [9,11], k=2               ? [11]
Python 
from collections import deque

def max_sliding_window(nums: list, k: int) -> list:
    dq, result = deque(), []
    for i in range(len(nums)):
        while dq and nums[dq[-1]] <= nums[i]: dq.pop()
        dq.append(i)
        if dq[0] < i - k + 1: dq.popleft()
        if i >= k - 1: result.append(nums[dq[0]])
    return result
